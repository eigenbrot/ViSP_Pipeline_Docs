    """The main, runner script for the entire ViSP Pipeline.

    The role of this function is to read in the configuration file and make calls to the appropriate pipeline modules.
    Unlike calling the modules directly, this function forces all calibration products to exist, but it will allow
    for missing error frames.
    """
    t1 = time.time()
    config = ConfigParser()
    config.read(config_file)
    print('{}: Loaded configuration file {}'.format(tag(), config_file))

    # Check to make sure raw science exists
    if not os.path.exists(config['Main']['raw_sci_dir']):
        print('{}: Raw science directory {} does not exist!'.format(tag(), config['Main']['raw_sci_dir']))
        return 1

    # Check for necessary calibration data and try to generate them if needed
    sci_kwargs = generate_intermediate_objects(config)

    # Now generate the demodulation matrices, if needed
    ispol = get_pol_mode(config['Main']['raw_sci_dir'], config['Main']['data_set_ID'])
    if ispol:

        pol_rdx_dir = '{}/proc'.format(config['PolarimetricCalibration']['raw_pol_dir'])
        dmod1_name = '{}/DMOD_beam1.fits'.format(pol_rdx_dir)
        dmod2_name = '{}/DMOD_beam2.fits'.format(pol_rdx_dir)
        if os.path.exists(dmod1_name) and os.path.exists(dmod2_name):
            print('{}: Found previous demodulation matrices in {}'.format(tag(), pol_rdx_dir))
            dmod1 = Data.DemodulationData()
            dmod1.load_from_file(dmod1_name)
            print('{}: loaded {}'.format(tag(), dmod1_name))

            dmod2 = Data.DemodulationData()
            dmod2.load_from_file(dmod2_name)
            print('{}: loaded {}'.format(tag(), dmod2_name))

        else:
            pol_args = dict(config['PolarimetricCalibration'])
            convert_config_to_int(pol_args)

            if not os.path.exists('{}/rdx_complete'.format(pol_rdx_dir)):
                run_polarization_calibration(**pol_args)

            dmod1, dmod2 = run_PAC_demod(pol_rdx_dir, **pol_args)

        sci_kwargs['telescope_db'] = config['PolarimetricCalibration']['telescope_db']
        sci_kwargs['demodulation'] = ScienceCalibration.get_demodulation_matrix(dmod1,dmod2)

    print('{}: Running science data'.format(tag()))
    print('{}: Reading files in {}'.format(tag(), config['Main']['raw_sci_dir']))
    all_Data = Data.DataSetData(config['Main']['raw_sci_dir'], dataset_ID=config['Main']['data_set_ID'])

    start_science_multiproc(config, sci_kwargs, all_Data)

    print('{}: All done in {} s'.format(tag(), time.time() - t1))

    return
    


    [docs]def start_science_multiproc(config, sci_kwargs, datasetData):
    """Manage a set of Science Calibration processes using multiprocessing.

    Each slit position (i.e., separate file) is assigned its own process and then the processes are started in such
    a way that there are never more than the user-defined maximum running at a single time.

    Parameters
    ----------
    config : configparser.ConfigParser
        The object containing user-defined reduction options

    sci_kwargs : dict
        The Science Calibration keyword arguments that contain the intermediate calibration objects and any other
        reduction options

    datasetData : `Data.DataSetData`
        Object containing file list and metadata for all files corresponding to the data set to be reduced

    Returns
    -------
    None
    """

    numthread = int(config['Main']['threads'])
    print('{}: Input directory contains {} slit positions'.format(tag(), datasetData.num_pos))
    slit_pos_list = datasetData.spatial_positions

    try:
        sci_kwargs['remove_tele'] = config.getboolean('Main','remove_tele_pol')
    except configparser.NoOptionError:
        pass

    # Let's see if the user wants to reduce only a subset of images
    try:
        sp_str = config['Main']['slit_pos']

        # They can provide a slice/range with 'XXX:YYY'
        slice_match = re.match('([0-9]*):([0-9]*)', sp_str)
        if slice_match:
            pos_subset = list(range(int(slice_match.groups()[0]), int(slice_match.groups()[1])))

        # If it's not a slice we'll try using ast.literal_eval to grab either an int or a list
        else:
            pos_subset = ast.literal_eval(sp_str)
            if type(pos_subset) is int:
                pos_subset = [pos_subset]
            elif type(pos_subset) is not list:
                raise ValueError('slit_pos must be either an int or list of ints')

        # Make sure that the subset actually exists in the full list of slit positions
        if all([i in slit_pos_list for i in pos_subset]):
            print('{}:  only reducing slit positions {}'.format(tag(), pos_subset))
            slit_pos_list = pos_subset
        else:
            raise ValueError('slit position {} is not in input directory'.format(pos_subset))

    except KeyError:
        print('{}:  running ScienceCalibration on all positions with {} threads'.format(tag(), numthread))
        pass

    # This loop sets up a list of mp processes. They have not been started.
    jobs = []
    sci_kwargs['dark_lock'] = mp.Lock()
    sci_kwargs['gain_lock'] = mp.Lock()
    sci_kwargs['geo_lock'] = mp.Lock()
    for spos in slit_pos_list:
        outpre = '{:}.{:04}'.format(config['Main']['output_prefix'], spos)
        jobs.append(mp.Process(target=ScienceCalibration.main,
                               args=[datasetData, spos, outpre],
                               kwargs=sci_kwargs,
                               name='POS_{:04n}'.format(spos)))

    # Now, start jobs and keep starting jobs until the max number of threads is reached
    #  Make a check every second to see if any jobs have finished and thus created room for new ones to start.
    i = 0
    while i < len(jobs):
        while [x.is_alive() for x in jobs].count(True) < numthread:
            jobs[i].start()
            i += 1
            if i >= len(jobs):
                break
        time.sleep(1)

    # Finally, make sure we wait for all jobs to complete before moving on.
    for p in jobs:
        p.join()

    return
    


    [docs]def generate_intermediate_objects(config):
    """Generate/collect the intermediate Calibration Objects needed by the Science Calibration module.

    If the an individual object already exists on disk, simply read it. Otherwise, call that object's generating module.

    Parameters
    ----------
    config : configparser.ConfigParser
        The object containing user-defined reduction options

    Returns
    -------
    dict
        The Science Calibration keyword arguments that contain the intermediate calibration objects.
    """

    mkerr = config.getboolean('Main', 'gen_noise_frames')

    # Dark
    ######
    filename = config['Main']['dark_cal']
    if not os.path.exists(filename):
        print('{}: Could not find {}, attempting to generate with DarkCalibration'.format(tag(), filename))
        DarkCal, DarkErr = DarkCalibration.main(config['DarkCalibration']['raw_dark_dir'], strip_fits(filename),
                                                gen_err=mkerr)
    else:
        DarkCal = Data.FitsData(filename)
        print('{}: Loaded {}'.format(tag(), filename))

        errname = gen_errname(filename)
        try:
            DarkErr = Data.FitsData(errname)
            print('{}:  and uncertainty {}'.format(tag(), errname))
        except FileNotFoundError:
            if mkerr:
                raise FileNotFoundError("gen_noise_frames is True, but I can't find an uncertainty frame "
                                        "associated with {}. You probably need to re-generate all Cal objects".format(filename))
            print('{}: Could not find {}. Setting error to None'.format(tag(), errname))
            DarkErr = None

    # Lamp Gain
    ###########
    filename = config['Main']['lamp_gain_cal']
    if not os.path.exists(filename):
        print('{}: Could not find {}, attempting to generate with LampGainCalibration'.format(tag(), filename))
        LampCal, LampErr = LampGainCalibration.main(config['LampGainCalibration']['raw_lamp_gain_dir'],
                                                    strip_fits(filename),
                                                    dark_cal=DarkCal, dark_cal_err=DarkErr, gen_err=mkerr)
    else:
        LampCal = Data.FitsData(filename)
        print('{}: Loaded {}'.format(tag(), filename))
        errname = gen_errname(filename)
        try:
            LampErr = Data.FitsData(errname)
            print('{}:  and uncertainty {}'.format(tag(), errname))
        except FileNotFoundError:
            if mkerr:
                raise FileNotFoundError("gen_noise_frames is True, but I can't find an uncertainty frame "
                                        "associated with {}. You probably need to re-generate all Cal objects".format(filename))
            print('{}: Could not find {}. Setting error to None'.format(tag(), errname))
            LampErr = None

    # Solar Gain / GeoCal
    #####################
    filename = config['Main']['solar_gain_cal']
    geoname = config['Main']['geometric_cal']
    if not os.path.exists(filename) or not os.path.exists(geoname):
        print('{}: Could not find {} or {}, attempting to generate with SolarGainCalibration'.format(tag(), filename,
                                                                                                     geoname))
        SolarCal, SolarErr, GeoCal = SolarGainCalibration.main(config['SolarGainCalibration']['raw_solar_gain_dir'],
                                                               strip_fits(filename), geoname, dark_cal=DarkCal,
                                                               lamp_gain_cal=LampCal, dark_cal_err=DarkErr,
                                                               lamp_cal_err=LampErr, gen_err=mkerr)
    else:
        GeoCal = Data.FitsData(geoname)
        print('{}: Loaded {}'.format(tag(), geoname))
        SolarCal = Data.FitsData(filename)
        print('{}: Loaded {}'.format(tag(), filename))
        errname = gen_errname(filename)
        try:
            SolarErr = Data.FitsData(errname)
            print('{}:  and uncertainty {}'.format(tag(), errname))
        except FileNotFoundError:
            if mkerr:
                raise FileNotFoundError("gen_noise_frames is True, but I can't find an uncertainty frame "
                                        "associated with {}. You probably need to re-generate all Cal objects".format(filename))
            print('{}: Could not find {}. Setting error to None'.format(tag(), errname))
            SolarErr = None

    # Construct the argument lists for the main pipeline
    sci_kwargs = {'dark_cal': DarkCal, 'solar_gain_cal': SolarCal, 'geometric_cal': GeoCal,
                  'dark_cal_err': DarkErr, 'solar_gain_cal_err': SolarErr, 'gen_err': mkerr}

    return sci_kwargs
    


    [docs]def get_pol_mode(data_dir, dataset_ID, suffix='.fits'):
    """Determine if the Science data corresponds to polarimetric observations or just basic intensity.

    Parameters
    ----------
    data_dir : str
        Name of input directory

    dataset_ID : str
        The ID_004 header value of data to use

    Returns
    -------
    bool
        True if the data are polarimetric. False otherwise.

    """
    file_list = glob('{}/*{}'.format(data_dir, suffix))
    pol = False
    found = False
    for f in file_list:
        hdulist = pyfits.open(f)
        header = hdulist[0].header
        try:
            if header['VISP_PMD'] == 'Full Stokes' and (header['ID___004'] == dataset_ID or dataset_ID == ''):
                print('{}: Polarimetric data detected'.format(tag()))
                pol = True
                found = True
                break
            elif header[0]['VISP_PMD'] == 'Stokes-I' and (header['ID___004'] == dataset_ID or dataset_ID == ''):
                print('{}: Basic intensity data detected'.format(tag()))
                found = True
                break
        except KeyError:
            pass

    if not found:
        print('{}: Could not determine polarimetric mode, assuming basic intensity'.format(tag()))

    return pol
    


    [docs]def run_polarization_calibration(raw_pol_dir, raw_dark_dir, raw_lamp_gain_dir, raw_solar_gain_dir,
                                 num_bins_x=1, num_bins_y=1, dark_cal='DarkCal.fits', lamp_gain_cal='LampGainCal.fits',
                                 solar_gain_cal='SolarGainCal.fits', geometric_cal='GeoCal.fits', threads=1,
                                 gen_noise_frames=False,
                                 **unused_kwargs):
    """
    Generate a set of reduced, macropixel'd ViSP data for use by the PACPipeline.

    This function basically constructs the correct config and kwarg objects and then does all of the actual processing
    via calls to `generate_intermediate_objects` and `ViSPPolCalPipeline.start_polcal_multiproc`.

    Parameters
    ----------
    raw_pol_dir : str
        Location of directory that contains raw Calibration Sequence (CS) data

    raw_dark_dir : str
        Location of the directory containing raw dark frames associated with the CS

    raw_lamp_gain_dir : str
        Location of the directory containing raw lamp gain frames associated with the CS

    raw_solar_gain_dir : str
        Location of the directory containing raw solar gain frames associated with the CS

    num_bins_x : int
        The final number of macropixels to extract/average from the processed CS data, spectral dimension.

    num_bins_y : int
        The final number of macropixels to extract/average from the processed CS data, spatial dimension.

    dark_cal : str
        Location of the processed DarkCalibration object to either load or save

    lamp_gain_cal : str
        Location of the processed LampGainCalibration object to either load or save

    solar_gain_cal : str
        Location of the processed SolarGainCalibration object to either load or save

    geometric_cal : str
        Location of the processed GeometricCalibration object to either load or save

    threads : int
        Maximum number of cpus/threads to use during reduction

    gen_noise_frames : bool
        If True then uncertainty estimates will be generated for the reduced CS step data

    unused_kwargs : dict
        Keyword arguments that exist in the config file, but are not used by this function

    Returns
    -------
    str
        The location of the directory associated with the reduction of the CS
    """
    outdir = '{}/proc'.format(raw_pol_dir)
    os.makedirs(outdir, exist_ok=True)

    # Make a dummy ConfigParser object with a Pol-only 'Main' section
    darkcal_file = '{}/{}'.format(outdir, dark_cal)
    lampcal_file = '{}/{}'.format(outdir, lamp_gain_cal)
    solarcal_file = '{}/{}'.format(outdir, solar_gain_cal)
    geocal_file = '{}/{}'.format(outdir, geometric_cal)
    run_args = {'Main': {'raw_pol_dir': raw_pol_dir,
                         'output_dir': outdir,
                         'dark_cal': darkcal_file,
                         'lamp_gain_cal': lampcal_file,
                         'solar_gain_cal': solarcal_file,
                         'geometric_cal': geocal_file,
                         'gen_noise_frames': gen_noise_frames,
                         'threads': threads},
                'DarkCalibration': {'raw_dark': raw_dark_dir},
                'LampGainCalibration': {'raw_lamp_gain': raw_lamp_gain_dir},
                'SolarGainCalibration': {'raw_solar_gain': raw_solar_gain_dir}}

    config = ConfigParser()
    config.read_dict(run_args)

    # Generate/load the necessary intermediate Calibration objects
    pol_kwargs = generate_intermediate_objects(config)
    pol_kwargs.update({'numpix_x': num_bins_x,
                       'numpix_y': num_bins_y})

    # Run the CS through the PolCal reduction pipeline
    ViSPPolCalPipeline.start_polcal_multiproc(config, pol_kwargs)

    # Create a file that indicates all files were successfully reduced
    with open('{}/rdx_complete'.format(outdir), 'w') as f:
        f.write('ViSP PolCal reduction of Calibration sequence in {}\nsaved to {}/beam[12] on {}'.\
                format(raw_pol_dir, outdir, time.asctime()))

    return
    


    [docs]def run_PAC_demod(run_dir, telescope_db='telescope_db.txt', threads=1, overwrite=False, **unused_kwargs):
    """Take a processed set of Calibration Sequence data and compute a set of demodulation matrices.

    The best demodulation is found by varying the parameters of the Calibration Unit (CU). See the PACPipeline for more
    details.

    Parameters
    ----------
    run_dir : str
        Location of the processed, macropixel'd ViSP data associated with a single Calibration Sequence (CS).

    telescope_db : str
        Location of the telescope parameter database

    threads : int
        Maximum number of cpus to use when fitting CU parameters. No more than 2 cpus will ever be used.

    overwrite : bool
        If False then demodulation matrices will not be computed if they already exist.

    unused_kwargs : dict
        Keyword arguments that exist in the config file, but are not used by this function

    Returns
    -------
    Data.FitsData
        The demodulation matrices for beam 1

    Data.FitsData
        The demodulation matrices for beam 2
    """
    from PAC_Pipeline.PACPipeline import main_demodulation_matrix

    jobs = []

    pac_kwargs = {'overwrite': overwrite,
                  'save_stats': True,
                  'telescope_db': telescope_db}
    for i in range(1,3):
        beam_dir = '{}/beam{}'.format(run_dir, i)
        beam_output = '{}/DMOD_beam{}.fits'.format(run_dir, i)
        jobs.append(mp.Process(target=main_demodulation_matrix,
                               args=[beam_dir, beam_output],
                               kwargs=pac_kwargs,
                               name='Beam{}'.format(i)))

    i = 0
    while i < len(jobs):
        while [x.is_alive() for x in jobs].count(True) < threads:
            jobs[i].start()
            i += 1
            if i >= len(jobs):
                break
        time.sleep(1)

    for p in jobs:
        p.join()

    dmod1 = Data.DemodulationData()
    dmod2 = Data.DemodulationData()
    dmod1.load_from_file('{}/DMOD_beam1.fits'.format(run_dir))
    dmod2.load_from_file('{}/DMOD_beam2.fits'.format(run_dir))

    return dmod1, dmod2

    [docs]def gen_blank_config(filename):
    """Create a blank configuration file.

    This is useful as a way to see all of the available options.

    Parameters
    ----------
    filename : str
        The location to save the blank file. Existing files will be overwritten.

    Returns
    -------
    None
    """
    config = ConfigParser(allow_no_value=True)
    config['Main'] = {'raw_sci_dir': '',
                      'data_set_ID': '',
                      'output_prefix': '',
                      'dark_cal': '',
                      'lamp_gain_cal': '',
                      'solar_gain_cal': '',
                      'geometric_cal': '',
                      'gen_noise_frames': '',
                      'threads': '',
                      '': None,
                      "; The raw files point to the location of the raw ViSP frames": None}
    config['DarkCalibration'] = {'raw_dark_dir': ''}
    config['LampGainCalibration'] = {'raw_lamp_gain_dir': ''}
    config['SolarGainCalibration'] = {'raw_solar_gain_dir': ''}
    config['PolarimetricCalibration'] = {'raw_pol_dir': '',
                                         'raw_dark_dir': '',
                                         'raw_lamp_gain_dir': '',
                                         'raw_solar_gain_dir': '',
                                         'gen_noise_frames': '',
                                         'telescope_db': '',
                                         'num_bins_x': '',
                                         'num_bins_y': '',
                                         'threads': ''}

    with open(filename, 'w') as f:
        config.write(f)

    return
    

def command_line():
    import argparse
    import sys

    parser = argparse.ArgumentParser(description='Run the ViSP Pipeline using the parameters specified in a config file. Config files follow the standard INI format')
    parser.add_argument('config_file', help="INI-style file containing pipeline options. Default is 'config.ini'.",
                        nargs='?', default='config.ini')
    parser.add_argument('-b', '--blank', action='store_true', help='Generate a blank configuration file')
    args = parser.parse_args()

    if args.blank:
        gen_blank_config(args.config_file)
        sys.exit(0)

    if os.path.exists(args.config_file):
        sys.exit(main(args.config_file))
    elif len(sys.argv) < 2:
        parser.print_usage()
        sys.exit(0)
    else:
        print('Could not load config file {}. Aborting.'.format(args.config_file))
        sys.exit(1)
