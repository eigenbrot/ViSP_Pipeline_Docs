    [docs]def main(config_file):
    """The main, runner script for the entire ViSP Pipeline.

    The role of this function is to read in the configuration file and make calls to the appropriate pipeline modules.
    Unlike calling the modules directly, this function forces all calibration products to exist, but it will allow
    for missing error frames.
    """
    t1 = time.time()
    config = ConfigParser()
    config.read(config_file)
    print('{}: Loaded configuration file {}'.format(tag(), config_file))

    if not check_files(config):
        return 1

    # Decide where to save any FitsData tmp files. This is here so that we never try to write anything to the directory
    # from which the code is run with explicate orders. On the DC there is no write access on the process machines.
    try:
        tmpdir = config['Main']['tmp_dir']
    except KeyError:
        tmpdir = os.path.dirname(config['Main']['output_prefix']) or '.'
    Data.TMPDIR = tmpdir
    os.makedirs(Data.TMPDIR, exist_ok=True)

    # Set a global suffix, if needed
    try:
        suffix = config['Main']['suffix']
        CONSTANTS['Data']['suffix'] = suffix
    except KeyError:
        pass

    # Check for necessary calibration data and try to generate them if needed
    sci_kwargs = generate_intermediate_objects(config)

    # Now generate the demodulation matrices, if needed
    ispol = get_pol_mode(config['Main']['raw_sci_dir'], config['Main']['data_set_ID'])
    save_M12 = False
    if ispol:
        pol_kwargs = generate_polarimetric_objects(config)
        save_M12 = pol_kwargs['save_M12_map']

        # Merge the two dictionaries together.
        sci_kwargs.update(pol_kwargs)

    print('{}: Running science data'.format(tag()))
    print('{}: Reading files in {}'.format(tag(), config['Main']['raw_sci_dir']))
    all_Data = Data.DataSetData(config['Main']['raw_sci_dir'], dataset_ID=config['Main']['data_set_ID'])

    # If the output prefix specifies a directory we'll create that now
    output_dir = os.path.dirname(config['Main']['output_prefix'])
    if output_dir:
        os.makedirs(output_dir, exist_ok=True)

    print('{}: Spawning ScienceCalibration'.format(tag()))
    start_science_multiproc(config, sci_kwargs, all_Data)

    if ispol and save_M12:
        from ViSP_Pipeline import PolarimetricUtils
        print('{}: Extracting full spatial map for PA&C'.format(tag()))
        savedir = os.path.dirname(config['Main']['output_prefix']) or '.'
        M12dir = savedir + '/M12_map'
        PolarimetricUtils.extract_M12_map(M12dir)

    print('{}: All done in {} s'.format(tag(), time.time() - t1))

    return
    


    [docs]def check_files(config):
    """Check that all the files/dirs needed for reduction at least exist.

    This makes sure we don't get an error caused by a missing file 30 minutes into reduction.

    Files are only checked for existance; the actual content is not verified.

    Parameters
    ----------
    config : configparser.ConfigParser
        The object containing user-defined reduction options

    Returns
    -------
    bool
        True if all files exist. False otherwise.
    """
    # Check to make sure raw science and calibration exists
    if not os.path.exists(config['Main']['raw_sci_dir']):
        print('{}: Raw science directory {} does not exist!'.format(tag(), config['Main']['raw_sci_dir']))
        return False

    if not os.path.exists(config['DarkCalibration']['raw_dark_dir']) and not os.path.exists(config['Main']['dark_cal']):
        print('{}: Raw dark directory {} does not exist!'.\
              format(tag(), config['DarkCalibration']['raw_dark_dir']))
        return False

    if not os.path.exists(config['LampGainCalibration']['raw_lamp_gain_dir']) and not os.path.exists(config['Main']['lamp_gain_cal']):
        print('{}: Raw lamp gain directory {} does not exist!'.\
              format(tag(), config['LampGainCalibration']['raw_lamp_gain_dir']))
        return False

    if not os.path.exists(config['SolarGainCalibration']['raw_solar_gain_dir']) and (not os.path.exists(config['Main']['solar_gain_cal']) or not os.path.exists(config['Main']['geometric_cal'])):
        print('{}: Raw solar gain directory {} does not exist!'.\
              format(tag(), config['SolarGainCalibration']['raw_solar_gain_dir']))
        return False

    # If polarized data is detected, check polcal
    ispol = get_pol_mode(config['Main']['raw_sci_dir'], config['Main']['data_set_ID'])
    if ispol:
        if not os.path.exists(config['Main']['instrument_pol_cal']):
            if not os.path.exists(config['InstrumentPolarimetricCalibration']['raw_pol_dir']):
                print('{}: Raw PolCal data {} does not exist!'. \
                format(tag(), config['InstrumentPolarimetricCalibration']['raw_pol_dir']))
                return False

    return True
    


    [docs]def start_science_multiproc(config, sci_kwargs, datasetData):
    """Manage a set of Science Calibration processes using multiprocessing.

    Each slit position (i.e., separate file) is assigned its own process and then the processes are started in such
    a way that there are never more than the user-defined maximum running at a single time.

    Parameters
    ----------
    config : configparser.ConfigParser
        The object containing user-defined reduction options

    sci_kwargs : dict
        The Science Calibration keyword arguments that contain the intermediate calibration objects and any other
        reduction options

    datasetData : `Data.DataSetData`
        Object containing file list and metadata for all files corresponding to the data set to be reduced

    Returns
    -------
    None
    """

    print('{}: Science kwargs: {}'.format(tag(), sci_kwargs))

    numthread = int(config['Main']['threads'])
    print('{}: Input directory contains {} slit positions'.format(tag(), datasetData.num_pos))
    slit_pos_list = datasetData.spatial_positions

    try:
        sci_kwargs['remove_tele'] = config.getboolean('Main', 'remove_tele_pol')
    except configparser.NoOptionError:
        pass

    try:
        sci_kwargs['include_parallactic'] = config.getboolean('Main','include_parallactic')
    except configparser.NoOptionError:
        pass

    # Let's see if the user wants to reduce only a subset of images
    try:
        sp_str = config['Main']['slit_pos']

        # They can provide a slice/range with 'XXX:YYY'
        slice_match = re.match('([0-9]*):([0-9]*)', sp_str)
        if slice_match:
            pos_subset = list(range(int(slice_match.groups()[0]), int(slice_match.groups()[1])))

        # If it's not a slice we'll try using ast.literal_eval to grab either an int or a list
        else:
            pos_subset = ast.literal_eval(sp_str)
            if type(pos_subset) is int:
                pos_subset = [pos_subset]
            elif type(pos_subset) is not list:
                raise ValueError('slit_pos must be either an int or list of ints')

        # Make sure that the subset actually exists in the full list of slit positions
        if all([i in slit_pos_list for i in pos_subset]):
            print('{}:  only reducing slit positions {}'.format(tag(), pos_subset))
            slit_pos_list = pos_subset
        else:
            raise ValueError('slit position {} is not in input directory'.format(pos_subset))

    except KeyError:
        print('{}:  running ScienceCalibration on all positions with {} threads'.format(tag(), numthread))
        pass

    # This loop sets up a list of mp processes. They have not been started.
    jobs = []
    sci_kwargs['dark_lock'] = mp.Lock()
    sci_kwargs['gain_lock'] = mp.Lock()
    sci_kwargs['geo_lock'] = mp.Lock()
    sci_kwargs['pol_lock'] = mp.Lock()
    for spos in slit_pos_list:
        outpre = '{:}.{:04}'.format(config['Main']['output_prefix'], spos)
        jobs.append(mp.Process(target=ScienceCalibration.main,
                               args=[datasetData, spos, outpre],
                               kwargs=sci_kwargs,
                               name='POS_{:04n}'.format(spos)))

    # Now, start jobs and keep starting jobs until the max number of threads is reached
    #  Make a check every second to see if any jobs have finished and thus created room for new ones to start.
    i = 0
    while i < len(jobs):
        while [x.is_alive() for x in jobs].count(True) < numthread:
            jobs[i].start()
            i += 1
            if i >= len(jobs):
                break
        time.sleep(1)

    # Finally, make sure we wait for all jobs to complete before moving on.
    for p in jobs:
        p.join()

    return
    



    [docs]def generate_intermediate_objects(config):
    """Generate/collect the intermediate Calibration Objects needed by the Science Calibration module.

    If the an individual object already exists on disk, simply read it. Otherwise, call that object's generating module.

    Parameters
    ----------
    config : configparser.ConfigParser
        The object containing user-defined reduction options

    Returns
    -------
    dict
        The Science Calibration keyword arguments that contain the intermediate calibration objects.
    """

    mkerr = config.getboolean('Main', 'gen_noise_frames')

    # Dark
    ######
    filename = config['Main']['dark_cal']
    no_err_msg = "gen_noise_frames is True, but I can't find an uncertainty frame " \
                 "associated with {}. You probably need to re-generate all Cal objects"

    if not os.path.exists(ensure_fits(filename)):
        print('{}: Could not find {}, attempting to generate with DarkCalibration'.format(tag(), filename))
        DarkCal, DarkErr = DarkCalibration.main(config['DarkCalibration']['raw_dark_dir'], strip_fits(filename),
                                                gen_err=mkerr)
    else:
        DarkCal = Data.FitsData(ensure_fits(filename))
        print('{}: Loaded {}'.format(tag(), ensure_fits(filename)))

        errname = gen_errname(filename)
        try:
            DarkErr = Data.FitsData(errname)
            print('{}:  and uncertainty {}'.format(tag(), errname))
        except FileNotFoundError:
            if mkerr:
                raise FileNotFoundError(no_err_msg.format(filename))
            print('{}: Could not find {}. Setting error to None'.format(tag(), errname))
            DarkErr = None

    # Lamp Gain
    ###########
    filename = config['Main']['lamp_gain_cal']
    if not os.path.exists(ensure_fits(filename)):
        print('{}: Could not find {}, attempting to generate with LampGainCalibration'.format(tag(), filename))
        LampCal, LampErr = LampGainCalibration.main(config['LampGainCalibration']['raw_lamp_gain_dir'],
                                                    strip_fits(filename),
                                                    dark_cal=DarkCal, dark_cal_err=DarkErr, gen_err=mkerr)
    else:
        LampCal = Data.FitsData(ensure_fits(filename))
        print('{}: Loaded {}'.format(tag(), ensure_fits(filename)))
        errname = gen_errname(filename)
        try:
            LampErr = Data.FitsData(errname)
            print('{}:  and uncertainty {}'.format(tag(), errname))
        except FileNotFoundError:
            if mkerr:
                raise FileNotFoundError(no_err_msg.format(filename))
            print('{}: Could not find {}. Setting error to None'.format(tag(), errname))
            LampErr = None

    # Solar Gain / GeoCal
    #####################
    filename = config['Main']['solar_gain_cal']
    geoname = config['Main']['geometric_cal']
    if not os.path.exists(ensure_fits(filename)) or not os.path.exists(ensure_fits(geoname)):
        print('{}: Could not find {} and/or {}, attempting to generate with SolarGainCalibration'.format(tag(),
                                                                                                         filename,
                                                                                                         geoname))
        SolarCal, SolarErr, GeoCal = SolarGainCalibration.main(config['SolarGainCalibration']['raw_solar_gain_dir'],
                                                               strip_fits(filename), ensure_fits(geoname),
                                                               dark_cal=DarkCal, lamp_gain_cal=LampCal,
                                                               dark_cal_err=DarkErr, lamp_cal_err=LampErr,
                                                               gen_err=mkerr)
    else:
        GeoCal = Data.FitsData(ensure_fits(geoname))
        print('{}: Loaded {}'.format(tag(), ensure_fits(geoname)))
        SolarCal = Data.FitsData(ensure_fits(filename))
        print('{}: Loaded {}'.format(tag(), ensure_fits(filename)))
        errname = gen_errname(filename)
        try:
            SolarErr = Data.FitsData(errname)
            print('{}:  and uncertainty {}'.format(tag(), errname))
        except FileNotFoundError:
            if mkerr:
                raise FileNotFoundError(no_err_msg.format(filename))
            print('{}: Could not find {}. Setting error to None'.format(tag(), errname))
            SolarErr = None

    # Construct the argument lists for the main pipeline
    sci_kwargs = {'dark_cal': DarkCal, 'solar_gain_cal': SolarCal, 'geometric_cal': GeoCal,
                  'dark_cal_err': DarkErr, 'solar_gain_cal_err': SolarErr, 'gen_err': mkerr}

    return sci_kwargs
    


    [docs]def generate_polarimetric_objects(config):
    """Generate/load InstrumentPolCal objects and update the main kwarg dict with relevant parameters

    The is basically the polarization analog of generate_intermediate_objects

    Parameters
    ----------
    config : configparser.ConfigParser
        The object containing user-defined reduction options. In this case it should have an
        "InstrumentPolarizationCalibration" section

    Returns
    -------
    dict
        Dictionary containing kwargs to Science Calibration that specify polarimetric options and Calibration products
    """
    from ViSP_Pipeline import InstrumentPolCalibration

    pol_kwargs = {'pol': True}

    inst_pol_name = config['Main']['instrument_pol_cal']
    if not os.path.exists(ensure_fits(inst_pol_name)):
        print('{}: Could not find {}, attempting to generate with InstrumentPolCalibration'.format(tag(),
                                                                                                   inst_pol_name))
        inst_pol_args = dict(config['InstrumentPolarimetricCalibration'])
        convert_config_to_int(inst_pol_args)
        inst_pol_args['output'] = ensure_fits(inst_pol_name)
        InstPolCal = InstrumentPolCalibration.main(**inst_pol_args)
    else:
        print('{}: Loaded {}'.format(tag(), ensure_fits(inst_pol_name)))
        InstPolCal = Data.DemodulationData()
        InstPolCal.load_from_file(ensure_fits(inst_pol_name))

    pol_kwargs['demodulation'] = InstPolCal

    try:
        cont_lims = [int(l) for l in config['Main']['continuum_lims'].split(':')]
    except KeyError:
        raise ValueError('continuum_lims option is either missing or malformed')
    else:
        pol_kwargs['continuum_lims'] = cont_lims

    try:
        save_M12 = config.getboolean('Main', 'save_M12_map')
    except configparser.NoOptionError:
        save_M12 = False

    pol_kwargs['save_M12_map'] = save_M12
    if save_M12:
        try:
            pol_kwargs['M12_line_lims'] = [int(l) for l in config['Main']['M12_line_lims'].split(':')]
        except Exception:
            raise ValueError('M12_line_lims options is either missing or malformed')

    return pol_kwargs
    


    [docs]def get_pol_mode(data_dir, dataset_ID, suffix=''):
    """Determine if the Science data corresponds to polarimetric observations or just basic intensity.

    Parameters
    ----------
    data_dir : str
        Name of input directory

    dataset_ID : str
        The ID_004 header value of data to use

    suffix : str
        File extension of the data files in data_dir

    Returns
    -------
    bool
        True if the data are polarimetric. False otherwise.

    """
    suffix = suffix or CONSTANTS['Data']['suffix']
    file_list = glob('{}/*{}'.format(data_dir, suffix))
    pol = False
    found = False
    for f in file_list:
        hdulist = pyfits.open(f)
        header = hdulist[0].header
        try:
            if header[S122['PolarimeterMode']] == 'Full Stokes' and \
                    (header[S122['DataSetParametersID']] == dataset_ID or dataset_ID == ''):
                print('{}: Polarimetric data detected'.format(tag()))
                pol = True
                found = True
                break
            elif header[S122['PolarimeterMode']] == 'Stokes-I' and \
                    (header[S122['DataSetParametersID']] == dataset_ID or dataset_ID == ''):
                print('{}: Basic intensity data detected'.format(tag()))
                found = True
                break
        except KeyError:
            pass

    if not found:
        print('{}: Could not determine polarimetric mode, assuming basic intensity'.format(tag()))

    return pol

    """Make sure that a string has only one '.fits' on the end of it
    """

    return strip_fits(filename) + '.' + CONSTANTS['Data']['suffix']
    


    [docs]def gen_errname(filename):
    """Add a '_err' suffix to a filename, before the fits extension
    """

    return '{}_err.{}'.format(strip_fits(filename), CONSTANTS['Data']['suffix'])
    



    [docs]def convert_config_to_int(configdict):
    """Convert configparser-style dictionary of arguments to integers, where appropriate
    """

    for k in configdict.keys():
        try:
            configdict[k] = int(configdict[k])
        except ValueError:
            pass

def command_line():
    import argparse
    import sys

    parser = argparse.ArgumentParser(
            description='Run the ViSP Pipeline using the parameters specified in a config file. Config files follow '
                        'the '
                        'standard INI format')
    parser.add_argument('config_file', help="INI-style file containing pipeline options. Default is 'config.ini'.",
                        nargs='?', default='config.ini')
    parser.add_argument('-b', '--blank', action='store_true', help='Generate a blank configuration file')
    parser.add_argument('-d', '--default', help='Generate a config file with sensible defaults for specified directory',
                        nargs=1, default=[None])
    args = parser.parse_args()

    if args.blank:
        gen_blank_config(args.config_file)
        sys.exit(0)

    if args.default != [None]:
        gen_sensible_config(args.config_file, args.default[0])
        sys.exit(0)

    if os.path.exists(args.config_file):
        sys.exit(main(args.config_file))
    elif len(sys.argv) < 2:
        parser.print_usage()
        sys.exit(0)
    else:
        print('Could not load config file {}. Aborting.'.format(args.config_file))
        sys.exit(1)
