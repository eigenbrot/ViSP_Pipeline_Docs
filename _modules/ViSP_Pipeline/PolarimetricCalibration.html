    [docs]def main(pol_dir, CS_step, outpre, dark_cal=None, solar_gain_cal=None, geometric_cal=None, numpix_x=1, numpix_y=1,
         lock=None):
    """Take in raw data and produce data suitable for use by the PA&C Pipeline.

    Dark, gain, and geometric effects are removed and the two beams are split into two separate objects/files. An
    arbitrary amount of binning of the final data is also possible. Binning the data results in faster PA&C analysis,
    at the expense of spatial/spectral resolution in the derived Calibration Unit parameters.

    Parameters
    ----------
    pol_dir : str
        Path to the directory containing a single SoCC

    CS_step : int
        Calibration Sequence step number to use for reduction

    outpre : str
        Prefix of the location to save processed data. Can include the full path. The processed data will be saved as
        outpre_beam1.fits and outpre_beam2.fits, and associated uncertainties (if provided) will be saved as
        outpre_beam1_err.fits and outpre_beam2_err.fits.

    dark_cal : Data.FitsData
        A Dark Calibration object. Probably the result of DarkCalibration.main().

    solar_gain_cal : Data.FitsData
        A Solar Gain Calibration object. Probably the result of SolarGainCalibration.main().

    geometric_cal : Data.FitsData
        A Geometric Calibration object. Probably produced by SolarGainCalibration.main().

    numpix_x : int
        The number of spatial pixels PER BEAM in the final output data

    numpix_y : int
        The number of spectral pixels PER BEAM in the final output data

    lock : multiprocessing.synchronize.locl
        If  provided, this lock will be used when reading information from the dark and gain calibration objects. This
        avoids some weirdness in astropy's FITS containers when accessing the same data at the same time.

    Returns
    -------
    Data.FitsData
        PolCal ViSP data for beam 1

    Data.FitsData
        PolCal ViSP data for beam 2

    Data.FitsData
        If uncertainty on the input data is provided, the uncertainty associated with beam 1 PolCal data

    Data.FitsData
        If uncertainty on the input data is provided, the uncertainty associated with beam 2 PolCal data
    """
    polData = Data.CalibrationCurveData(pol_dir, CS_step, suffix='VISP*.fits')
    print('{}: Loaded CS step {} from {}'.format(tag(), CS_step, pol_dir))
    polErr = None

    if polData.headers[0]['VISP_PMD'] != 'Full Stokes':
        raise Exception('Data are not from a full stokes run')

    if dark_cal is not None:
        print('{}: doing Dark Correction'.format(tag()))
        if lock is not None:
            print('{}: acquiring MultiProcessing Lock'.format(tag()))
            lock.acquire()
        generic.dark_correction(polData, dark_cal)
        if lock is not None:
            print('{}: releasing MultiProcessing Lock'.format(tag()))
            lock.release()

    if solar_gain_cal is not None:
        print('{}: doing Gain Correction with Solar Flat'.format(tag()))
        if lock is not None:
            print('{}: acquiring MultiProcessing Lock'.format(tag()))
            lock.acquire()
        generic.gain_correction(polData, solar_gain_cal)
        if lock is not None:
            print('{}: releasing MultiProcessing Lock'.format(tag()))
            lock.release()

    if geometric_cal is not None:
        print('{}: removing geometric/spectral distortions'.format(tag()))
        polData, polErr = ScienceCalibration.geometric_correction(polData, geometric_cal)

    print('{}: extracting macropixels'.format(tag()))
    binData_1, binData_2, binErr_1, binErr_2 = extract_macropixels(polData, numpix_x, numpix_y, errData=polErr)

    print('{}: generating PolCal data'.format(tag()))
    binData_1, binData_2, binErr_1, binErr_2 = generate_polarization_calibration(binData_1, binData_2, outpre,
                                                                                 CS_step, binErr_1, binErr_2)

    return binData_1, binData_2, binErr_1, binErr_2
    


    [docs]def generate_polarization_calibration(binData_1, binData_2, outpre, CS_step, err_1=None, err_2=None):
    """Write reduced PolCal to a sensible place.

    The sensible place is DIR/beam[1|2]/VISP_CS_STEP.PAC.beam[1|2].fits where DIR and STEP are defined by inputs.

    Parameters
    ----------
    binData_1 : Data.FitsData
        PolCal ViSP data for beam 1. Each modulator state in a separate ImageHDU.

    binData_2 : Data.FitsData
        PolCal ViSP data for beam 2. Each modulator state in a separate ImageHDU.

    outpre : str
        Path to top level directory that contains the reduction products

    CS_step : int
        The number of the CS step represented by the input data

    err_1 : Data.FitsData
        Uncertainty associated with binData1

    err_2 : Data.FitsData
        Uncertainty associated with binData2

    Returns
    -------
    Data.FitsData
        Passthrough of binData1

    Data.FitsData
        Passthrough of binData2

    Data.FitsData
        Passthrough of err_1

    Data.FitsData
        Passthrough of err_2

    """
    os.makedirs('{}/beam1'.format(outpre), exist_ok=True)
    os.makedirs('{}/beam2'.format(outpre), exist_ok=True)
    for obj, post, beamnum in zip([binData_1, binData_2, err_1, err_2],
                                  ['.fits', '.fits', '_err.fits', '_err.fits'],
                                  [1, 2, 1, 2]):
        if obj is not None:
            outname = '{0}/beam{1}/VISP_CS_{2:02}.PAC.beam{1}{3}'.format(outpre, beamnum, CS_step, post)
            obj.write_out(outname, overwrite=True)

    return binData_1, binData_2, err_1, err_2
    


    [docs]def extract_macropixels(fitsData, numpix_x, numpix_y, errData=None):
    """Bin the input data down to a specified number of pixels.

    Because this function is designed to produce PolCal data the two ViSP beams are split apart and placed into separate
    Data.FitsData objects.

    If the number of output pixels does not divide evenly into the number of input pixels then some input pixels will be
    lost. This might or might not be a bad thing.

    Parameters
    ----------
    fitsData : Data.FitsData
        Object containing data to bin. Should contain both ViSP beams and is probably the output of
        ScienceCalibration.geometric_correction()

    numpix_x : int
        The number of spatial output pixels. Both beams will have the same number of spatial pixels.

    numpix_y : int
        The number of spectral output pixels PER BEAM

    errData : Data.FitsData
        Uncertainty associated with the input data

    Returns
    -------
    Data.FitsData
        PolCal ViSP data for beam 1. Each modulator state in a separate ImageHDU.

    Data.FitsData
        PolCal ViSP data for beam 2. Each modulator state in a separate ImageHDU.

    Data.FitsData
        If uncertainty on the input data is provided, the uncertainty associated with beam 1 PolCal data

    Data.FitsData
        If uncertainty on the input data is provided, the uncertainty associated with beam 2 PolCal data
    """
    print('{}: Binning beams to shape ({}, {})'.format(tag(), numpix_x, numpix_y))

    pivot = fitsData.headers[0]['VISP_PPP']
    nummod = fitsData.headers[0]['VISP_XXX']
    numx, numy_all = fitsData.data_hdus[0].data.shape
    numy_2 = (numy_all - pivot)

    print('{}: full beam 1 shape: ({}, {})'.format(tag(), numx, pivot))
    print('{}: full beam 2 shape: ({}, {})'.format(tag(), numx, numy_2))

    if numpix_x > numx or numpix_y > pivot or numpix_y > numy_2:
        raise IndexError('Cannot bin to more pixels than input data')

    xbin = numx // numpix_x
    ybin_1 = pivot // numpix_y
    ybin_2 = numy_2 // numpix_y

    print('{}: x binning: {}'.format(tag(), xbin))
    print('{}: beam 1 y binning: {}'.format(tag(), ybin_1))
    print('{}: beam 2 y binning: {}'.format(tag(), ybin_2))

    xend = numpix_x * xbin - numx or None
    yend_1 = numpix_y * ybin_1 - pivot or None
    yend_2 = numpix_y * ybin_2 - numy_2 or None

    if (numx / numpix_x) % 1 != 0:
        print('{}: Spatial binning amount is non-integer; {} pixels will be lost from beams 1 and 2'.\
                      format(tag(), abs(xend)))

    if (pivot / numpix_y) % 1 !=0 or (numy_2 / numpix_y) % 1 != 0:
        print('{}: Spectral binning amount is non-integer; {}, {} pixels will be lost from beams 1 and 2'.\
                      format(tag(), abs(yend_1), abs(yend_2)))

    hdulist_1 = [pyfits.PrimaryHDU(header=fitsData.hdulist[0].header)]
    hdulist_1[0].header['CAM_010'] = xbin
    hdulist_1[0].header['CAM_011'] = ybin_1
    hdulist_1[0].header['VISP_OGX'] = numx
    hdulist_1[0].header['VISP_OGY'] = pivot
    hdulist_1[0].header['VISP_BEM'] = 1

    hdulist_2 = [pyfits.PrimaryHDU(header=fitsData.hdulist[0].header)]
    hdulist_2[0].header['CAM_010'] = xbin
    hdulist_2[0].header['CAM_011'] = ybin_2
    hdulist_2[0].header['VISP_OGX'] = numx
    hdulist_2[0].header['VISP_OGY'] = numy_2
    hdulist_2[0].header['VISP_BEM'] = 2

    for i in range(nummod):
        print('{}: modulator state {}:'.format(tag(), i))
        idx = fitsData.find_data_keyword_idx('VISP_YYY', i)
        for j in idx:
            print('{}:  extracting from data HDU {}'.format(tag(),j))
            data = fitsData.data_hdus[j].data
            tmp1 = data[:, :pivot]
            tmp2 = data[:, pivot:]

            x_binned_data1 = np.zeros((numpix_x, pivot))
            binned_data1 = np.zeros((numpix_x, numpix_y))
            x_binned_data2 = np.zeros((numpix_x, numy_all - pivot))
            binned_data2 = np.zeros((numpix_x, numpix_y))

            for k in range(xbin):
                x_binned_data1 += tmp1[k:xend:xbin, :]
                x_binned_data2 += tmp2[k:xend:xbin, :]
            for l in range(ybin_1):
                binned_data1 += x_binned_data1[:, l:yend_1:ybin_1]
            for l in range(ybin_2):
                binned_data2 += x_binned_data2[:, l:yend_2:ybin_2]

            ih_1 = pyfits.ImageHDU(np.asarray(binned_data1 / (xbin * ybin_1), dtype=np.float32),
                                   header=fitsData.data_hdus[j].header)
            ih_1.header['CAM_010'] = xbin
            ih_1.header['CAM_011'] = ybin_1
            ih_1.header['VISP_OGX'] = numx
            ih_1.header['VISP_OGY'] = pivot
            ih_1.header['VISP_BEM'] = 1
            hdulist_1.append(ih_1)

            ih_2 = pyfits.ImageHDU(np.asarray(binned_data2 / (xbin * ybin_2), dtype=np.float32),
                                   header=fitsData.data_hdus[j].header)
            ih_2.header['CAM_010'] = xbin
            ih_2.header['CAM_011'] = ybin_2
            ih_2.header['VISP_OGX'] = numx
            ih_2.header['VISP_OGY'] = numy_2
            ih_2.header['VISP_BEM'] = 2
            hdulist_2.append(ih_2)

    binData_1 = Data.FitsData()
    binData_1.origin = 'extract macropixels'
    binData_1.hdulist = hdulist_1
    binData_2 = Data.FitsData()
    binData_2.origin = 'extract macropixels'
    binData_2.hdulist = hdulist_2

    return binData_1, binData_2, None, None
    


    [docs]def upsample_demodulation(beam1_dmod, beam2_dmod):
    """Take a pair of demodulation matrices computed for binned data and up-sample them to the full ViSP frame.

    Up-sampling is done using nearest-neighbor interpolation in such a way to ensure that each macropixel (i.e. bin)
    is equally represented in the final output matrices. The exception to this is when the number of macropixels was
    not an integer divisor of the full frame. In this case the macropixel on the edge of each beam will be slightly
    (typically less than 5 px) over represented in the final matrices.

    All information on the shape of the full frame is pulled from the FITS headers, so they'd better be correct! Note
    that beam 1 and beam 2 do NOT need to have the same spectral size. In other words the split between the two beams
    is not required to be exactly in the middle of the chip.

    The output contains both beams stitched together so that it may be used directly by
    ScienceCalibration.polarimetric_demodulation.

    Parameters
    ----------
    beam1_dmod : Data.DemodulationData
        Demodulation matrix for beam 1. This object should contain a single ImageHDU (i.e., data hdu) with shape
        (x, y1, 4, M)

    beam2_dmod : Data.DemodulationData
        Demodulation matrix for beam 1. This object should contain a single ImageHDU (i.e., data hdu) with shape
        (x, y2, 4, M)

    Returns
    -------
    Data.DemodulationData
        Object containing array of shape (X, Y, 4, M) where X and Y are the dimensions of the orginal ViSP frame.

    Notes
    -----
    This function will work exactly as expected if x = y1 = y2 = 1, but in this case you're probably better off just
    squeezing out the extra dimensions and taking advantage of the (mostly) shape-agnostic input of
    ScienceCalibration.polarimetric_demodulation
    """
    if beam1_dmod.headers[0]['VISP_OGX'] != beam2_dmod.headers[0]['VISP_OGX']:
        raise ValueError('Beams 1 and 2 not not have the same original spatial dimension')

    full_x = beam1_dmod.headers[0]['VISP_OGX']
    ogy_1 = beam1_dmod.headers[0]['VISP_OGY']
    ogy_2 = beam2_dmod.headers[0]['VISP_OGY']
    full_y= ogy_1 + ogy_2
    nums, numm = beam1_dmod.data_hdus[0].data.shape[-2:]
    full_output = np.zeros((full_x, full_y, nums, numm), dtype=np.float32)
    print('{}: original shape of beam 1: ({}, {}, {}, {})'.format(tag(), full_x, ogy_1, nums, numm))
    print('{}: original shape of beam 2: ({}, {}, {}, {})'.format(tag(), full_x, ogy_2, nums, numm))
    print('{}: full frame output: ({}, {}, {}, {})'.format(tag(), full_x, full_y, nums, numm))

    startidx = 0
    endidx = ogy_1
    for beam_hdu in [beam1_dmod.data_hdus[0], beam2_dmod.data_hdus[0]]:

        print('{}: filling output spectral pixels {} - {}'.format(tag(), startidx, endidx))
        newy = beam_hdu.header['VISP_OGY']
        data = beam_hdu.data

        numx, numy, nums, numm = data.shape
        print('{}:  demodulation matrix shape: ({}, {}, {}, {})'.format(tag(), numx, numy, nums, numm))

        x = np.arange(full_x // numx / 2, full_x, full_x // numx)
        y = np.arange(newy // numy / 2, newy, newy // numy)

        if numx == 1 and numy == 1:
            print('{}:  full-average demodulation matrices detected'.format(tag()))
        else:
            print('{}:  interpolating'.format(tag()))

        xi, yi = np.mgrid[0:full_x:full_x * 1j, 0:newy:newy * 1j]
        for i in range(nums):
            print('{}:  stokes parameter {}'.format(tag(), i))
            for j in range(numm):
                print('{}:   modulator state {}'.format(tag(), j))
                if numx == 1 and numy == 1:
                    full_output[:, startidx:endidx, i, j] = data[0, 0, i, j]
                else:
                    full_output[:, startidx:endidx, i, j] = spi.interpn((x, y), data[:,:,i,j], (xi, yi), bounds_error=False,
                                                                        fill_value=None, method='nearest')

        startidx = newy
        endidx = None

    newhead = beam1_dmod.headers[0]
    newhead.remove('VISP_BEM')
    newhead['CAM_010'] = 1
    newhead['CAM_011'] = 1
    dmodData = Data.DemodulationData(full_output, header=newhead)

    return dmodData
