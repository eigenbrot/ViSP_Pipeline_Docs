    cu_numbins_y: int = 1,
    demod_numbins_x: int = 1,
    demod_numbins_y: int = 1,
) -> Data.DemodulationData:
    """Run raw data (i.e., a PolCal CS observation) through the Instrument Polarization Calibration pipeline

    The steps in the pipeline are:

    1. Perform ViSP reductions on entire CS data set

    2. Fit CU parameters and generate demodulation matrices from the resulting SoCCs

    So basically, this pipeline calls the normal ViSP pipeline and then uses PA&C Modules to further analyze the
    reduced PolCal data.

    Parameters
    ----------
    raw_pol_dir : str
        Path to directory containing raw PolCal data

    output : str
        Path were final InstrumentPolCalibration (i.e., demodulation matrices) will be saved

    dark_cal : `Data.FitsData`
        Dark Calibration object

    solar_gain_cal : `Data.FitsData`
        Solar Gain Calibration object

    geometric_cal : `Data.FitsData`
        Geometric Calibration object

    dark_cal_err : `Data.FitsData`
        Uncertainties on ``dark_cal``

    solar_gain_cal_err : `Data.FitsData`
        Uncertainties on ``solar_gain_cal``

    gen_err : bool
        If True then uncertainty estimates will be generated for all Calibration and data products

    telescope_db : str
        Path to the database of telescope properties (x, \tau). If None then the PA&C default will be used.

    fit_mode : str
        Name of the set of fitting switches to use

    init_set : str
        Name of the set of initial values for the CU parameters to use

    cu_numbins_x : int
        Each beam will have this number of spatial bins when computing the best CU parameters

    cu_numbins_y : int
        Each beam will have this number of spectral bins when computing the best CU parameters

    demod_numbins_x : int
        Each beam will have this number of spatial bins in the final demodulation matrices

    demod_numbins_y : int
        Each beam will have this number of spectral bins in the final demodulation matrices

    threads : int
        The number of independent processes to use when performing ViSP reductions on the PolCal data

    Returns
    -------
    Data.DemodulationData
    """
    pol_rdx_dir = "{}/polcal_proc".format(Data.TMPDIR)
    dmod1_name = "{}/DMOD_beam1.{}".format(pol_rdx_dir, CONSTANTS["Data"]["suffix"])
    dmod2_name = "{}/DMOD_beam2.{}".format(pol_rdx_dir, CONSTANTS["Data"]["suffix"])
    cmp1_name = "{}/CUpars_beam1.{}".format(pol_rdx_dir, CONSTANTS["Data"]["suffix"])
    cmp2_name = "{}/CUpars_beam2.{}".format(pol_rdx_dir, CONSTANTS["Data"]["suffix"])
    cmp1_final_name = "{}/CUpars_resamp_beam1.{}".format(pol_rdx_dir, CONSTANTS["Data"]["suffix"])
    cmp2_final_name = "{}/CUpars_resamp_beam2.{}".format(pol_rdx_dir, CONSTANTS["Data"]["suffix"])

    if os.path.exists(dmod1_name) and os.path.exists(dmod2_name):
        print("{}: Found previous demodulation matrices in {}".format(tag(), pol_rdx_dir))
        # TODO: move actual reading after if/else and remove redundant load from run_PAC_demod
        raw_dmod = Data.DemodulationData()
        raw_dmod.load_beam_from_file(dmod1_name, 1)
        print("{}: loaded {}".format(tag(), dmod1_name))

        raw_dmod.load_beam_from_file(dmod2_name, 2)
        print("{}: loaded {}".format(tag(), dmod2_name))

    else:
        if telescope_db is None:
            from PAC_Pipeline.generic import get_default_telescope_db

            print("{}: Loading telescope parameter database from PA&C defualts".format(tag()))
            telescope_db = get_default_telescope_db()

        if not os.path.exists("{}/rdx_complete".format(pol_rdx_dir)):
            pol_args = {
                "dark_cal": dark_cal,
                "solar_gain_cal": solar_gain_cal,
                "geometric_cal": geometric_cal,
                "dark_cal_err": dark_cal_err,
                "solar_gain_cal_err": solar_gain_cal_err,
                "cu_numbins_x": cu_numbins_x,
                "cu_numbins_y": cu_numbins_y,
                "demod_numbins_x": demod_numbins_x,
                "demod_numbins_y": demod_numbins_y,
                "gen_err": gen_err,
            }
            print("{}: Performing ViSP reduction on CS data in {}".format(tag(), raw_pol_dir))
            reduce_CS(raw_pol_dir, pol_args, threads=threads)

        if os.path.exists(cmp1_final_name) and os.path.exists(cmp2_final_name):
            print("{}: Found previous fit to CU parameters in {}".format(tag(), pol_rdx_dir))
        else:
            if not (os.path.exists(cmp1_name) and os.path.exists(cmp2_name)):
                print("{}: Using PA&C Modules to fit CU parameters".format(tag()))
                run_PAC_CUpar(
                    pol_rdx_dir,
                    fit_mode=fit_mode,
                    init_set=init_set,
                    telescope_db=telescope_db,
                    threads=threads,
                )
            else:
                print("{}: Found un-resampled CU parameter fits in {}".format(tag(), pol_rdx_dir))

            print("{}: Resampling CU Parameters to Demodulation RCUH".format(tag()))
            cmpData1 = Data.FitsData(cmp1_name, ignore_shape_check=True)
            resamp1 = upsample_CU_parameters(cmpData1, demod_numbins_x, demod_numbins_y)
            resamp1.write_out(cmp1_final_name)

            cmpData2 = Data.FitsData(cmp2_name, ignore_shape_check=True)
            resamp2 = upsample_CU_parameters(cmpData2, demod_numbins_x, demod_numbins_y)
            resamp2.write_out(cmp2_final_name)

        print("{}: Using PA&C Modules to compute demodulation matrices".format(tag()))
        raw_dmod = run_PAC_demod(pol_rdx_dir, telescope_db=telescope_db, threads=threads)

    print("{}: Generating Instrument Polarimetric Calibration".format(tag()))
    instpol_calibration = generate_instrument_pol_calibration(raw_dmod, output)

    return instpol_calibration

    [docs]def reduce_CS(raw_pol_dir: str, run_kwargs: dict, threads: int = 1) -> None:
    """ViSP reduce all steps in a set of PolCal Calibration Sequence (CS) observations

    This function basically just sets up some directories and calls `manage_CS_rdx`.

    Parameters
    ----------
    raw_pol_dir : str
        Location of directory that contains raw Calibration Sequence (CS) data

    run_kwargs : dict
        Keyword arguments to be passed to `reduce_single_step`. Here is where all the Calibration objects live

    threads : int
        Maximum number of cpus/threads to use during reduction

    Returns
    -------
    None
    """
    outdir = "{}/polcal_proc".format(Data.TMPDIR)
    os.makedirs(outdir, exist_ok=True)
    os.makedirs("{}/cu_RCUH".format(outdir), exist_ok=True)
    os.makedirs("{}/demod_RCUH".format(outdir), exist_ok=True)

    print("{}: Starting PolCal data ViSP reduction".format(tag()))
    manage_CS_rdx(raw_pol_dir, outdir, run_kwargs, threads=threads)

    # Create a file that indicates all files were successfully reduced
    # TODO: This file gets made even if the multiproc reduction failed. Fix this.
    with open("{}/rdx_complete".format(outdir), "w") as f:
        f.write(
            "ViSP PolCal reduction of Calibration sequence in {}\nsaved to {}/beam[12] on {}".format(
                raw_pol_dir, outdir, time.asctime()
            )
        )

    return
    



    [docs]def manage_CS_rdx(raw_pol_dir: str, outpre: str, run_kwargs: dict, threads: int = 1) -> None:
    """Manage a set of PolCal reduction processes using multiprocessing.

    Each step in the Calibration Sequence (i.e., separate file) is assigned its own process and then the processes are
    started in such a way that there are never more than the user-defined maximum running at a single time.

    Parameters
    ----------
    raw_pol_dir : str
        Location of raw PolCal frames

    outpre : str
        Prefix of location to save reduced frames

    pol_kwargs : dict
        The Polarimetric Calibration keyword arguments that contain the intermediate calibration objects and any other
        reduction options

    threads : int
        Maximum number of cpus/threads to use during reduction

    Returns
    -------
    None
    """

    print("{}: Running ViSP reduction on all files with {} threads".format(tag(), threads))
    print("{}: Scanning polcal files in {}".format(tag(), raw_pol_dir))
    SoCCData = Data.SoCCData(raw_pol_dir)
    num_steps = SoCCData.num_CS_steps
    if num_steps == 0:
        raise FileNotFoundError(
            "Did not find any calibration sequence steps in {}".format(raw_pol_dir)
        )

    print("{}: Found {} calibration sequence steps".format(tag(), num_steps))

    # This loop sets up a list of mp processes. They have not been started.
    jobs = []
    run_kwargs["dark_lock"] = mp.Lock()
    run_kwargs["gain_lock"] = mp.Lock()
    run_kwargs["geo_lock"] = mp.Lock()
    for j in range(num_steps):
        jobs.append(
            mp.Process(
                target=reduce_single_step,
                args=[SoCCData, j, outpre],
                kwargs=run_kwargs,
                name="step_{:02}".format(j),
            )
        )

    # Now, start jobs and keep starting jobs until the max number of threads is reached
    #  Make a check every second to see if any jobs have finished and thus created room for new ones to start.
    i = 0
    while i < len(jobs):
        while [x.is_alive() for x in jobs].count(True) < threads:
            jobs[i].start()
            i += 1
            if i >= len(jobs):
                break
        time.sleep(1)

    # Finally, make sure we wait for all jobs to complete before moving on.
    for p in jobs:
        p.join()

    return
    



    [docs]def reduce_single_step(
    SoCCData: Data.SoCCData,
    CS_step: int,
    outpre: str,
    dark_cal: Optional[Data.FitsData] = None,
    solar_gain_cal: Optional[Data.FitsData] = None,
    geometric_cal: Optional[Data.FitsData] = None,
    dark_cal_err: Optional[Data.FitsData] = None,
    solar_gain_cal_err: Optional[Data.FitsData] = None,
    gen_err: bool = False,
    cu_numbins_x: int = 1,
    cu_numbins_y: int = 1,
    demod_numbins_x: int = 1,
    demod_numbins_y: int = 1,
    dark_lock: Union[mps.Lock, ContextManager] = NULLCONTEXT,
    gain_lock: Union[mps.Lock, ContextManager] = NULLCONTEXT,
    geo_lock: Union[mps.Lock, ContextManager] = NULLCONTEXT,
) -> Tuple[Data.FitsData, Data.FitsData, Optional[Data.FitsData], Optional[Data.FitsData]]:
    """Perform full ViSP reduction on a single step from a PolCal Calibration Sequence (CS)

    Dark, gain, and geometric effects are removed and the two beams are split into two separate objects/files. An
    arbitrary amount of binning of the final data is also possible. Binning the data results in faster PA&C analysis,
    at the expense of spatial/spectral resolution in the derived Calibration Unit parameters.

    Parameters
    ----------
    SoCCData : `Data.SoCCData`
        Object containing a mapping from CS step to a list of raw files

    CS_step : int
        Calibration Sequence step number to use for reduction

    outpre : str
        Prefix of the location to save processed data. Can include the full path. The processed data will be saved as
        outpre_beam1.fits and outpre_beam2.fits, and associated uncertainties (if provided) will be saved as
        outpre_beam1_err.fits and outpre_beam2_err.fits.

    dark_cal : Data.FitsData
        A Dark Calibration object. Probably the result of DarkCalibration.main().

    solar_gain_cal : Data.FitsData
        A Solar Gain Calibration object. Probably the result of SolarGainCalibration.main().

    geometric_cal : Data.FitsData
        A Geometric Calibration object. Probably produced by SolarGainCalibration.main().

    dark_cal_err : Data.FitsData
        Uncertainty associated with dark_cal

    solar_gain_cal_err : Data.FitsData
        Uncertainty associated with solar_gain_cal

    gen_err : bool
        If True then pixel uncertainty estimates will be generated from the raw data frames

    cu_numbins_x : int
        Each beam will have this number of spatial bins when computing the best CU parameters

    cu_numbins_y : int
        Each beam will have this number of spectral bins when computing the best CU parameters

    demod_numbins_x : int
        Each beam will have this number of spatial bins in the final demodulation matrices

    demod_numbins_y : int
        Each beam will have this number of spectral bins in the final demodulation matrices

    dark_lock : multiprocessing.synchronize.lock
        This lock will be used when reading information from the dark calibration object. This avoids some weirdness in
        astropy's FITS containers when accessing the same data at the same time.

    gain_lock : multiprocessing.synchronize.lock
        This lock will be used when reading information from the gain calibration object. This avoids some weirdness in
        astropy's FITS containers when accessing the same data at the same time.

    geo_lock : multiprocessing.synchronize.lock
        This lock will be used when reading information from the geometric calibration object. This avoids some
        weirdness in astropy's FITS containers when accessing the same data at the same time.

    Returns
    -------
    Data.FitsData
        PolCal ViSP data for beam 1

    Data.FitsData
        PolCal ViSP data for beam 2

    Data.FitsData
        If uncertainty on the input data is provided, the uncertainty associated with beam 1 PolCal data

    Data.FitsData
        If uncertainty on the input data is provided, the uncertainty associated with beam 2 PolCal data
    """
    polData = SoCCData.get_CC_data(CS_step)
    print("{}: Loaded CS step {} from {}".format(tag(), CS_step, SoCCData))

    if gen_err:
        polErr = generic.generate_noise_frame(polData)
    else:
        polErr = None

    if (
        polData.headers[0][S122["PolarimeterMode"]].lower()
        not in CONSTANTS["InstrumentPolarizationCalibration"]["polmode_matches"]
    ):
        raise Exception("Data are not from a full stokes run")

    if dark_cal is not None:
        print("{}: doing Dark Correction".format(tag()))
        with dark_lock:
            print("{}: acquired dark MultiProcessing Lock".format(tag()))
            generic.dark_correction(polData, dark_cal, polErr, dark_cal_err)
        print("{}: released dark MultiProcessing Lock".format(tag()))

    if solar_gain_cal is not None:
        print("{}: doing Gain Correction with Solar Flat".format(tag()))
        with gain_lock:
            print("{}: acquired gain MultiProcessing Lock".format(tag()))
            generic.gain_correction(polData, solar_gain_cal, polErr, solar_gain_cal_err)
        print("{}: released gain MultiProcessing Lock".format(tag()))

    if geometric_cal is not None:
        print("{}: removing geometric/spectral distortions".format(tag()))
        with geo_lock:
            print("{}: acquired geo MultiProcessing Lock".format(tag()))
            geo_numbers = ScienceCalibration.extract_geometric_numbers(geometric_cal)
        print("{}: released geo MultiProcessing Lock".format(tag()))
        polData, polErr = ScienceCalibration.geometric_correction(
            polData, geo_numbers, errData=polErr
        )

    for xbin, ybin, suffix, name in zip(
        [cu_numbins_x, demod_numbins_x],
        [cu_numbins_y, demod_numbins_y],
        ["cu_RCUH", "demod_RCUH"],
        ["CU parameter fit", "demodulation matrix calculation"],
    ):
        print("{}: extracting macropixels for {}".format(tag(), name))
        binData_1, binData_2, binErr_1, binErr_2 = extract_macropixels(
            polData, xbin, ybin, errData=polErr
        )

        for beamData in [binData_1, binData_2]:
            update_header_noise_floor(beamData)

        print("{}: saving reduced PolCal data for {}".format(tag(), name))
        write_reduced_CS_step(
            binData_1, binData_2, "{}/{}".format(outpre, suffix), CS_step, binErr_1, binErr_2
        )

    return binData_1, binData_2, binErr_1, binErr_2
    



    [docs]def write_reduced_CS_step(
    binData_1: Data.FitsData,
    binData_2: Data.FitsData,
    outpre: str,
    CS_step: int,
    err_1: Optional[Data.FitsData] = None,
    err_2: Optional[Data.FitsData] = None,
) -> None:
    """Write reduced PolCal to a sensible place.

    The sensible place is DIR/beam[1|2]/VISP_CS_STEP.PAC.beam[1|2].fits where DIR and STEP are defined by inputs.

    Parameters
    ----------
    binData_1 : Data.FitsData
        PolCal ViSP data for beam 1. Each modulator state in a separate ImageHDU.

    binData_2 : Data.FitsData
        PolCal ViSP data for beam 2. Each modulator state in a separate ImageHDU.

    outpre : str
        Path to top level directory that contains the reduction products

    CS_step : int
        The number of the CS step represented by the input data

    err_1 : Data.FitsData
        Uncertainty associated with binData1

    err_2 : Data.FitsData
        Uncertainty associated with binData2

    """
    os.makedirs("{}/beam1".format(outpre), exist_ok=True)
    os.makedirs("{}/beam2".format(outpre), exist_ok=True)
    for obj, post, beamnum in zip(
        [binData_1, binData_2, err_1, err_2],
        [
            "." + CONSTANTS["Data"]["suffix"],
            "." + CONSTANTS["Data"]["suffix"],
            "_err." + CONSTANTS["Data"]["suffix"],
            "_err." + CONSTANTS["Data"]["suffix"],
        ],
        [1, 2, 1, 2],
    ):
        if obj is not None:
            outname = "{0}/beam{1}/VISP_CS_{2:02}.PAC.beam{1}{3}".format(
                outpre, beamnum, CS_step, post
            )
            obj.write_out(outname, overwrite=True)

    return

    [docs]def extract_macropixels(
    fitsData: Data.FitsData, numpix_x: int, numpix_y: int, errData: Optional[Data.FitsData] = None
) -> Tuple[Data.FitsData, Data.FitsData, Optional[Data.FitsData], Optional[Data.FitsData]]:
    """Bin the input data down to a specified number of pixels.

    Because this function is designed to produce PolCal data the two ViSP beams are split apart and placed into separate
    Data.FitsData objects. Additionally, a third, dummy dimension is added to conform with PAC inputs.

    If the number of output pixels does not divide evenly into the number of input pixels then some input pixels will be
    lost. This might or might not be a bad thing.

    Parameters
    ----------
    fitsData : Data.FitsData
        Object containing data to bin. Should contain both ViSP beams and is probably the output of
        ScienceCalibration.geometric_correction()

    numpix_x : int
        The number of spectral output pixels PER BEAM

    numpix_y : int
        The number of spatial output pixels. Both beams will have the same number of spatial pixels.

    errData : Data.FitsData
        Uncertainty associated with the input data

    Returns
    -------
    Data.FitsData
        PolCal ViSP data for beam 1. Each modulator state in a separate ImageHDU.

    Data.FitsData
        PolCal ViSP data for beam 2. Each modulator state in a separate ImageHDU.

    Data.FitsData
        If uncertainty on the input data is provided, the uncertainty associated with beam 1 PolCal data

    Data.FitsData
        If uncertainty on the input data is provided, the uncertainty associated with beam 2 PolCal data
    """
    print("{}: Binning beams to shape ({}, {}, 1)".format(tag(), numpix_x, numpix_y))

    pivot = CONSTANTS["Data"]["beam_border"]
    nummod = fitsData.headers[0][S122["NumberOfModulatorStates"]]
    numx_all, numy = fitsData.data[0].shape
    numx_2 = numx_all - pivot

    print("{}: full beam 1 shape: ({}, {})".format(tag(), pivot, numy))
    print("{}: full beam 2 shape: ({}, {})".format(tag(), numx_2, numy))

    if numpix_x > pivot or numpix_x > numx_2 or numpix_y > numy:
        raise IndexError("Cannot bin to more pixels than input data")

    xbin_1 = pivot // numpix_x
    xbin_2 = numx_2 // numpix_x
    ybin = numy // numpix_y

    print("{}: beam 1 x binning: {}".format(tag(), xbin_1))
    print("{}: beam 2 x binning: {}".format(tag(), xbin_2))
    print("{}: y binning: {}".format(tag(), ybin))

    xend_1 = numpix_x * xbin_1 - pivot or None
    xend_2 = numpix_x * xbin_2 - numx_2 or None
    yend = numpix_y * ybin - numy or None

    if (numy / numpix_y) % 1 != 0:
        print(
            "{}: Spatial binning amount is non-integer; {} pixels will be lost from beams 1 and 2".format(
                tag(), abs(yend)
            )
        )

    if (pivot / numpix_x) % 1 != 0 or (numx_2 / numpix_x) % 1 != 0:
        print(
            "{}: Spectral binning amount is non-integer; {}, {} pixels will be lost from beams 1 and 2".format(
                tag(), abs(xend_1), abs(xend_2)
            )
        )

    hdulist_1 = [pyfits.PrimaryHDU(header=fitsData.hdulist[0].header)]
    hdulist_1[0].header[S122["SoftwareBinningX"]] = xbin_1
    hdulist_1[0].header[S122["SoftwareBinningY"]] = ybin
    hdulist_1[0].header["VISP_OGX"] = pivot
    hdulist_1[0].header["VISP_OGY"] = numy
    hdulist_1[0].header["VISP_BEM"] = 1

    hdulist_2 = [pyfits.PrimaryHDU(header=fitsData.hdulist[0].header)]
    hdulist_2[0].header[S122["SoftwareBinningX"]] = xbin_2
    hdulist_2[0].header[S122["SoftwareBinningY"]] = ybin
    hdulist_2[0].header["VISP_OGX"] = numx_2
    hdulist_2[0].header["VISP_OGY"] = numy
    hdulist_2[0].header["VISP_BEM"] = 2

    if errData is not None:
        hdulist_1_err = [pyfits.PrimaryHDU(header=hdulist_1[0].header)]
        hdulist_2_err = [pyfits.PrimaryHDU(header=hdulist_2[0].header)]
    else:
        binData_1_err = None
        binData_2_err = None

    for i in range(1, nummod + 1):
        print("{}: modulator state {}:".format(tag(), i))
        idx = fitsData.find_data_keyword_idx(S122["CurrentState"], i)
        # TODO: is len(idx) ever > 1? If it is, should it be?
        for j in idx:
            print("{}:  extracting from data HDU {}".format(tag(), j))
            data = fitsData.data[j]
            tmp1 = data[:pivot, :]
            tmp2 = data[pivot:, :]

            noise_floor1 = np.nanstd(tmp1) * CONSTANTS["Data"]["noise_floor_err_fraction"]
            noise_floor2 = np.nanstd(tmp2) * CONSTANTS["Data"]["noise_floor_err_fraction"]

            x_binned_data1 = np.zeros((numpix_x, numy))
            binned_data1 = np.zeros((numpix_x, numpix_y))
            x_binned_data2 = np.zeros((numpix_x, numy))
            binned_data2 = np.zeros((numpix_x, numpix_y))

            for l in range(xbin_1):
                x_binned_data1 += tmp1[l:xend_1:xbin_1, :]
            for l in range(xbin_2):
                x_binned_data2 += tmp2[l:xend_2:xbin_2, :]
            for k in range(ybin):
                binned_data1 += x_binned_data1[:, k:yend:ybin]
                binned_data2 += x_binned_data2[:, k:yend:ybin]

            # Add the 3rd dummy dimension here with the [:, :, None]
            macro_data_1 = np.asarray(binned_data1 / (xbin_1 * ybin), dtype=np.float32)
            ih_1 = pyfits.ImageHDU(macro_data_1[:, :, None], header=fitsData.data_hdus[j].header)
            ih_1.header[S122["SoftwareBinningX"]] = xbin_1
            ih_1.header[S122["SoftwareBinningY"]] = ybin
            ih_1.header["VISP_OGX"] = pivot
            ih_1.header["VISP_OGY"] = numy
            ih_1.header["VISP_BEM"] = 1
            ih_1.header[S122["NoiseFloor"]] = noise_floor1
            hdulist_1.append(ih_1)

            macro_data_2 = np.asarray(binned_data2 / (xbin_2 * ybin), dtype=np.float32)
            ih_2 = pyfits.ImageHDU(macro_data_2[:, :, None], header=fitsData.data_hdus[j].header)
            ih_2.header[S122["SoftwareBinningX"]] = xbin_2
            ih_2.header[S122["SoftwareBinningY"]] = ybin
            ih_2.header["VISP_OGX"] = numx_2
            ih_2.header["VISP_OGY"] = numy
            ih_2.header["VISP_BEM"] = 2
            ih_2.header[S122["NoiseFloor"]] = noise_floor2
            hdulist_2.append(ih_2)

            if errData is not None:
                print("{}:  extracting from uncertinaty HDU {}".format(tag(), j))
                err = errData.data[j]
                tmp1_err = err[:pivot, :]
                tmp2_err = err[pivot:, :]

                noise_floor1 = np.nanmean(tmp1_err) * CONSTANTS["Data"]["noise_floor_err_fraction"]
                noise_floor2 = np.nanmean(tmp2_err) * CONSTANTS["Data"]["noise_floor_err_fraction"]
                hdulist_1[j].header[S122["NoiseFloor"]] = noise_floor1
                hdulist_2[j].header[S122["NoiseFloor"]] = noise_floor2

                x_binned_data1_err = np.zeros((numpix_x, numy))
                binned_data1_err = np.zeros((numpix_x, numpix_y))
                x_binned_data2_err = np.zeros((numpix_x, numy))
                binned_data2_err = np.zeros((numpix_x, numpix_y))

                for l in range(xbin_1):
                    x_binned_data1_err += tmp1_err[l:xend_1:xbin_1, :]
                for l in range(xbin_2):
                    x_binned_data2_err += tmp2_err[l:xend_2:xbin_2, :]
                for k in range(ybin):
                    x_binned_data1_err += x_binned_data1[:, k:yend:ybin]
                    x_binned_data2_err += x_binned_data2[:, k:yend:ybin]

                macro_data_1_err = np.asarray(binned_data1_err / (xbin_1 * ybin), dtype=np.float32)
                ih_1_err = pyfits.ImageHDU(macro_data_1_err[:, :, None], header=hdulist_1[j].header)
                hdulist_1_err.append(ih_1_err)

                macro_data_2_err = np.asarray(binned_data2_err / (xbin_2 * ybin), dtype=np.float32)
                ih_2_err = pyfits.ImageHDU(macro_data_2_err[:, :, None], header=hdulist_2[j].header)
                hdulist_2_err.append(ih_2_err)

    binData_1 = Data.FitsData(ignore_shape_check=True)
    binData_1.origin = "extract macropixels"
    binData_1.hdulist = hdulist_1
    binData_2 = Data.FitsData(ignore_shape_check=True)
    binData_2.origin = "extract macropixels"
    binData_2.hdulist = hdulist_2

    if errData is not None:
        binData_1_err = Data.FitsData(ignore_shape_check=True)
        binData_1_err.origin = "extract macropixels"
        binData_1_err.hdulist = hdulist_1_err
        binData_2_err = Data.FitsData(ignore_shape_check=True)
        binData_2_err.origin = "extract macropixels"
        binData_2_err.hdulist = hdulist_2_err

    return binData_1, binData_2, binData_1_err, binData_2_err

    [docs]def upsample_demodulation(data: np.nanstd, header: pyfits.header.Header) -> np.ndarray:
    """Take a single demodulation matrix computed for binned data and up-sample it to the full ViSP frame.

    Up-sampling is done using nearest-neighbor interpolation in such a way to ensure that each macropixel (i.e. bin)
    is equally represented in the final output matrices. The exception to this is when the number of macropixels was
    not an integer divisor of the full frame. In this case the macropixel on the edge of each beam will be slightly
    (typically less than 5 px) over represented in the final matrices.

    All information on the shape of the full frame is pulled from the FITS headers, so they'd better be correct!

    Parameters
    ----------
    data : `numpy.ndarray`
        Data for a single beam's demodulation matrix. Should have shape (x, y, 4, M).

    header : `astropy.io.fits.header.Header`
        Header object associated with the data. Used to define beam sizes.

    Returns
    -------
    numpy.ndarray
        The input data upsampled to match the full size of the corresponding ViSP beam.

    Notes
    -----
    This function will work exactly as expected if x = y = 1, but in this case you're probably better off just
    squeezing out the extra dimensions and taking advantage of the (mostly) shape-agnostic input of
    ScienceCalibration.polarimetric_multiplication
    """
    full_x = header["VISP_OGX"]
    full_y = header["VISP_OGY"]
    numx, numy, nums, numm = data.shape
    full_output = np.zeros((full_x, full_y, nums, numm), dtype=np.float32)
    print("{}: original shape of beam: ({}, {}, {}, {})".format(tag(), full_x, full_y, nums, numm))

    print("{}:  demodulation matrix shape: ({}, {}, {}, {})".format(tag(), numx, numy, nums, numm))

    x = np.arange(full_x // numx / 2, full_x, full_x // numx)
    y = np.arange(full_y // numy / 2, full_y, full_y // numy)

    print("{}:  interpolating".format(tag()))

    xi, yi = np.mgrid[0 : full_x : full_x * 1j, 0 : full_y : full_y * 1j]
    for i in range(nums):
        print("{}:  stokes parameter {}".format(tag(), i))
        for j in range(numm):
            print("{}:   modulator state {}".format(tag(), j))
            full_output[:, :, i, j] = spi.interpn(
                (x, y),
                data[:, :, i, j],
                (xi, yi),
                bounds_error=False,
                fill_value=None,
                method="nearest",
            )

    return full_output
    



    [docs]def upsample_CU_parameters(CuData: Data.FitsData, newx: int, newy: int) -> Data.FitsData:
    """Take CU parameters fit to a subsample of the ViSP frame and upsample to some new sampling

    Up-sampling is done using nearest-neighbor interpolation in such a way to ensure that each macropixel (i.e. bin)
    is equally represented in the upsampled data. The exception to this is when there is not an integer multiple between
    the current binning and desired binning.

    Despite the 2D nature of ViSP's detector, CU parameters are saved/loaded by the PA&C Modules with three dimensions.
    Thus, the shapes of the input and output will be (X, Y, L, Z), where L is the extra PAC dimension and Z is the
    number of parameters.

    Parameters
    ----------
    CuData : `Data.FitsData`
        Object containing the CU parameters. Probably the result of PA&C's `pac_cu_pars`

    newx : int
        Number of x pixels in umsampled field

    newy : int
        Number of y pixels in upsampled field

    Returns
    -------
    Data.FitsData
        Object containing the resampled CU parameters
    """
    # This function probably produces non-ideal behavior if downsampling, with non-idealness increasing with
    # downsampling amount.

    numx, numy, numl, numz = CuData.data[0].shape
    print("{}: shape of CU parameters: ({}, {}, {}, {})".format(tag(), numx, numy, numl, numz))

    full_output = np.zeros((newx, newy, numl, numz), dtype=np.float64)
    print("{}: resampling to new shape ({}, {}, {}, {})".format(tag(), newx, newy, numl, numz))

    x = np.arange(newx / 2, newx * numx, newx) / numx
    y = np.arange(newy / 2, newy * numy, newy) / numy

    print("{}:  interpolating".format(tag()))

    xi, yi = np.mgrid[0 : newx * numx : newx * 1j, 0 : newy * numy : newy * 1j]
    xi /= numx
    yi /= numy
    for i in range(numz):
        print("{}:  CU parameter {}".format(tag(), i))
        full_output[:, :, :, i] = spi.interpn(
            (x, y),
            CuData.data[0][:, :, :, i],
            (xi, yi),
            bounds_error=False,
            fill_value=None,
            method="nearest",
        )

    out_data = Data.FitsData(ignore_shape_check=True)
    out_data.hdulist = [pyfits.PrimaryHDU(full_output, header=CuData.headers[0])]

    return out_data

    [docs]def generate_instrument_pol_calibration(
    dmodData: Data.DemodulationData, output: str
) -> Data.DemodulationData:
    """Read raw demodulation matrices and prep them for use in Science data reduction

    The demodulation matrices for each ViSP beam are treated separately and if either one was a full-frame average
    then the spatial dimensions are simply squeezed out. Otherwise the data are passed to `upsample_demodulation` for
    interpolation to the full beam size.

    Parameters
    ----------
    dmodData : Data.DemodulationData
        The demodulation matrix for both ViSP beams. Should have a single data HDU for each beam with shape
        (x, y, 4, M) where (x, y) is the shape of the binned ViSP beam 1 frame passed to PA&C and M is the number of
        unique modulator states.

    Returns
    -------
    Data.DemodulationData
        Object containing a data HDU for each ViSP beam. The data shape of these HDUs will either be (X, Y, 4, M) or
        (4, M), where X and Y are the full read-out size of a single ViSP beam.
    """
    print("{}: extracting demoulation matrices".format(tag()))
    numx1, numy1, _, numm1 = dmodData.beam1.shape
    numx2, numy2, _, numm2 = dmodData.beam2.shape

    if numm1 != numm2:
        raise ValueError("The demodulation matrices don't have the same number of modulator states")
    if dmodData.beam1_header["VISP_OGX"] != dmodData.beam2_header["VISP_OGX"]:
        raise ValueError("Beams 1 and 2 not not have the same original spatial dimension")

    final_dmod = Data.DemodulationData()
    for i, (numx, numy, data, head) in enumerate(
        zip(
            [numx1, numx2],
            [numy1, numy2],
            [dmodData.beam1, dmodData.beam2],
            [dmodData.beam1_header, dmodData.beam2_header],
        )
    ):

        if numx == numy == 1:
            print("{}: Beam {} is a whole beam, passing through".format(tag(), i + 1))
            final_dmod.init_beam_with_ndarray(data[0, 0, :, :], i + 1, header=head)
        else:
            print("{}: Upsampling beam {}".format(tag(), i + 1))
            data = upsample_demodulation(data, head)
            final_dmod.init_beam_with_ndarray(data, i + 1, header=head)

    final_dmod.write_out(output, overwrite=True)

    return final_dmod
