    [docs]def main(raw_pol_dir, output, raw_dark_dir, raw_lamp_gain_dir, raw_solar_gain_dir,
         gen_noise_frames=False, telescope_db='telescope_db.txt', num_bins_x=1, num_bins_y=1, threads=1):
    """Run raw data (i.e., a PolCal CS observation) through the Instrument Polarization Calibration pipeline

    The steps in the pipeline are:

    1. Generate intermediate (dark, gain, etc.) Calibration objects associated with CS

    2. Perform ViSP reductions on entire CS data set

    3. Fit CU parameters and generate demodulation matrices from the resulting SoCCs

    So basically, this pipeline calls the normal ViSP pipeline and then uses PA&C Modules to further analyize the
    reduced PolCal data.

    Parameters
    ----------
    raw_pol_dir : str
        Path to directory containing raw PolCal data

    output : str
        Path were final InstrumentPolCalibration (i.e., demodulation matrices) will be saved

    raw_dark_dir : str
        Path to directory containing raw dark data associated with PolCal observations

    raw_lamp_gain_dir : str
        Path to directory containing raw lamp gain data associated with PolCal observations

    raw_solar_gain_dir : str
        Path to directory containing raw solar gain data associated with PolCal observations

    gen_noise_frames : bool
        If True then uncertainty estimates will be generated for all Calibration and data products

    telescope_db : str
        Path to the database of telescope properties (x, \tau)

    num_bins_x : int
        Each beam will have this number of spatial bins in the final demodulation matrices

    num_bins_y : int
        Each beam will have this number of spectral bins in the final demodulation matrices

    threads : int
        The number of independent processes to use when performing ViSP reductions on the PolCal data

    Returns
    -------
    None
    """
    pol_rdx_dir = '{}/proc'.format(raw_pol_dir)
    dmod1_name = '{}/DMOD_beam1.fits'.format(pol_rdx_dir)
    dmod2_name = '{}/DMOD_beam2.fits'.format(pol_rdx_dir)
    if os.path.exists(dmod1_name) and os.path.exists(dmod2_name):
        print('{}: Found previous demodulation matrices in {}'.format(tag(), pol_rdx_dir))
        raw_dmod = Data.DemodulationData()
        raw_dmod.load_beam_from_file(dmod1_name, 1)
        print('{}: loaded {}'.format(tag(), dmod1_name))

        raw_dmod.load_beam_from_file(dmod2_name, 2)
        print('{}: loaded {}'.format(tag(), dmod2_name))

    else:
        pol_args = {'raw_pol_dir': raw_pol_dir,
                    'raw_dark_dir': raw_dark_dir,
                    'raw_lamp_gain_dir': raw_lamp_gain_dir,
                    'raw_solar_gain_dir': raw_solar_gain_dir,
                    'num_bins_x': num_bins_x,
                    'num_bins_y': num_bins_y,
                    'gen_noise_frames': gen_noise_frames,
                    'threads': threads}

        if not os.path.exists('{}/rdx_complete'.format(pol_rdx_dir)):
            print('{}: Performing ViSP reduction on CS data in {}'.format(tag(), raw_pol_dir))
            reduce_CS(**pol_args)

        print('{}: Using PA&C Modules to compute demodulation matrices'.format(tag()))
        raw_dmod = run_PAC_demod(pol_rdx_dir, telescope_db = telescope_db, threads = threads)

    print('{}: Generating Instrument Polarimetric Calibration'.format(tag()))
    instpol_calibration = generate_instrument_pol_calibration(raw_dmod, output)

    return instpol_calibration

              num_bins_x=1, num_bins_y=1, dark_cal='DarkCal.fits', lamp_gain_cal='LampGainCal.fits',
              solar_gain_cal='SolarGainCal.fits', geometric_cal='GeoCal.fits', threads=1,
              gen_noise_frames=False):
    """ViSP reduce all steps in a set of PolCal Calibration Sequence (CS) observations

    This function basically constructs the correct config and kwarg objects and then does all of the actual processing
    via calls to `generate_intermediate_objects` and `manage_CS_rdx`.

    Parameters
    ----------
    raw_pol_dir : str
        Location of directory that contains raw Calibration Sequence (CS) data

    raw_dark_dir : str
        Location of the directory containing raw dark frames associated with the CS

    raw_lamp_gain_dir : str
        Location of the directory containing raw lamp gain frames associated with the CS

    raw_solar_gain_dir : str
        Location of the directory containing raw solar gain frames associated with the CS

    num_bins_x : int
        The final number of macropixels to extract/average from the processed CS data, spectral dimension.

    num_bins_y : int
        The final number of macropixels to extract/average from the processed CS data, spatial dimension.

    dark_cal : str
        Location of the processed DarkCalibration object to either load or save

    lamp_gain_cal : str
        Location of the processed LampGainCalibration object to either load or save

    solar_gain_cal : str
        Location of the processed SolarGainCalibration object to either load or save

    geometric_cal : str
        Location of the processed GeometricCalibration object to either load or save

    threads : int
        Maximum number of cpus/threads to use during reduction

    gen_noise_frames : bool
        If True then uncertainty estimates will be generated for the reduced CS step data

    Returns
    -------
    None
    """
    outdir = '{}/proc'.format(raw_pol_dir)
    os.makedirs(outdir, exist_ok=True)

    # Make a dummy ConfigParser object with a Pol-only 'Main' section
    darkcal_file = '{}/{}'.format(outdir, dark_cal)
    lampcal_file = '{}/{}'.format(outdir, lamp_gain_cal)
    solarcal_file = '{}/{}'.format(outdir, solar_gain_cal)
    geocal_file = '{}/{}'.format(outdir, geometric_cal)
    run_args = {'Main': {'raw_pol_dir': raw_pol_dir,
                         'output_dir': outdir,
                         'dark_cal': darkcal_file,
                         'lamp_gain_cal': lampcal_file,
                         'solar_gain_cal': solarcal_file,
                         'geometric_cal': geocal_file,
                         'gen_noise_frames': gen_noise_frames,
                         'threads': threads},
                'DarkCalibration': {'raw_dark_dir': raw_dark_dir},
                'LampGainCalibration': {'raw_lamp_gain_dir': raw_lamp_gain_dir},
                'SolarGainCalibration': {'raw_solar_gain_dir': raw_solar_gain_dir}}

    config = ConfigParser()
    config.read_dict(run_args)

    # Generate/load the necessary intermediate Calibration objects
    pol_kwargs = ViSPPipeline.generate_intermediate_objects(config)
    pol_kwargs.update({'numpix_x': num_bins_x,
                       'numpix_y': num_bins_y})

    # Run the CS through the PolCal reduction pipeline
    print('{}: Starting PolCal data ViSP reduction'.format(tag()))
    manage_CS_rdx(config, pol_kwargs)

    # Create a file that indicates all files were successfully reduced
    with open('{}/rdx_complete'.format(outdir), 'w') as f:
        f.write('ViSP PolCal reduction of Calibration sequence in {}\nsaved to {}/beam[12] on {}'. \
                format(raw_pol_dir, outdir, time.asctime()))

    return
    



    [docs]def manage_CS_rdx(config, pol_kwargs):
    """Manage a set of PolCal reduction processes using multiprocessing.

    Each step in the Calibration Sequence (i.e., separate file) is assigned its own process and then the processes are
    started in such a way that there are never more than the user-defined maximum running at a single time.

    Parameters
    ----------
    config : configparser.ConfigParser
        The object containing user-defined reduction options

    pol_kwargs : dict
        The Polarimetric Calibration keyword arguments that contain the intermediate calibration objects and any other
        reduction options

    Returns
    -------
    None
    """

    numthread = int(config['Main']['threads'])
    print('{}: Running ViSP reduction on all files with {} threads'.format(tag(), numthread))
    num_steps = find_number_of_CS_steps(config['Main']['raw_pol_dir'])
    print('{}: Found {} calibration sequence steps'.format(tag(), num_steps))

    # This loop sets up a list of mp processes. They have not been started.
    jobs = []
    pol_kwargs['dark_lock'] = mp.Lock()
    pol_kwargs['gain_lock'] = mp.Lock()
    pol_kwargs['geo_lock'] = mp.Lock()
    for j in range(num_steps):
        outpre = config['Main']['output_dir']
        jobs.append(mp.Process(target=reduce_single_step,
                               args=[config['Main']['raw_pol_dir'], j, outpre],
                               kwargs=pol_kwargs,
                               name='step_{:02}'.format(j)))

    # Now, start jobs and keep starting jobs until the max number of threads is reached
    #  Make a check every second to see if any jobs have finished and thus created room for new ones to start.
    i = 0
    while i < len(jobs):
        while [x.is_alive() for x in jobs].count(True) < numthread:
            jobs[i].start()
            i += 1
            if i >= len(jobs):
                break
        time.sleep(1)

    # Finally, make sure we wait for all jobs to complete before moving on.
    for p in jobs:
        p.join()

    return
    



    [docs]def reduce_single_step(pol_dir, CS_step, outpre, dark_cal=None, solar_gain_cal=None, geometric_cal=None,
                       dark_cal_err=None, solar_gain_cal_err=None, gen_err=False,
                       numpix_x=1, numpix_y=1,
                       dark_lock=None, gain_lock=None, geo_lock=None):
    """Perform full ViSP reduction on a single step from a PolCal Calibration Sequence (CS)

    Dark, gain, and geometric effects are removed and the two beams are split into two separate objects/files. An
    arbitrary amount of binning of the final data is also possible. Binning the data results in faster PA&C analysis,
    at the expense of spatial/spectral resolution in the derived Calibration Unit parameters.

    Parameters
    ----------
    pol_dir : str
        Path to the directory containing a single SoCC

    CS_step : int
        Calibration Sequence step number to use for reduction

    outpre : str
        Prefix of the location to save processed data. Can include the full path. The processed data will be saved as
        outpre_beam1.fits and outpre_beam2.fits, and associated uncertainties (if provided) will be saved as
        outpre_beam1_err.fits and outpre_beam2_err.fits.

    dark_cal : Data.FitsData
        A Dark Calibration object. Probably the result of DarkCalibration.main().

    solar_gain_cal : Data.FitsData
        A Solar Gain Calibration object. Probably the result of SolarGainCalibration.main().

    geometric_cal : Data.FitsData
        A Geometric Calibration object. Probably produced by SolarGainCalibration.main().

    dark_cal_err : Data.FitsData
        Uncertainty associated with dark_cal

    solar_gain_cal_err : Data.FitsData
        Uncertainty associated with solar_gain_cal

    gen_err : bool
        If True then pixel uncertainty estimates will be generated from the raw data frames

    numpix_x : int
        The number of spatial pixels PER BEAM in the final output data

    numpix_y : int
        The number of spectral pixels PER BEAM in the final output data

    dark_lock : multiprocessing.synchronize.lock
        This lock will be used when reading information from the dark calibration object. This avoids some weirdness in
        astropy's FITS containers when accessing the same data at the same time.

    gain_lock : multiprocessing.synchronize.lock
        This lock will be used when reading information from the gain calibration object. This avoids some weirdness in
        astropy's FITS containers when accessing the same data at the same time.

    geo_lock : multiprocessing.synchronize.lock
        This lock will be used when reading information from the geometric calibration object. This avoids some
        weirdness in astropy's FITS containers when accessing the same data at the same time.

    Returns
    -------
    Data.FitsData
        PolCal ViSP data for beam 1

    Data.FitsData
        PolCal ViSP data for beam 2

    Data.FitsData
        If uncertainty on the input data is provided, the uncertainty associated with beam 1 PolCal data

    Data.FitsData
        If uncertainty on the input data is provided, the uncertainty associated with beam 2 PolCal data
    """
    polData = Data.CalibrationCurveData(pol_dir, CS_step, suffix='VISP*.fits')
    print('{}: Loaded CS step {} from {}'.format(tag(), CS_step, pol_dir))

    if gen_err:
        polErr = generic.generate_noise_frame(polData)
    else:
        polErr = None

    if polData.headers[0]['VISP_PMD'] != 'Full Stokes':
        raise Exception('Data are not from a full stokes run')

    if dark_cal is not None:
        print('{}: doing Dark Correction'.format(tag()))
        with dark_lock:
            print('{}: acquired dark MultiProcessing Lock'.format(tag()))
            generic.dark_correction(polData, dark_cal, polErr, dark_cal_err)
        print('{}: released dark MultiProcessing Lock'.format(tag()))

    if solar_gain_cal is not None:
        print('{}: doing Gain Correction with Solar Flat'.format(tag()))
        with gain_lock:
            print('{}: acquired gain MultiProcessing Lock'.format(tag()))
            generic.gain_correction(polData, solar_gain_cal, polErr, solar_gain_cal_err)
        print('{}: released gain MultiProcessing Lock'.format(tag()))

    if geometric_cal is not None:
        print('{}: removing geometric/spectral distortions'.format(tag()))
        with geo_lock:
            print('{}: acquired geo MultiProcessing Lock'.format(tag()))
            geo_numbers = ScienceCalibration.extract_geometric_numbers(geometric_cal)
        print('{}: released geo MultiProcessing Lock'.format(tag()))
        polData, polErr = ScienceCalibration.geometric_correction(polData, geo_numbers, errData=polErr)

    print('{}: extracting macropixels'.format(tag()))
    binData_1, binData_2, binErr_1, binErr_2 = extract_macropixels(polData, numpix_x, numpix_y, errData=polErr)

    print('{}: saving reduced PolCal data'.format(tag()))
    write_reduced_CS_step(binData_1, binData_2, outpre, CS_step, binErr_1, binErr_2)

    return binData_1, binData_2, binErr_1, binErr_2
    



    [docs]def write_reduced_CS_step(binData_1, binData_2, outpre, CS_step, err_1=None, err_2=None):
    """Write reduced PolCal to a sensible place.

    The sensible place is DIR/beam[1|2]/VISP_CS_STEP.PAC.beam[1|2].fits where DIR and STEP are defined by inputs.

    Parameters
    ----------
    binData_1 : Data.FitsData
        PolCal ViSP data for beam 1. Each modulator state in a separate ImageHDU.

    binData_2 : Data.FitsData
        PolCal ViSP data for beam 2. Each modulator state in a separate ImageHDU.

    outpre : str
        Path to top level directory that contains the reduction products

    CS_step : int
        The number of the CS step represented by the input data

    err_1 : Data.FitsData
        Uncertainty associated with binData1

    err_2 : Data.FitsData

    """
    os.makedirs('{}/beam1'.format(outpre), exist_ok=True)
    os.makedirs('{}/beam2'.format(outpre), exist_ok=True)
    for obj, post, beamnum in zip([binData_1, binData_2, err_1, err_2],
                                  ['.fits', '.fits', '_err.fits', '_err.fits'],
                                  [1, 2, 1, 2]):
        if obj is not None:
            outname = '{0}/beam{1}/VISP_CS_{2:02}.PAC.beam{1}{3}'.format(outpre, beamnum, CS_step, post)
            obj.write_out(outname, overwrite=True)

    return
    



    [docs]def extract_macropixels(fitsData, numpix_x, numpix_y, errData=None):
    """Bin the input data down to a specified number of pixels.

    Because this function is designed to produce PolCal data the two ViSP beams are split apart and placed into separate
    Data.FitsData objects.

    If the number of output pixels does not divide evenly into the number of input pixels then some input pixels will be
    lost. This might or might not be a bad thing.

    Parameters
    ----------
    fitsData : Data.FitsData
        Object containing data to bin. Should contain both ViSP beams and is probably the output of
        ScienceCalibration.geometric_correction()

    numpix_x : int
        The number of spatial output pixels. Both beams will have the same number of spatial pixels.

    numpix_y : int
        The number of spectral output pixels PER BEAM

    errData : Data.FitsData
        Uncertainty associated with the input data

    Returns
    -------
    Data.FitsData
        PolCal ViSP data for beam 1. Each modulator state in a separate ImageHDU.

    Data.FitsData
        PolCal ViSP data for beam 2. Each modulator state in a separate ImageHDU.

    Data.FitsData
        If uncertainty on the input data is provided, the uncertainty associated with beam 1 PolCal data

    Data.FitsData
        If uncertainty on the input data is provided, the uncertainty associated with beam 2 PolCal data
    """
    print('{}: Binning beams to shape ({}, {})'.format(tag(), numpix_x, numpix_y))

    pivot = fitsData.headers[0]['VISP_PPP']
    nummod = fitsData.headers[0]['VISP_XXX']
    numx, numy_all = fitsData.data_hdus[0].data.shape
    numy_2 = (numy_all - pivot)

    print('{}: full beam 1 shape: ({}, {})'.format(tag(), numx, pivot))
    print('{}: full beam 2 shape: ({}, {})'.format(tag(), numx, numy_2))

    if numpix_x > numx or numpix_y > pivot or numpix_y > numy_2:
        raise IndexError('Cannot bin to more pixels than input data')

    xbin = numx // numpix_x
    ybin_1 = pivot // numpix_y
    ybin_2 = numy_2 // numpix_y

    print('{}: x binning: {}'.format(tag(), xbin))
    print('{}: beam 1 y binning: {}'.format(tag(), ybin_1))
    print('{}: beam 2 y binning: {}'.format(tag(), ybin_2))

    xend = numpix_x * xbin - numx or None
    yend_1 = numpix_y * ybin_1 - pivot or None
    yend_2 = numpix_y * ybin_2 - numy_2 or None

    if (numx / numpix_x) % 1 != 0:
        print('{}: Spatial binning amount is non-integer; {} pixels will be lost from beams 1 and 2'. \
              format(tag(), abs(xend)))

    if (pivot / numpix_y) % 1 != 0 or (numy_2 / numpix_y) % 1 != 0:
        print('{}: Spectral binning amount is non-integer; {}, {} pixels will be lost from beams 1 and 2'. \
              format(tag(), abs(yend_1), abs(yend_2)))

    hdulist_1 = [pyfits.PrimaryHDU(header=fitsData.hdulist[0].header)]
    hdulist_1[0].header['CAM__010'] = xbin
    hdulist_1[0].header['CAM__011'] = ybin_1
    hdulist_1[0].header['VISP_OGX'] = numx
    hdulist_1[0].header['VISP_OGY'] = pivot
    hdulist_1[0].header['VISP_BEM'] = 1

    hdulist_2 = [pyfits.PrimaryHDU(header=fitsData.hdulist[0].header)]
    hdulist_2[0].header['CAM__010'] = xbin
    hdulist_2[0].header['CAM__011'] = ybin_2
    hdulist_2[0].header['VISP_OGX'] = numx
    hdulist_2[0].header['VISP_OGY'] = numy_2
    hdulist_2[0].header['VISP_BEM'] = 2

    if errData is not None:
        hdulist_1_err = [pyfits.PrimaryHDU(header=hdulist_1[0].header)]
        hdulist_2_err = [pyfits.PrimaryHDU(header=hdulist_2[0].header)]
    else:
        binData_1_err = None
        binData_2_err = None

    for i in range(nummod):
        print('{}: modulator state {}:'.format(tag(), i))
        idx = fitsData.find_data_keyword_idx('VISP_YYY', i)
        for j in idx:
            print('{}:  extracting from data HDU {}'.format(tag(), j))
            data = fitsData.data_hdus[j].data
            tmp1 = data[:, :pivot]
            tmp2 = data[:, pivot:]

            x_binned_data1 = np.zeros((numpix_x, pivot))
            binned_data1 = np.zeros((numpix_x, numpix_y))
            x_binned_data2 = np.zeros((numpix_x, numy_all - pivot))
            binned_data2 = np.zeros((numpix_x, numpix_y))

            for k in range(xbin):
                x_binned_data1 += tmp1[k:xend:xbin, :]
                x_binned_data2 += tmp2[k:xend:xbin, :]
            for l in range(ybin_1):
                binned_data1 += x_binned_data1[:, l:yend_1:ybin_1]
            for l in range(ybin_2):
                binned_data2 += x_binned_data2[:, l:yend_2:ybin_2]

            ih_1 = pyfits.ImageHDU(np.asarray(binned_data1 / (xbin * ybin_1), dtype=np.float32),
                                   header=fitsData.data_hdus[j].header)
            ih_1.header['CAM__010'] = xbin
            ih_1.header['CAM__011'] = ybin_1
            ih_1.header['VISP_OGX'] = numx
            ih_1.header['VISP_OGY'] = pivot
            ih_1.header['VISP_BEM'] = 1
            hdulist_1.append(ih_1)

            ih_2 = pyfits.ImageHDU(np.asarray(binned_data2 / (xbin * ybin_2), dtype=np.float32),
                                   header=fitsData.data_hdus[j].header)
            ih_2.header['CAM__010'] = xbin
            ih_2.header['CAM__011'] = ybin_2
            ih_2.header['VISP_OGX'] = numx
            ih_2.header['VISP_OGY'] = numy_2
            ih_2.header['VISP_BEM'] = 2
            hdulist_2.append(ih_2)

            if errData is not None:
                print('{}:  extracting from uncertinaty HDU {}'.format(tag(), j))
                err = errData.data_hdus[j].data
                tmp1_err = err[:, :pivot]
                tmp2_err = err[:, pivot:]

                x_binned_data1_err = np.zeros((numpix_x, pivot))
                binned_data1_err = np.zeros((numpix_x, numpix_y))
                x_binned_data2_err = np.zeros((numpix_x, numy_all - pivot))
                binned_data2_err = np.zeros((numpix_x, numpix_y))

                for k in range(xbin):
                    x_binned_data1_err += tmp1_err[k:xend:xbin, :]
                    x_binned_data2_err += tmp2_err[k:xend:xbin, :]
                for l in range(ybin_1):
                    binned_data1_err += x_binned_data1_err[:, l:yend_1:ybin_1]
                for l in range(ybin_2):
                    binned_data2_err += x_binned_data2_err[:, l:yend_2:ybin_2]

                ih_1_err = pyfits.ImageHDU(np.asarray(binned_data1_err / (xbin * ybin_1), dtype=np.float32),
                                           header=hdulist_1[j].header)
                hdulist_1_err.append(ih_1_err)

                ih_2_err = pyfits.ImageHDU(np.asarray(binned_data2_err / (xbin * ybin_2), dtype=np.float32),
                                           header=hdulist_2[j].header)
                hdulist_2_err.append(ih_2_err)

    binData_1 = Data.FitsData()
    binData_1.origin = 'extract macropixels'
    binData_1.hdulist = hdulist_1
    binData_2 = Data.FitsData()
    binData_2.origin = 'extract macropixels'
    binData_2.hdulist = hdulist_2

    if errData is not None:
        binData_1_err = Data.FitsData()
        binData_1_err.origin = 'extract macropixels'
        binData_1_err.hdulist = hdulist_1_err
        binData_2_err = Data.FitsData()
        binData_2_err.origin = 'extract macropixels'
        binData_2_err.hdulist = hdulist_2_err

    return binData_1, binData_2, binData_1_err, binData_2_err
    


    [docs]def upsample_demodulation(dmodHDU):
    """Take a single demodulation matrix computed for binned data and up-sample it to the full ViSP frame.

    Up-sampling is done using nearest-neighbor interpolation in such a way to ensure that each macropixel (i.e. bin)
    is equally represented in the final output matrices. The exception to this is when the number of macropixels was
    not an integer divisor of the full frame. In this case the macropixel on the edge of each beam will be slightly
    (typically less than 5 px) over represented in the final matrices.

    All information on the shape of the full frame is pulled from the FITS headers, so they'd better be correct!

    Parameters
    ----------
    dmodHDU : `astropy.io.fits.hdu.image.ImageHDU`
        HDU containing data for a single beam's demodulation matrix. Should have data with shape (x, y, 4, M).

    Returns
    -------
    numpy.ndarray
        The input data upsampled to match the full size of the corresponding ViSP beam.

    Notes
    -----
    This function will work exactly as expected if x = y = 1, but in this case you're probably better off just
    squeezing out the extra dimensions and taking advantage of the (mostly) shape-agnostic input of
    ScienceCalibration.polarimetric_multiplication
    """
    full_x = dmodHDU.header['VISP_OGX']
    full_y = dmodHDU.header['VISP_OGY']
    nums, numm = dmodHDU.data.shape[-2:]
    full_output = np.zeros((full_x, full_y, nums, numm), dtype=np.float32)
    print('{}: original shape of beam: ({}, {}, {}, {})'.format(tag(), full_x, full_y, nums, numm))

    numx, numy, nums, numm = dmodHDU.data.shape
    print('{}:  demodulation matrix shape: ({}, {}, {}, {})'.format(tag(), numx, numy, nums, numm))

    x = np.arange(full_x // numx / 2, full_x, full_x // numx)
    y = np.arange(full_y // numy / 2, full_y, full_y // numy)

    print('{}:  interpolating'.format(tag()))

    xi, yi = np.mgrid[0:full_x:full_x * 1j, 0:full_y:full_y * 1j]
    for i in range(nums):
        print('{}:  stokes parameter {}'.format(tag(), i))
        for j in range(numm):
            print('{}:   modulator state {}'.format(tag(), j))
            full_output[:, :, i, j] = spi.interpn((x, y), dmodHDU.data[:, :, i, j], (xi, yi), bounds_error=False,
                                                  fill_value=None, method='nearest')

    return full_output
    


    [docs]def run_PAC_demod(run_dir, telescope_db='telescope_db.txt', threads=1, overwrite=False):
    """Take a ViSP reduced set of Calibration Sequence data and compute a set of demodulation matrices.

    The best demodulation is found by varying the parameters of the Calibration Unit (CU). See the PACPipeline for more
    details.

    Parameters
    ----------
    run_dir : str
        Location of the processed, macropixel'd ViSP data associated with a single Calibration Sequence (CS).

    telescope_db : str
        Location of the telescope parameter database

    threads : int
        Maximum number of cpus to use when fitting CU parameters. No more than 2 cpus will ever be used.

    overwrite : bool
        If False then demodulation matrices will not be computed if they already exist.

    Returns
    -------
    Data.DemodulationData
        Object containing the demodulation matrices for both ViSP beams
    """
    from PAC_Pipeline.PACPipeline import main_demodulation_matrix

    jobs = []

    pac_kwargs = {'overwrite': overwrite,
                  'save_stats': True,
                  'telescope_db': telescope_db}
    for i in range(1, 3):
        beam_dir = '{}/beam{}'.format(run_dir, i)
        beam_output = '{}/DMOD_beam{}.fits'.format(run_dir, i)
        jobs.append(mp.Process(target=main_demodulation_matrix,
                               args=[beam_dir, beam_output],
                               kwargs=pac_kwargs,
                               name='Beam{}'.format(i)))

    i = 0
    while i < len(jobs):
        while [x.is_alive() for x in jobs].count(True) < threads:
            jobs[i].start()
            i += 1
            if i >= len(jobs):
                break
        time.sleep(1)

    for p in jobs:
        p.join()

    raw_dmod = Data.DemodulationData()
    raw_dmod.load_beam_from_file('{}/DMOD_beam1.fits'.format(run_dir), 1)
    raw_dmod.load_beam_from_file('{}/DMOD_beam2.fits'.format(run_dir), 2)

    return raw_dmod
    



    [docs]def generate_instrument_pol_calibration(dmodData, output):
    """Read raw demodulation matrices and prep them for use in Science data reduction

    The demodulation matrices for each ViSP beam are treated separately and if either one was a full-frame average
    then the spatial dimensions are simply squeezed out. Otherwise the data are passed to `upsample_demodulation` for
    interpolation to the full beam size.

    Parameters
    ----------
    dmodData : Data.DemodulationData
        The demodulation matrix for both ViSP beams. Should have a single data HDU for each beam with shape
        (x, y, 4, M) where (x, y) is the shape of the binned ViSP beam 1 frame passed to PA&C and M is the number of
        unique modulator states.

    Returns
    -------
    Data.DemodulationData
        Object containing a data HDU for each ViSP beam. The data shape of these HDUs will either be (X, Y, 4, M) or
        (4, M), where X and Y are the full read-out size of a single ViSP beam.
    """
    print('{}: extracting demoulation matrices'.format(tag()))
    numx1, numy1, _, numm1 = dmodData.beam1.data.shape
    numx2, numy2, _, numm2 = dmodData.beam2.data.shape

    if numm1 != numm2:
        raise ValueError("The demodulation matrices don't have the same number of modulator states")
    if dmodData.beam1.header['VISP_OGX'] != dmodData.beam2.header['VISP_OGX']:
        raise ValueError('Beams 1 and 2 not not have the same original spatial dimension')

    final_dmod = Data.DemodulationData()
    for i, (numx, numy, hdu) in enumerate(zip([numx1, numx2], [numy1, numy2], [dmodData.beam1, dmodData.beam2])):

        if numx == numy == 1:
            print('{}: Beam {} is a whole beam, passing through'.format(tag(), i + 1))
            final_dmod.init_beam_with_ndarray(hdu.data[0, 0, :, :], i + 1, header=hdu.header)
        else:
            print('{}: Upsampling beam {}'.format(tag(), i + 1))
            data = upsample_demodulation(hdu)
            final_dmod.init_beam_with_ndarray(data, i + 1, header=hdu.header)

    final_dmod.write_out(output, overwrite=True)

    return final_dmod
    



    [docs]def find_number_of_CS_steps(dirname):
    """Discover the number of Calibration Sequence steps present in a SoCC directory

    Parameters
    ----------
    dirname : str
        Path to a directory containing a single SoCC

    Returns
    -------
    int
        The number of Calibration Sequence steps in the SoCC
    """
    step_list = []
    file_list = glob('{}/*.fits'.format(dirname))

    for f in file_list:
        h = pyfits.open(f)
        try:
            if h[0].header['DKIST004'] == 'PolCal':
                step_list.append(h[0].header['PAC__999'])
        except KeyError:
            pass

    unique = np.unique(step_list)

    if not np.all(np.sort(unique) == np.arange(unique.size)):
        raise ValueError('SoCC appears to be missing some CS steps')

    return unique.size
