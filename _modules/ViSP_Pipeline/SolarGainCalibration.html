    [docs]def main(
    solar_dir: str,
    outpre: str,
    geo_out: str,
    dark_cal: Optional[Data.FitsData] = None,
    lamp_gain_cal: Optional[Data.FitsData] = None,
    dark_cal_err: Optional[Data.FitsData] = None,
    lamp_cal_err: Optional[Data.FitsData] = None,
    debug: bool = False,
    gen_err: bool = True,
) -> Tuple[Data.FitsData, Optional[Data.FitsData], Data.FitsData]:
    """Run a raw frame through the Solar Gain Calibration pipeline.

    If the associated Calibration objects are provided then the raw data will be dark corrected and have the lamp
    gain removed prior to further analysis. These steps are not strictly required but are highly recommended.

    The result of this pipeline is a Solar Gain Calibration object suitable for flattening the response of the ViSP
    detector. As a side effect a Geometric Calibration object is produced. This object is used to remove the
    distorting effects of the ViSP optics from the data.

    Parameters
    ----------
    solar_dir : str
        Location of the directory containing raw solar gain data.

    outpre : str
        Prefix of the location to save processed data. Can include the full path. The processed data will be saved as
        outpre.fits and associated uncertainties (if provided) will be saved as outpre_err.fits.

    geo_out : str
        The location/filename to save the Geometric Calibration object. This is NOT a prefix.

    dark_cal : Data.FitsData
        A Dark Calibration object. Probably the result of DarkCalibration.main().

    lamp_gain_cal : data.FitsData
        A Lamp Gain Calibration object. Probably the result of LampGainCalibration.main().


    dark_cal_err : Data.FitsData or None
        Uncertainty assocaited with the Dark Calibration object.


    lamp_cal_err : Data.FitsData or None
        Uncertainty associated with the Lamp Gain Calibration object.

    gen_err : bool
        If True then pixel uncertainties will be generated from the raw data

    Returns
    -------
    Data.FitsData
        Processed Solar Gain Calibration.

    Data.FitsData
        The uncertainty associated with the Solar Gain Calibration.

    Data.FitsData
        The Geometric Calibration data.

    """
    gainData = Data.SolarGainData(solar_dir)

    if gen_err:
        errData = generic.generate_noise_frame(gainData)
    else:
        errData = None

    validate_gain_task_data(gainData, CONSTANTS["SolarGainCalibration"]["min_solargain_frames"])

    print("{}: averaging data".format(tag()))
    avgData, avgErr = generic.average_FitsData(
        gainData, errData=errData, preserve_modulator=True, delete_source=True
    )
    if debug:
        print("{}: DEBUG: Writing avg image to SGd_avg.FITS".format(tag()))
        avgData.write_out("SGd_avg.FITS", overwrite=True)

    if dark_cal:
        print("{}: doing Dark Correction".format(tag()))
        generic.dark_correction(avgData, dark_cal, avgErr, dark_cal_err)
        if debug:
            print("{}: DEBUG: Writing dark subtracted image to SGd_ds.FITS".format(tag()))
            avgData.write_out("SGd_ds.FITS", overwrite=True)

    # Save the data before (potentially, hopefully) doing lamp correction
    guaranteed_no_lamp_avgData = avgData.copy()
    if avgErr is not None:
        guaranteed_no_lamp_avgErr = avgErr.copy()
    else:
        guaranteed_no_lamp_avgErr = avgErr

    if debug:
        print("{}: DEBUG: Writing guaranteed no lamp data to SGd_gnlc.FITS".format(tag()))
        guaranteed_no_lamp_avgData.write_out("SGd_gnlc.FITS", overwrite=True)

    print("{}: finding beam geometry".format(tag()))
    angles, beam_offset, state_offset = find_beam_geometry(avgData, debug=debug)

    if lamp_gain_cal is not None:
        print("{}: removing lamp gain signal".format(tag()))
        generic.gain_correction(avgData, lamp_gain_cal, avgErr, lamp_cal_err)
        if debug:
            print("{}: DEBUG: Writing lamp corrected image to SGd_lc.FITS".format(tag()))
            avgData.write_out("SGd_lc.FITS", overwrite=True)

    print("{}: extracting un-distorted beams".format(tag()))
    beam1, beam2, beam1_err, beam2_err = get_matching_beams(
        avgData, angles, beam_offset, state_offset, errData=avgErr
    )

    print("{}: finding spectral tilt/skew".format(tag()))
    spec_shifts = find_spec_geometry(
        beam1, beam2, max_shift=CONSTANTS["SolarGainCalibration"]["max_spec_shift_px"], debug=debug
    )

    print("{}: removing spectral tilt/skew".format(tag()))
    perfect_beams, perfect_beams_err = remove_spec_geometry(
        beam1, beam2, spec_shifts, beam1_err, beam2_err
    )

    print("{}: computing average spectrum".format(tag()))
    avg_spec_1d, avg_spec_1d_err = average_spectrum(
        perfect_beams,
        perfect_beams_err,
        window=CONSTANTS["SolarGainCalibration"]["spectral_avg_window"],
        hairline_fraction=CONSTANTS["SolarGainCalibration"]["hairline_fraction"],
    )

    print("{}: generating distorted average spectra".format(tag()))
    wave = int(avgData.headers[0][S122["Wavelength"]])
    if wave not in CONSTANTS["SolarGainCalibration"]["zone_prominence_by_wave"].keys():
        wave = "default"
    z_prominence = CONSTANTS["SolarGainCalibration"]["zone_prominence_by_wave"][wave]
    z_width = CONSTANTS["SolarGainCalibration"]["zone_width_by_wave"][wave]
    z_bg_order = CONSTANTS["SolarGainCalibration"]["zone_bg_order_by_wave"][wave]
    z_norm_frac = CONSTANTS["SolarGainCalibration"]["zone_norm_frac_by_wave"][wave]
    print(
        "{}: using zone parameters for {}: ({}, {}, {}, {})".format(
            tag(), wave, z_prominence, z_width, z_bg_order, z_norm_frac
        )
    )
    avg_spec, avg_spec_err = generate_average_spectra(
        avg_spec_1d,
        avg_spec_1d_err,
        beam1,
        beam2,
        spec_shifts,
        angles,
        beam_offset,
        state_offset,
        CONSTANTS["Data"]["beam_border"],
        zone_prominence=z_prominence,
        zone_width=z_width,
        zone_bg_order=z_bg_order,
        zone_normalization_percentile=z_norm_frac,
        debug=debug,
    )
    if debug:
        avg_FD = Data.FitsData()
        avg_FD.init_with_ndarray(avg_spec)
        print("{}: DEBUG: Writing 2d avg agin to SGd_avg2d.FITS".format(tag()))
        avg_FD.write_out("SGd_avg2d.FITS", overwrite=True)
        del avg_FD

    print("{}: removing average spectra from solar gain".format(tag()))
    remove_average_spectra(
        guaranteed_no_lamp_avgData, avg_spec, guaranteed_no_lamp_avgErr, avg_spec_err
    )
    if debug:
        print("{}: DEBUG: Writing removed spectra to SGd_gnlc_rem.FITS".format(tag()))
        guaranteed_no_lamp_avgData.write_out("SGd_gnlc_rem.FITS", overwrite=True)

    validate_solar_gain_calibration(
        guaranteed_no_lamp_avgErr, CONSTANTS["SolarGainCalibration"]["max_solargain_noise"]
    )

    print("{}: generating Solar Gain Calibration".format(tag()))
    solar_gain_calibration, solar_gain_calibration_err = generate_solar_gain_calibration(
        guaranteed_no_lamp_avgData, guaranteed_no_lamp_avgErr, outpre
    )

    print("{}: generating Geometric Calibration".format(tag()))
    geometric_calibration = generate_geometric_calibration(
        angles,
        beam_offset,
        state_offset,
        spec_shifts,
        geo_out,
        avgData.headers[0],
        avgData.headers[1],
    )

    return solar_gain_calibration, solar_gain_calibration_err, geometric_calibration
    



    [docs]def validate_gain_task_data(gainData: Data.FitsData, min_gain_frames: int) -> None:
    """
    Confirm that the gain data form a consistent set, there are enough gain frames, and are from a Gain task.

    If any of these conditions are not met, an error is raised.

    Parameters
    ----------
    gainData : Data.FitsData
        The raw frame data object to validate

    min_gain_frames : int
        The minimum number of gain frames required to produce a valid SolarGain Calibration object

    """
    if (
        gainData.headers[0][S122["InstrumentProgramTask"]]
        not in TASK_DATA_DICTS["SolarGain"]["InstrumentProgramTask"]
    ):
        raise ValueError(
            "Instrument task is not Gain. (is {})".format(
                gainData.headers[0][S122["InstrumentProgramTask"]]
            )
        )

    if len(gainData.data) < min_gain_frames:
        raise ValueError("Not enough solar gain frames to compute acceptable Calibration Object")

    IP0 = gainData.headers[1][S122["InstrumentProgramID"]]
    for h in gainData.data_hdus:
        if h.header[S122["InstrumentProgramID"]] != IP0:
            raise ValueError("Not all solar gain frames have the same IP ID")

    return
    



    [docs]def validate_solar_gain_calibration(avgErr: Optional[Data.FitsData], max_noise: float) -> None:
    """
    Confirm that the Gain Calibration object does not have any out-of-range values.

    If problems are detected an error is raised.

    Parameters
    ----------
    avgErr : Data.FitsData or None
        Uncertainties associated with avgData

    max_noise : float
        The maximum allowable value in the uncertainty frame.

    """
    if avgErr is None:
        return

    for i in range(len(avgErr.data)):
        if np.mean(avgErr.data[i]) > max_noise:
            raise ValueError("SolarGain Calibration Object noise is too high")

    return
    



    [docs]def find_beam_geometry(
    fitsData: Data.FitsData,
    numotsu: int = 5,
    numtheta: int = 1500,
    peak_width: int = 4,
    debug: bool = False,
) -> Tuple[np.ndarray, np.ndarray, np.ndarray]:
    """Compute the spectral angle, beam offset, and modulator state offset for a set of spectra.

    Because the beam splitter doesn't impart a rotation the angle should be the same for both beams.

    The beam offset is the offset required to match the mirrored beam2 to beam1. This last part is important so I'll
    say it again: the beam offset is the shift between beam1 and beam2[:,::-1]

    The modulator state offset gives the offset requires to match all modulator states to the mean location across
    all states (for each beam).

    Given the above, to match beam1 and beam2 from modulator state i the transform is:

    beam1 -> rotate(shift(beam1, -modulator_offset[i]), -angle[0])
    beam2 -> rotate(shift(beam2, -modulator_offset[i] - beam_offset), -angle[1])

    Parameters
    ----------
    fitsData : Data.FitsData
        The spectra to be used. Should probably contain multiple HDU's, each corresponding to a different modulator
        state

    numotsu : int
        The number of Otsu thresholding iterations to make when finding the slit hairlines. This is passed directly
        to TargetCalibration.find_beam_angles()

    numtheta : int
        The number of theta samples to use when construction the Hough transform used to find the slit hairlines.

    peak_width : int
        The +/- window width (in pixels) to use when fitting the peak of the beam correlation function

    debug : bool
        If True then plots of the correlation function will be produced for the gross beam matching and the modulator
        state matching. The user will have to press enter between each match.

    Returns
    -------
    numpy.ndarray
        An array of length 2 containing the angle of the slit hairlines relative to the spectral image axis

    numpy.ndarray
        An array of length 2 containing the (x, y) shift between beam2[:,::-1] and beam1

    numpy.ndarray
        An array of shape (N, 2, 2) where N is the number of modulator states. The second dimension corresponds to
        the two beams and the last dimension contains the (x, y) shift between the beam and the mean location of the
        beam across all modulator states.

    """
    print("{}: Finding relative angle and position between beams".format(tag()))
    pivot = CONSTANTS["Data"]["beam_border"]
    beam0_1 = fitsData.data[0][:pivot, :]
    beam0_2 = fitsData.data[0][pivot:, :][::-1, :]

    angles = find_beam_angles(beam0_1, beam0_2, numotsu=numotsu, numtheta=numtheta, debug=debug)
    zero_slug = np.array([0, 0])

    beam0_1_ac = correct_geometry(beam0_1, zero_slug, angles[0])
    beam0_2_ac = correct_geometry(beam0_2, zero_slug, angles[1])

    beam_offset = find_beam_offset(beam0_1_ac, beam0_2_ac, width=peak_width, debug=debug)
    if debug:
        input("{}: Just finished matching the beams, press enter to continue".format(tag()))

    # Now find the relative offset between all of the modulator states
    nummod = fitsData.headers[0][S122["NumberOfModulatorStates"]]
    state_offset = np.zeros((nummod, 2, 2))
    for i in range(nummod):
        print("{}: Finding offset between HDU 0 and HDU {}".format(tag(), i))
        beam1 = fitsData.data[i][:pivot, :]
        beam2 = fitsData.data[i][pivot:, :][::-1, :]

        beam1_ac = correct_geometry(beam1, zero_slug, angles[0])
        beam2_ac = correct_geometry(beam2, zero_slug, angles[1])

        state_offset[i, 0, :] = find_beam_offset(
            beam0_1_ac, beam1_ac, width=peak_width, debug=debug
        )
        if debug:
            input("{}: Just finished HDU {} beam 1, press enter to continue".format(tag(), i))
        state_offset[i, 1, :] = find_beam_offset(
            beam0_2_ac, beam2_ac, width=peak_width, debug=debug
        )
        if debug:
            input("{}: Just finished HDU {} beam 2, press enter to continue".format(tag(), i))

    # Remove the mean so that all states will be shifted to a common mean
    state_offset -= np.mean(state_offset, axis=0)

    return angles, beam_offset, state_offset
    



    [docs]def find_beam_angles(
    beam1: np.ndarray,
    beam2: np.ndarray,
    numotsu: int = 5,
    numtheta: int = 1500,
    theta_min: float = -np.pi / 90,
    theta_max: float = np.pi / 90,
    debug: bool = False,
) -> np.ndarray:
    """For both beams, find the angle between the slit hairlines and the spectral image axis

    Spectral image axis refers to the pixel grid

    Parameters
    ----------
    beam1 : numpy.ndarray
        Beam1 data

    beam2 : numpy.ndarray
        Beam2 data, mirrored

    numotsu : int
        The number of Otsu thresholding iterations to perform

    numtheta : int
        The number of theta samples to use when constructing the Hough transform

    Returns
    -------
    numpy.ndarray
        A length 2 array containing the hairline angle for each beam. They should probably be the same.

    """
    angles = []

    for beam in [beam1, beam2]:
        binary = TargetCalibration.make_binary(beam, numotsu=numotsu, debug=debug)
        H, t, r = TargetCalibration.do_hough(
            binary, theta_min, theta_max, numtheta=numtheta, debug=debug
        )
        peak_theta, _ = TargetCalibration.find_px_angles(H, t, debug=debug)
        angles.append(peak_theta)
        if debug:
            input("{}: Just finished the next beam press enter to continue".format(tag()))

    return -1 * np.squeeze(np.array(angles))
    



    [docs]def find_beam_offset(
    beam1: np.ndarray, beam2: np.ndarray, width: int = 4, debug: bool = False
) -> np.ndarray:
    """Find the (x, y) shift between 2 beams.

    The shift is found by fitting the peak of the correlation of the two beams

    Parameters
    ----------
    beam1 : numpy.ndarray
        Beam 1 data

    beam2 : numpy.ndarray
        Beam 2 data, mirrored

    width : int
        The width (in pixels) of the window used to fit the peak of the correlation will. This parameter is actually
        half of the width because it specifies the radius of the fitting window.

    debug : bool
        If True plots of the correlation in x and y, along with the interpolated peaks, are saved to "corr_peaks.pdf"
        in the current directory.

    Returns
    -------
    numpy.ndarray
        The (x, y) shift between beam1 and beam2

    """
    sig1 = beam1 - np.nanmean(beam1)
    sig2 = beam2 - np.nanmean(beam2)

    sig2 = ske.match_histograms(sig2, sig1)

    corr = ss.fftconvolve(sig1, sig2[::-1, ::-1], mode="same")

    xidx, yidx = np.unravel_index(np.argmax(corr), corr.shape)
    xpeak = (
        pku.interpolate(np.arange(corr.shape[0]), np.sum(corr, axis=1), ind=[xidx], width=width)[0]
        - corr.shape[0] // 2
    )
    ypeak = (
        pku.interpolate(np.arange(corr.shape[1]), np.sum(corr, axis=0), ind=[yidx], width=width)[0]
        - corr.shape[1] // 2
    )

    if np.abs(xpeak - (xidx - corr.shape[0] // 2)) > width:
        xpeak = xidx - corr.shape[0] // 2

    if np.abs(ypeak - (yidx - corr.shape[1] // 2)) > width:
        ypeak = yidx - corr.shape[1] // 2

    print(
        "{}: peak of correlation at ({}, {})".format(
            tag(), xidx - corr.shape[0] // 2, yidx - corr.shape[1] // 2
        )
    )
    print("{:}: interpolated shift = ({:5.3f}, {:5.3f})".format(tag(), xpeak, ypeak))

    if debug and MPLT:
        pyfits.PrimaryHDU(corr).writeto("corr.fits", overwrite=True)
        pyfits.PrimaryHDU(sig1).writeto("sig1.fits", overwrite=True)
        pyfits.PrimaryHDU(sig2).writeto("sig2.fits", overwrite=True)
        x = np.arange(corr.shape[0]) - corr.shape[0] // 2
        y = np.arange(corr.shape[1]) - corr.shape[1] // 2
        pxidx = np.where(np.abs(x - xidx + corr.shape[0] // 2) < 100)
        pyidx = np.where(np.abs(y - yidx + corr.shape[1] // 2) < 100)
        fig = plt.figure()
        xax = fig.add_subplot(211)
        xax.set_title("X corr")
        xax.plot(x[pxidx], np.sum(corr, axis=1)[pxidx])
        xax.plot([xidx - corr.shape[0] // 2], [np.sum(corr, axis=1)[xidx]], "r.")
        xax.axvline(x=xpeak, ls=":", color="g", lw=0.8)
        xax.axvline(x=xidx - width - corr.shape[0] // 2, ls="--", color="r", lw=0.4)
        xax.axvline(x=xidx + width - corr.shape[0] // 2, ls="--", color="r", lw=0.4)
        yax = fig.add_subplot(212)
        yax.set_title("Y corr")
        yax.plot(y[pyidx], np.sum(corr, axis=0)[pyidx])
        yax.plot([yidx - corr.shape[1] // 2], [np.sum(corr, axis=0)[yidx]], "r.")
        yax.axvline(x=ypeak, ls=":", color="g", lw=0.8)
        yax.axvline(x=yidx - width - corr.shape[1] // 2, ls="--", color="r", lw=0.4)
        yax.axvline(x=yidx + width - corr.shape[1] // 2, ls="--", color="r", lw=0.4)
        fig.savefig("corr_peaks.pdf")

    # Multiply by -1 so that the output is the shift needed to move from "perfect" to the current state.
    #  In other words, applying a shift equal to the negative of the output of this function will undo the measured
    #  shift.
    return -1 * np.array([xpeak, ypeak])
    



    [docs]def correct_geometry(data: np.ndarray, shift: np.ndarray, angle: float) -> np.ndarray:
    """A simple function to shift and rotate data. It applies the inverse of the given shift and angle."""

    deg_angle = 180.0 * angle / np.pi
    data[np.where(data == np.inf)] = np.max(data[np.isfinite(data)])
    data[np.where(data == -np.inf)] = np.min(data[np.isfinite(data)])
    data[np.isnan(data)] = np.nanmedian(data)

    return spnd.interpolation.shift(
        spnd.interpolation.rotate(data, -deg_angle, mode="nearest", reshape=False),
        -shift,
        mode="reflect",
    )
    



    [docs]def get_matching_beams(
    fitsData: Data.FitsData,
    angles: np.ndarray,
    beam_offset: np.ndarray,
    state_offset: np.ndarray,
    errData: Optional[Data.FitsData] = None,
    clean_data: bool = False,
) -> Tuple[np.ndarray, np.ndarray, np.ndarray, np.ndarray]:
    """Extract beam 1 and 2 and transform them so they share the same coordinates.

    In other words beam_1[i,j] and beam_2[i,j] will correspond to the same _physical_ location (i.e., field position
    before entering ViSP optics). This is done by first mirroring beam 2 and then applying the appropriate rotations
    and shifts.

    Each HDU (i.e., modulator state) in the input data will added to the third dimension of the output data.

    Parameters
    ----------
    fitsData : Data.FitsData
        The distorted (raw) input spectra

    angles : numpy.ndarray
        The hairline angles for the two beams

    beam_offset : numpy.ndarray
        The (x,y) offsets that match the mirrored beam 2 to beam 1

    state_offset : numpy.ndarray
        Shape (N,2,2) where N is the number of modulator states. The 2nd dimension corresponds to beam 1 or beam 2
        and the last dimension contains the (c,y) shift needed to shift each modulator to a global mean.

    errData : Data.FitsData
        Uncertainty assocaited with fitsData

    Returns
    -------
    numpy.ndarray
        A 3D array containing the data of the un-distorted beam 1 at N modulator states

    numpy.ndarray
        A 3D array containing the data of the un-distorted beam 2 at N modulator states

    numpy.ndarray or None
        A 3D array containing the uncertainty for the un-distorted beam 1 at N modulator states

    numpy.ndarray or None
        A 3D array containing the uncertainty for the un-distorted beam 2 at N modulator states

    """
    pivot = CONSTANTS["Data"]["beam_border"]
    nummod = fitsData.headers[0][S122["NumberOfModulatorStates"]]
    numx, numy = fitsData.data[0].shape
    beam1 = np.zeros((pivot, numy, nummod), dtype=np.float32)
    beam2 = np.zeros((numx - pivot, numy, nummod), dtype=np.float32)
    if errData is not None:
        beam1_err = np.zeros((pivot, numy, nummod), dtype=np.float32)
        beam2_err = np.zeros((numx - pivot, numy, nummod), dtype=np.float32)
    else:
        beam1_err = None
        beam2_err = None

    for i in range(nummod):
        print("{}: removing beam distortions from data in HDU {}".format(tag(), i))
        data = fitsData.data[i]
        tmp1 = data[:pivot, :]
        tmp2 = data[pivot:, :][::-1, :]
        if clean_data:
            tmp1 = clean_salt(tmp1.astype(np.float64))
            tmp2 = clean_salt(tmp2.astype(np.float64))
        beam1[:, :, i] = correct_geometry(tmp1, state_offset[i, 0], angles[0])
        beam2[:, :, i] = correct_geometry(tmp2, state_offset[i, 1] + beam_offset, angles[1])
        if errData is not None:
            print("{}: removing beam distortions from uncertainty in HDU {}".format(tag(), i))
            err = fitsData.data[i] ** 2
            tmp1_err = err[:pivot, :]
            tmp2_err = err[pivot:, :][::-1, :]
            beam1_err[:, :, i] = np.sqrt(correct_geometry(tmp1_err, state_offset[i, 0], angles[0]))
            beam2_err[:, :, i] = np.sqrt(
                correct_geometry(tmp2_err, state_offset[i, 1] + beam_offset, angles[1])
            )

    return beam1, beam2, beam1_err, beam2_err
    



    [docs]def find_spec_geometry(
    all_beam1: np.ndarray,
    all_beam2: np.ndarray,
    peak_width: int = 4,
    max_shift: float = 40.0,
    poly_fit_order: int = 3,
    debug: bool = False,
) -> np.ndarray:
    """Compute a function that will shift all slit positions to be on the same wavelength grid.

    The resulting function can be used to remove any spectral tilt or shift from the data.

    Shifts are found by correlating each spatial pixel with the spectrum of the slit center. To improve the signal
    all modulator states are averaged together prior to correlation.

    Parameters
    ----------
    all_beam1 : numpy.ndarray
        A 3D array containing the un-distorted beam 1 data at N modulator states

    all_beam2 : numpy.ndarray
        A 3D array containing the un-distorted beam 2 data at N modulator states

    peak_width : int
        The +/- width used to fit the correlation peaks

    max_shift : float
        The maximum spectral shift (in pixels) allowed. Any spatial pixels with larger shifts will just assume the shift
        from the last spectral pixel measured.

    poly_fit_order : int
        Order of the polynomial to use when fitting spectral curvature

    debug : bool
        If True a plot of the computed shifts is produced for each beam and saved to "spec_shifts.pdf".
        The input beams will also be saved as "beam[1|2].fits".

    Returns
    -------
    numpy.ndarray
        Array with shape (2, X), where X is the number of pixels in the spatial dimension. The first dimension
        corresponds to the two beams and the second dimension gives the shift in the spectral dimension needed to
        bring each slit position to the mean location of all positions.

    """
    numy = all_beam1.shape[1]
    avg_shifts = np.zeros((numy, 2))

    print("{}: Finding spectral shifts".format(tag()))
    # TODO: Maybe a refspec for each mod state?
    beam1 = np.nanmean(all_beam1, axis=2)
    beam2 = np.nanmean(all_beam2, axis=2)
    if debug:
        pyfits.PrimaryHDU(beam1).writeto("beam1.fits", overwrite=True)
        pyfits.PrimaryHDU(beam2).writeto("beam2.fits", overwrite=True)

    ref_spec_1 = beam1[:, numy // 2]
    ref_spec_2 = beam2[:, numy // 2]

    if debug and MPLT:
        fig = plt.figure()
        ax1 = fig.add_subplot(211)
        ax1.set_title("Beam 1")
        ax2 = fig.add_subplot(212)
        ax2.set_title("Beam 2")

    for j in range(numy):
        spec1 = beam1[:, j]
        spec2 = beam2[:, j]

        corr_1 = np.correlate(
            spec1 - np.nanmean(spec1), ref_spec_1 - np.nanmean(ref_spec_1), mode="same"
        )
        corr_2 = np.correlate(
            spec2 - np.nanmean(spec2), ref_spec_2 - np.nanmean(ref_spec_2), mode="same"
        )

        idx_1 = pku.indexes(
            corr_1, min_dist=corr_1.size
        )  # Only find a single peak in each correlation signal
        idx_2 = pku.indexes(corr_2, min_dist=corr_2.size)

        initial_guess_1 = 1 * (idx_1 - corr_1.size // 2)
        initial_guess_2 = 1 * (idx_2 - corr_2.size // 2)

        shift_1 = spo.minimize(
            shift_chisq,
            np.array([float(initial_guess_1)]),
            args=(ref_spec_1, spec1),
            method="nelder-mead",
        ).x[0]
        if np.abs(shift_1) > max_shift:
            # Didn't find a good peak, probably because of a hairline
            print(
                "{}: shift in beam 1 at spatial pixel {} out of range ({})".format(
                    tag(), j, shift_1
                )
            )
            print("{}:   using the shift from the previous pixel".format(tag()))
            shift_1 = avg_shifts[j - 1, 0]

        shift_2 = spo.minimize(
            shift_chisq,
            np.array([float(initial_guess_2)]),
            args=(ref_spec_2, spec2),
            method="nelder-mead",
        ).x[0]
        if np.abs(shift_2) > max_shift:
            print(
                "{}: shift in beam 2 at spatial pixel {} out of range ({})".format(
                    tag(), j, shift_2
                )
            )
            print("{}:   using the shift from the previous pixel".format(tag()))
            shift_2 = avg_shifts[j - 1, 1]

        avg_shifts[j, 0] = shift_1
        avg_shifts[j, 1] = shift_2

        if debug and MPLT:
            ax1.plot(spec1, ls="-", color="k", alpha=0.1, lw=0.5)
            ax2.plot(spec2, ls="-", color="k", alpha=0.1, lw=0.5)

    avg_shifts -= np.mean(avg_shifts, axis=0)

    poly_1 = np.poly1d(np.polyfit(np.arange(numy), avg_shifts[:, 0], poly_fit_order))
    poly_2 = np.poly1d(np.polyfit(np.arange(numy), avg_shifts[:, 1], poly_fit_order))
    fit_shifts = np.stack((poly_1(np.arange(numy)), poly_2(np.arange(numy))), 1)

    if debug and MPLT:
        ax1.plot(ref_spec_1, ls="-", color="r")
        ax2.plot(ref_spec_2, ls="-", color="r")
        fig.savefig("specs.png", dpi=300)

        x = np.arange(numy)
        fig = plt.figure()
        ax1 = fig.add_subplot(211)
        ax1.plot(x, avg_shifts[:, 0], ".", ms=0.8)
        ax1.plot(x, fit_shifts[:, 0], "-r")
        ax2 = fig.add_subplot(212)
        ax2.plot(x, avg_shifts[:, 1], ".", ms=1)
        ax2.plot(x, fit_shifts[:, 1], "-r")
        ax2.set_xlabel("Spatial pixel")
        ax1.set_ylabel("Spectral shift [px]")
        # ax1.set_ylim(-2, 2)
        # ax2.set_ylim(-2, 2)
        ax1.set_title("Beam 1")
        ax2.set_title("Beam 2")
        fig.savefig("spec_shifts.pdf")
        plt.close(fig)

    return avg_shifts.T

    """Remove the spectral tilt/skew found in find_spec_geometry().

    This is a pretty simple function that simply undoes the computed spectral shifts.

    The output contains both beams in the same array to simplify the calculation of the average spectrum, but this is
    just a convenience.

    Parameters
    ----------
    all_beam1 : numpy.ndarray
        A 3D array containing the data for the un-distorted beam 1 in N modulator states

    all_beam2 : numpy.ndarray
        A 3D array containing the data for the un-distorted beam 2 in N modulator states

    spec_shifts : numpy.ndarray
        Array with shape (2, X), where X is the number of pixels in the spatial dimension. The first dimension

    all_beam2_err : numpy.ndarray
        Uncertainty estimates for all_beam2

    numpy.ndarray

        Otherwise it is simply None.

    """

        all_beam1_err is None and all_beam2_err is not None
    ):
        raise Exception(
            "Uncertainty estimates must be supplied for either both beams or none of them"
        )

    nummod = all_beam1.shape[2]
    numy = all_beam1.shape[1]
    beam1_output = np.zeros((all_beam1.shape))
    beam2_output = np.zeros((all_beam2.shape))
    if all_beam1_err is not None:
        beam1_err_output = np.zeros(beam1_output.shape)
        beam2_err_output = np.zeros(beam2_output.shape)

    for i in range(nummod):
        print("{}: removing spectral tilt/skew from HDU {}".format(tag(), i))
        beam1 = all_beam1[:, :, i]
        beam2 = all_beam2[:, :, i]
        if all_beam1_err is not None:
            beam1_err = all_beam1_err[:, :, i]
            beam2_err = all_beam2_err[:, :, i]
        for j in range(numy):
            beam1_output[:, j, i] = shift_spec(-spec_shifts[0, j], beam1[:, j])
            beam2_output[:, j, i] = shift_spec(-spec_shifts[1, j], beam2[:, j])
            if all_beam1_err is not None:
                beam1_err_output[:, j, i] = np.sqrt(
                    spnd.interpolation.shift(
                        beam1_err[:, j] ** 2, -spec_shifts[0, j], mode="reflect"
                    )
                )
                beam2_err_output[:, j, i] = np.sqrt(
                    spnd.interpolation.shift(
                        beam2_err[:, j] ** 2, -spec_shifts[1, j], mode="reflect"
                    )
                )

    output = np.vstack((beam1_output, beam2_output))
    if all_beam1_err is not None:
        output_err = np.vstack((beam1_err_output, beam2_err_output))
    else:
        output_err = None

    return output, output_err
    



    [docs]def average_spectrum(
    spectra: np.ndarray,
    error: Optional[np.ndarray],
    window: int = 50,
    hairline_fraction: float = 0.8,
) -> Tuple[np.ndarray, Optional[np.ndarray]]:
    """Compute the spectral average across all spatial pixels.

    Optionally, compute the associated error spectrum.

    Modulator states are preserved, so the output has shape (X, Y, N) where Y is the number of pixels in the spectral
    dimension and N is the number of modulator states.
    """
    med_spec = np.nanmedian(spectra, axis=1)
    ratio = spectra / med_spec[:, None, :]
    lidx = np.where(ratio < hairline_fraction)
    tmp = spectra.copy()
    tmp[lidx] = med_spec[lidx[0], lidx[-1]]
    avg_spec = spnd.gaussian_filter1d(tmp, window, axis=1)
    del tmp

    if error is not None:
        avg_err = np.sqrt(spnd.gaussian_filter1d(error ** 2, window, axis=1))
    else:
        avg_err = None

    return avg_spec, avg_err
    



    [docs]def generate_average_spectra(
    avg_spec: np.ndarray,
    avg_err: Optional[np.ndarray],
    all_beam1: np.ndarray,
    all_beam2: np.ndarray,
    spec_shifts: np.ndarray,
    angles: np.ndarray,
    beam_offset: np.ndarray,
    state_offset: np.ndarray,
    pivot: int,
    zone_prominence: float = 0.2,
    zone_width: float = 2,
    zone_bg_order: int = 22,
    zone_normalization_percentile: int = 90,
    debug: bool = False,
) -> Tuple[np.ndarray, Optional[np.ndarray]]:
    """Take a single spectrum and produce spectra for all spatial positions, then distort these spectra to match the
    detector-space distortions.

    The final result is an average spectral image where output[i,j] corresponds to the raw detector location (i,j)

    Modulator states are preserved and are each given a different spectrum.

    Parameters
    ----------
    avg_spec : numpy.ndarray
        An array with shape (Y, N) where Y is the number of spatial pixels and N is the number of modulator states

    avg_err : numpy.ndarray or None
        Uncertainty associated with avg_spec.

    all_beam1 : numpy.ndarray
        A 3D array containing the data for the un-distorted beam 1 in N modulator states

    all_beam2 : numpy.ndarray
        A 3D array containing the data for the un-distorted beam 2 in N modulator states

    spec_shifts : numpy.ndarray
        Array with shape (2, X), where X is the number of pixels in the spatial dimension. The first dimension
        corresponds to the two beams and the second dimension is the spectral shift. Probably the output of
        find_spec_geometry()

    angles : numpy.ndarray
        A 2 element array containing the hairline angles of beam 1 and beam 2

    beam_offset : numpy.ndarray
        A 2 element array containing the (x, y) shift beteween beam 1 and the reflected beam 2

    state_offset : numpy.ndarray
        Shape (N,2,2) where N is the number of modulator states. The 2nd dimension corresponds to beam 1 or beam 2
        and the last dimension contains the (c,y) shift needed to shift each modulator to a global mean.

    pivot : int
        The spectral pixel that defines the boundary between beam 1 and beam 2. This is in the original, raw,
        detector coordinates.

    Returns
    -------
    numpy.ndarray
        An array of shape (X, Y, N) containing the normalized, distorted average spectrum for all spatial locations

    numpy.ndarray or None
        If avg_err was not None then this contains the uncertainty associated with the above. Otherwise None.

    """
    numy = spec_shifts.shape[1]
    numx, _, nummod = avg_spec.shape
    rot_angle_1 = 180 * angles[0] / np.pi
    rot_angle_2 = 180 * angles[1] / np.pi
    output = np.zeros((numx, numy, nummod))
    if avg_err is not None:
        output_err = np.zeros(output.shape)
    else:
        output_err = None

    means1 = np.nanmedian(all_beam1, axis=0)
    means2 = np.nanmedian(all_beam2, axis=0)

    for i in range(nummod):
        print("{}: creating average spectra for HDU {}".format(tag(), i))
        tmp1 = np.zeros((pivot, numy))
        tmp2 = np.zeros((numx - pivot, numy))
        if avg_err is not None:
            tmp1_err = np.zeros(tmp1.shape)
            tmp2_err = np.zeros(tmp2.shape)
        print("{}:  computing line zones".format(tag()))
        zones1 = compute_line_zones(
            avg_spec[:pivot, :, i],
            prominence=zone_prominence,
            width=zone_width,
            bg_order=zone_bg_order,
            normalization_percentile=zone_normalization_percentile,
            debug=1 if debug else False,
        )
        zones2 = compute_line_zones(
            avg_spec[pivot:, :, i],
            prominence=zone_prominence,
            width=zone_width,
            bg_order=zone_bg_order,
            normalization_percentile=zone_normalization_percentile,
            debug=2 if debug else False,
        )

        if len(zones1) == 0 or len(zones2) == 0:
            raise ValueError("No zones found! Please adjust the zone parameters and try again.")

        print("{}:  fitting reshift".format(tag()))
        for j in range(numy):
            ref_spec_1 = all_beam1[:, j, i]
            ref_spec_2 = all_beam2[:, j, i]
            spec1 = avg_spec[:pivot, j, i]
            spec2 = avg_spec[pivot:, j, i]
            shift_1 = reshift(spec1, ref_spec_1, zones=zones1, x_init=spec_shifts[0, j])
            shift_2 = reshift(spec2, ref_spec_2, zones=zones2, x_init=spec_shifts[1, j])
            tmp1[:, j] = shift_spec(shift_1, avg_spec[:pivot, j, i], mode="reflect")
            tmp2[:, j] = shift_spec(shift_2, avg_spec[pivot:, j, i], mode="reflect")
            if avg_err is not None:
                tmp1_err[:, j] = np.sqrt(
                    spnd.interpolation.shift(
                        avg_err[:pivot, i] ** 2, spec_shifts[0, j], mode="reflect"
                    )
                )
                tmp2_err[:, j] = np.sqrt(
                    spnd.interpolation.shift(
                        avg_err[pivot:, i] ** 2, spec_shifts[1, j], mode="reflect"
                    )
                )

        tmp1 *= (means1[:, i] / np.nanmedian(tmp1, axis=0))[None, :]
        tmp2 *= (means2[:, i] / np.nanmedian(tmp2, axis=0))[None, :]

        if debug:
            pyfits.PrimaryHDU(tmp1).writeto("SGd_beam1_perfectshift.FITS", overwrite=True)
            pyfits.PrimaryHDU(tmp2).writeto("SGd_beam2_perfectshift.FITS", overwrite=True)
            print(
                "{}: DEBUG: Wrote re-shifted avg spec to SGd_beam?_perfectshift.FIT".format(tag())
            )

        output[:pivot, :, i] = spnd.interpolation.shift(
            spnd.interpolation.rotate(tmp1, rot_angle_1, reshape=False, mode="nearest"),
            state_offset[i, 0],
            mode="reflect",
        )
        output[pivot:, :, i] = spnd.interpolation.shift(
            spnd.interpolation.rotate(tmp2, rot_angle_2, reshape=False, mode="nearest"),
            state_offset[i, 1] + beam_offset,
            mode="reflect",
        )[::-1, :]
        if avg_err is not None:
            print("{}: creating average uncertainty spectra for HDU {}".format(tag(), i))
            tmp1_err *= (means1[:, i] / np.mean(tmp1, axis=0))[None, :]
            tmp2_err *= (means2[:, i] / np.mean(tmp2, axis=0))[None, :]
            output_err[:pivot, :, i] = np.sqrt(
                spnd.interpolation.shift(
                    spnd.interpolation.rotate(
                        tmp1_err ** 2, rot_angle_1, reshape=False, mode="nearest"
                    ),
                    state_offset[i, 0],
                    mode="reflect",
                )
            )
            output_err[pivot:, :, i] = np.sqrt(
                spnd.interpolation.shift(
                    spnd.interpolation.rotate(
                        tmp2_err ** 2, rot_angle_2, reshape=False, mode="nearest"
                    ),
                    state_offset[i, 1] + beam_offset,
                    mode="reflect",
                )
            )[:, ::-1]

    return output, output_err

    [docs]def compute_line_zones(
    spec_2d: np.ndarray,
    prominence: float = 0.5,
    width: float = 5,
    bg_order: int = 12,
    normalization_percentile: int = 90,
    debug: int = 0,
) -> List[Tuple[int, int]]:
    """ Identify spectral regions around strong spectra features """
    avg_1d = np.mean(spec_2d, axis=1)
    em_spec = -1 * avg_1d + avg_1d.max()
    em_spec /= np.nanpercentile(em_spec, normalization_percentile)
    baseline = pku.baseline(em_spec, bg_order)
    em_spec -= baseline
    first_idx = sps.find_peaks(em_spec, prominence=prominence, width=width)[0]
    _, _, rips, lips = sps.peak_widths(em_spec, first_idx, rel_height=0.97)
    rips = np.floor(rips).astype(int)
    lips = np.ceil(lips).astype(int)

    ranges_to_remove = identify_overlapping_zones(rips, lips)
    rips = np.delete(rips, ranges_to_remove)
    lips = np.delete(lips, ranges_to_remove)

    if debug and MPLT:
        ax = plt.figure().add_subplot(111)
        ax1 = ax.twinx()
        ax.plot(avg_1d, "-k")
        ax1.plot(em_spec, "-b", alpha=0.3)
        for r, l in zip(rips, lips):
            ax.axvspan(r, l, alpha=0.2, color="r")
        for p in first_idx:
            ax.scatter([p], [avg_1d[p]])
        for p in first_idx[ranges_to_remove]:
            ax.scatter([p], [avg_1d[p]], color="r", marker="x", s=50)
        ax.figure.savefig("line_zones_{}.pdf".format(debug))

    return list(zip(rips, lips))

    for i in range(len(all_ranges)):
        target_range = all_ranges[i]
        for j in range(i + 1, len(all_ranges)):
            if (
                np.intersect1d(target_range, all_ranges[j]).size > 1
            ):  # Allow for a single overlap just to be nice
                if target_range.size > all_ranges[j].size:
                    ranges_to_remove.append(j)
                    print(
                        "{}: Zone ({}, {}) inside zone ({}, {})".format(
                            tag(),
                            all_ranges[j][0],
                            all_ranges[j][-1],
                            target_range[0],
                            target_range[-1],
                        )
                    )
                else:
                    ranges_to_remove.append(i)
                    print(
                        "{}: Zone ({}, {}) inside zone ({}, {})".format(
                            tag(),
                            target_range[0],
                            target_range[-1],
                            all_ranges[j][0],
                            all_ranges[j][-1],
                        )
                    )

    return ranges_to_remove
    



    [docs]def shift_chisq(par: np.ndarray, ref_spec: np.ndarray, spec: np.ndarray) -> float:
    """
    Goodness of fit calculation for a simple shift. Uses simple chisq as goodness of fit.
    Less robust than `reshift`, but waaaay faster
    """
    shift = par[0]
    shifted_spec = shift_spec(-shift, spec, mode="constant", cval=np.nan)
    chisq = np.nansum((ref_spec - shifted_spec) ** 2 / ref_spec)
    return chisq
    



    [docs]def reshift(
    spec: np.ndarray, ref_spec: np.ndarray, zones: List[Tuple[int, int]], x_init: float = 0.0
) -> float:
    """
    Compute the shift for a single spatial position back from rectified spectra to the OG (curved) position

    Line zones are used to increase the SNR of the chisq and the final shift is the mean of the shifts computed
    for each zone.

    Parameters
    ----------
    spec
        The 1D spectrum to shift back

    ref_spec
        The reference spectrum. This should be the un-shifted, raw spectrum at the same position as `spec`

    zones
        List of zone borders (in px coords)

    x_init
        Initial guess for the shift. This is used to shift the zones so it needs to be pretty good, but not perfect.

    Returns
    -------
    The shift value
    """
    shifts = np.zeros(len(zones))
    for i, z in enumerate(zones):
        if z[1] + int(x_init) >= spec.size:
            continue
        idx = np.arange(z[0], z[1]) + int(x_init)
        shift = spo.minimize(
            shift_func, [x_init], args=(ref_spec, spec, idx), method="nelder-mead"
        ).x[0]
        shifts[i] = shift

    return np.mean(shifts)
    



    [docs]def shift_func(par: List[float], ref_spec: np.ndarray, spec: np.ndarray, idx: np.ndarray) -> float:
    """
    Non-chisq based goodness of fit calculator for compute spectral shifts

    Instead of chisq, the metric approximates the final Gain image
    """
    shift = par[0]
    shifted_spec = shift_spec(shift, spec, mode="constant", cval=np.nan)
    final_gain = (ref_spec / shifted_spec)[idx]
    slope = (final_gain[-1] - final_gain[0]) / final_gain.size
    bg = slope * np.arange(final_gain.size) + final_gain[0]
    subbed_gain = np.abs((final_gain) - bg)
    fit_metric = np.nansum(subbed_gain[np.isfinite(subbed_gain)])
    return fit_metric

def command_line() -> None:
    import argparse
    import os
    import sys
    from ViSP_Pipeline.ViSPPipeline import ensure_fits, strip_fits

    parser = argparse.ArgumentParser(
        description="Run the ViSP Pipeline's Solar Gain Calibration module (i.e., generate a solar gain frame and "
        "a geometric calibration). "
        "This is a direct hook into SolarGainCalibration.main() and all keyword options are exposed as "
        "command line options."
    )
    parser.add_argument("solar_dir", help="Location of directory holding raw Solar Gain frames")
    parser.add_argument(
        "output",
        help="Name of FITS SolarGain output file. A {} suffix will be enforced".format(
            CONSTANTS["Data"]["suffix"]
        ),
    )
    parser.add_argument(
        "geo_output",
        help="Name of the FITS geometric cal output file. A {} suffix will be enforced".format(
            CONSTANTS["Data"]["suffix"]
        ),
    )
    parser.add_argument(
        "-D", "--dark-cal", help="Location of previously computed Dark Calibration FITS file"
    )
    parser.add_argument(
        "-d",
        "--dark-cal-err",
        help="FITS file containing errors on provided Dark. "
        "This doesn't do anything unless -E is also set.",
    )
    parser.add_argument(
        "-L", "--lamp-cal", help="Location of previously computed Lamp Gain Calibration FITS file"
    )
    parser.add_argument(
        "-l",
        "--lamp-cal-err",
        help="FITS file containing errors on provided Lamp Gain. "
        "This doesn't do anything unless -E is also set.",
    )
    parser.add_argument(
        "-f", "--force", help="Overwrite output if it already exists", action="store_true"
    )
    parser.add_argument(
        "-E", "--gen-err", help="Also generate a sigma uncertainty frame", action="store_true"
    )
    parser.add_argument("-K", "--debug", help="Activate debug mode", action="store_true")

    args = parser.parse_args()

    if not (os.path.exists(args.solar_dir) and os.path.isdir(args.solar_dir)):
        print("Input directory {} does not exist!".format(args.solar_dir))
        sys.exit(1)

    for f in [args.dark_cal, args.dark_cal_err, args.lamp_cal, args.lamp_cal_err]:
        if f is not None and not os.path.exists(f):
            print("File {} does not exist!".format(f))
            sys.exit(1)

    for output_file in [ensure_fits(args.output), ensure_fits(args.geo_output)]:
        if os.path.exists(output_file) and not args.force:
            print("Output file {} already exists. Use -f to override".format(output_file))
            sys.exit(1)

    func_args = [args.solar_dir, strip_fits(args.output), ensure_fits(args.geo_output)]
    kwargs = {
        "dark_cal": Data.FitsData(args.dark_cal) if args.dark_cal is not None else None,
        "dark_cal_err": Data.FitsData(ensure_fits(args.dark_cal_err))
        if args.dark_cal_err is not None
        else None,
        "lamp_gain_cal": Data.FitsData(args.lamp_cal) if args.lamp_cal is not None else None,
        "lamp_cal_err": Data.FitsData(ensure_fits(args.lamp_cal_err))
        if args.lamp_cal_err is not None
        else None,
        "debug": args.debug,
        "gen_err": args.gen_err,
    }

    print(
        "Calling SolarGainCalibration.main\n\twith args: {}\n\t and kwargs: {}".format(
            func_args, kwargs
        )
    )

    main(*func_args, **kwargs)
