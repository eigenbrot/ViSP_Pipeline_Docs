    datasetData: Data.DataSetData,
    slit_pos: int,
    outpre: str,
    dark_cal: Optional[Data.FitsData] = None,
    solar_gain_cal: Optional[Data.FitsData] = None,
    geometric_cal: Optional[Data.FitsData] = None,
    dark_cal_err: Optional[Data.FitsData] = None,
    solar_gain_cal_err: Optional[Data.FitsData] = None,
    gen_err: bool = True,
    dark_lock=NULLCONTEXT,
    gain_lock=NULLCONTEXT,
    geo_lock=NULLCONTEXT,
    pol_lock=NULLCONTEXT,
    pol: bool = False,
    demodulation: Optional[Data.DemodulationData] = None,
    telescope_db: Optional[str] = None,
    remove_tele: bool = True,
    include_parallactic: bool = True,
    save_M12_map: bool = True,
    skip_cal_verification: bool = False,
    M12_line_lims: List[int] = [0, -1],
    continuum_lims: List[int] = [0, -1],
) -> Tuple[Data.FitsData, Optional[Data.FitsData]]:
    """Run raw data through the Science Calibration pipeline.

    If no intermediate Calibration objects are provided then the pipeline simply averages the (potentially
    distorted!) beams. This is probably not what you want.

    Parameters
    ----------
    datasetData : `Data.DataSetData`
        Object containing file metadata for all files associated with the data set to be reduced

    slit_pos : int
        The slit (i.e. spatial) position to load. Files are matched to this parameter with the VISP_CSS header keyword

    outpre : str
        Prefix of the location to save processed data. Can include the full path. The processed data will be saved as
        outpre.fits and associated uncertainties (if provided) will be saved as outpre_err.fits.

    dark_cal : Data.FitsData
        A Dark Calibration object. Probably the result of DarkCalibration.main().

    solar_gain_cal : Data.FitsData
        A Solar Gain Calibration object. Probably the result of SolarGainCalibration.main().

    geometric_cal : Data.FitsData
        A Geometric Calibration object. Probably produced by SolarGainCalibration.main().

    dark_cal_err : Data.FitsData
        Uncertainty associated with dark_cal

    solar_gain_cal_err : Data.FitsData
        Uncertainty associated with solar_gain_cal

    gen_err : bool
        If True then pixel uncertainties will be generated from the raw data frames

    demodulation : Data.DemodulationData
        Object containing an array of shape (X, Y, 4, M) containing the demodulation matrix for each (x, y) pixel.
        This is probably the output of PolarimetricCalibration.upsample_demodulation.

    telescope_db : str
        Path to the database of telescope properties (x, \tau).

    dark_lock : multiprocessing.synchronize.lock
        This lock will be used when reading information from the dark calibration object. This avoids some weirdness in
        astropy's FITS containers when accessing the same data at the same time.

    gain_lock : multiprocessing.synchronize.lock
        This lock will be used when reading information from the gain calibration object. This avoids some weirdness in
        astropy's FITS containers when accessing the same data at the same time.

    geo_lock : multiprocessing.synchronize.lock
        This lock will be used when reading information from the geometric calibration object. This avoids some
        weirdness in astropy's FITS containers when accessing the same data at the same time.

    pol_lock : multiprocessing.synchronize.lock
        This lock will be used when reading information from the demodulation calibration object. This avoids some
        weirdness in astropy's FITS containers when accessing the same data at the same time.

    remove_tele : bool
        If True the inverse Telescope Model will be applied to the demodulated Stokes data

    include_parallactic : bool
        If True the data will be rotated into the DKIST common reference frame

    save_M12_map : bool
        If True a copy of the reduced data WITHOUT an inverse M12 correction will be saved for later use by the PA&C
        Modules. This copy will be truncated in wavelength to save space.

    skip_cal_verification : bool
        If True then none of the *Cal objects will be verified to be valid for use with the current science data.
        Use at your own risk!

    M12_line_lims : list
        If `save_M12_map = True` then this list defines the wavelength pixel range that contains a strong polarization
        signal (i.e., a spectral line).

    continuum_lims : list
        The wavelength pixel range that contains primarily continuum signal. These data are extracted to update a
        database of X12 mirror parameters.

    Returns
    -------
    Data.FitsData
        Level 1 ViSP Science Data with all of the provided Calibrations applied

    Data.FitsData
        If uncertainty on the input data is provided, the uncertainty associated with the Level 1 Science Data product.

    """
    # This list will hold objects we want to ensure are cleaned on exit
    to_clean_list = []

    print("{}: Loading ViSP slit position {} data from {}".format(tag(), slit_pos, datasetData))
    sciData = datasetData.get_slit_pos_data(slit_pos)

    if gen_err:
        sciErr = generic.generate_noise_frame(sciData)
    else:
        sciErr = None

    if dark_cal is not None:
        print("{}: doing Dark Correction".format(tag()))
        with dark_lock:
            print("{}: acquired dark MultiProcessing Lock".format(tag()))
            if not skip_cal_verification:
                verify_useful_dark(sciData, dark_cal)
            generic.dark_correction(sciData, dark_cal, sciErr, dark_cal_err)
        print("{}: released dark MultiProcessing Lock".format(tag()))

    if solar_gain_cal is not None:
        print("{}: doing Gain Correction with Solar Flat".format(tag()))
        with gain_lock:
            print("{}: acquired gain MultiProcessing Lock".format(tag()))
            if not skip_cal_verification:
                verify_useful_gain(sciData, solar_gain_cal)
            generic.gain_correction(sciData, solar_gain_cal, sciErr, solar_gain_cal_err)
        print("{}: released gain MultiProcessing Lock".format(tag()))

    # Because Geo and Pol corrections need only a single HDU per modulator state
    sciData, sciErr = generic.average_FitsData(sciData, sciErr, preserve_modulator=True)

    if geometric_cal is not None:
        print("{}: removing geometric/spectral distortions".format(tag()))
        with geo_lock:
            print("{}: acquired geo MultiProcessing Lock".format(tag()))
            if not skip_cal_verification:
                verify_useful_geometric(sciData, geometric_cal)
            geo_numbers = extract_geometric_numbers(geometric_cal)
        print("{}: released geo MultiProcessing Lock".format(tag()))
        sciData, sciErr = geometric_correction(sciData, geo_numbers, errData=sciErr)

    if pol:
        from ViSP_Pipeline import PolarimetricUtils

        if demodulation is None:
            raise Exception(
                "Polarimetric data requires a demodulation matrix and a telescope polaization calibration"
            )
        if (telescope_db is None or not os.path.exists(telescope_db)) and remove_tele:
            print("{}: Loading telescope parameter database from PA&C defaults".format(tag()))
            from PAC_Pipeline.generic import get_default_telescope_db

            telescope_db = get_default_telescope_db()

        print("{}: demodulating ViSP polarlization".format(tag()))
        with pol_lock:
            print("{}: acquired pol MultiProcessing Lock".format(tag()))
            if not skip_cal_verification:
                verify_useful_demodulation(sciData, demodulation)
        print("{}: released pol MultiProcessing Lock".format(tag()))
        (beam1Data, beam2Data), (beam1Err, beam2Err) = instrument_pol_correction(
            sciData, demodulation, sciErr
        )

        to_clean_list += [sciData, sciErr]

        print("{}: combining polarimetric beams".format(tag()))
        avgBeams, avgErr = average_polarimetric_beams(beam1Data, beam2Data, beam1Err, beam2Err)

        to_clean_list += [beam1Data, beam2Data, beam1Err, beam2Err]

        if save_M12_map:
            savedir = os.path.dirname(outpre) or "."
            M12dir = savedir + "/M12_map"
            if not os.path.exists(M12dir):
                os.makedirs(M12dir)

            print("{}: saving line data for PA&C M12 determination to {}".format(tag(), M12dir))
            M12outname = "{}/M12.{}.{}".format(
                M12dir, outpre.split(".")[1], CONSTANTS["Data"]["suffix"]
            )
            PolarimetricUtils.save_M12_map_frame(avgBeams, telescope_db, M12_line_lims, M12outname)

        if remove_tele:
            print("{}: removing telescope polarization matrix".format(tag()))
            teleCal = TelescopePolCalibration.generate_telescope_pol_calibration(
                avgBeams, telescope_db, M12=True, include_parallactic=include_parallactic
            )
            outData, outErr = telescope_pol_correction(avgBeams, teleCal, avgErr)

            to_clean_list += [avgBeams, avgErr]

        else:
            print("{}: skipping removal of telescope polarization matrix!".format(tag()))
            outData, outErr = avgBeams, avgErr
    else:
        print("{}: combining non-polarimetric beams".format(tag()))
        outData, outErr = average_beams(sciData, sciErr)

    print("{}: Generating level-1 science data".format(tag()))
    ViSPData, ViSPErr = generate_science_calibration(outData, outErr, outpre, pol)

    if pol and remove_tele:
        # The x-talk measurements will only be useful if the TM was applied
        print("{}: Reporting continuum cross-talk to PA&C Modules".format(tag()))
        try:
            PolarimetricUtils.report_continuum_xtalk(outData, continuum_lims)
        except Exception as e:
            print('{}: Could not report cross-talk. Error was: "{}"'.format(tag(), e))

    to_clean_list += [outData, outErr]
    for obj in to_clean_list:
        if obj is not None:
            obj._cleanup(True)

    return ViSPData, ViSPErr

    """
    Confirm that the provided DarkCalibration object can be used with the provided Science frame.

    This is done by checking the relevant headers.

    Parameters
    ----------
    sciData : Data.FitsData
        Science Data

    darkData : Data.FitsData
        Dark Calibration object
    """
    check_keys = VERIFICATION_KEYS["Dark"]
    dark_head = darkData.data_hdus[0].header
    for k in check_keys:
        for h in sciData.data_hdus:
            if h.header[k] != dark_head[k]:
                raise ValueError("Dark keyword {} does not match data".format(k))

    [docs]def verify_useful_demodulation(sciData: Data.FitsData, demodData: Data.FitsData) -> None:
    """
    Confirm that the provided demodulation matrices can be used with the provided Science frame.

    This is done by checking the relevant headers.

    Parameters
    ----------
    sciData : Data.FitsData
        Science Data

    demodData : Data.DemodulationData
        Demodulation matrices
    """
    check_keys = VERIFICATION_KEYS["Demodulation"]
    demod_head = demodData.headers[1]
    pivot = CONSTANTS["Data"]["beam_border"]
    for i, h in enumerate(sciData.data_hdus):
        for k in check_keys:
            if h.header[k] != demod_head[k]:
                raise ValueError(
                    "Demodulation keyword {} does not match data ({} != {})".format(
                        k, h.header[k], demod_head[k]
                    )
                )
        # Check if the demodulation matrices are full-frame or single averages
        # TODO: This does not for completely unprocessed science data because it still has a dummy dimension, which is technically allowed
        numx, numy = sciData.data[i].shape
        if len(demodData.beam1.shape) == 4:
            if numy != demodData.beam1.shape[1] or pivot != demodData.beam1.shape[0]:
                raise ValueError("Demodulation beam1 shape does not match data")
        if len(demodData.beam2.shape) == 4:
            if numy != demodData.beam2.shape[1] or (numx - pivot) != demodData.beam2.shape[0]:
                raise ValueError("Demodulation beam2 shape does not match data")
    



    [docs]def generate_science_calibration(
    sciData: Data.FitsData, sciErr: Optional[Data.FitsData], outpre: str, pol: bool = True
) -> Tuple[Data.FitsData, Optional[Data.FitsData]]:
    """Update header information to contain Stokes parameter names and data to disk

    Parameters
    ----------
    sciData : Data.FitsData
        Science data. Should have 4 ImageHDU's, corresponding to the 4 Stokes parameters [I, Q, U, V] IN THAT ORDER!

    sciErr : Data.FitsData or None
        Uncertainties associated with sciData

    outpre : str
        The filename (minus the .fits extension) to save the data to

    pol : bool
        If True then update the header with Stokes information. Otherwise just write the files to disk.

    Returns
    -------
    Data.FitsData
        Passthrough of sciData

    Data.FitsData
        Passthrough of sciErr

    """
    for obj, post in zip(
        [sciData, sciErr],
        ["." + CONSTANTS["Data"]["suffix"], "_err." + CONSTANTS["Data"]["suffix"]],
    ):
        if obj is not None:
            if pol:
                for i, name in enumerate(["I", "Q", "U", "V"]):
                    obj.data_hdus[i].header["VISP_STK"] = (name, "Stokes parameter")
                    obj.data_hdus[i].header["EXTNAME"] = "Stokes {}".format(name)
            outname = outpre + post
            obj.write_out(outname, overwrite=True)

    return sciData, sciErr
    



    [docs]def extract_geometric_numbers(
    geometric_cal: Data.FitsData,
) -> Tuple[np.ndarray, np.ndarray, np.ndarray, np.ndarray]:
    """Simply load the necessary geometric parameters from disk.

    This function exists because, in very rare cases, reduction threads for different slit positions were colliding
    when reading the Geometric Calibration object from its FITS file. The solution is to use a Multiprocessing Lock, but
    in order to still get the speedup from multiprocessing on the geometric correction (a time intensive step) we need
    to load the Geometric Calibration object separately under Lock and then release the Lock to actually do the
    correction.

    Parameters
    ----------
    geometric_cal : Data.FitsData
        Object containing the hairline angles, beam/modulator offsets, and spectral shifts of the distorted system

    Returns
    -------
    numpy.ndarray
         Shape (2,). The hairline angles for the two beams.

    numpy.ndarray
        Shape (2,). The (x, y) offset between beam 2 and beam 1.

    numpy.ndarray
        Shape (M, 2, 2). The first dimension corresponds to M modulator states, the second dimension corresponds to
        the two beams, and the last dimension contains the (x, y) offsets between each modulator state and the mean
        location of all modulator states for each beam.

    numpy.ndarray
        Shape (2, X), where X is the number of pixels in the spatial dimension. For each of the 2 beams, the spectral
        shift at each of the X spatial pixels.
    """
    print("{}: loading relevant values from {}".format(tag(), geometric_cal))
    angles = geometric_cal.hdulist[geometric_cal.find_keyword_idx("NAME", "ANGLES")[0]].data
    beam_offset = geometric_cal.hdulist[geometric_cal.find_keyword_idx("NAME", "BEAMOFF")[0]].data
    state_offset = geometric_cal.hdulist[geometric_cal.find_keyword_idx("NAME", "STATEOFF")[0]].data
    spec_shifts = geometric_cal.hdulist[geometric_cal.find_keyword_idx("NAME", "SPECSHFT")[0]].data

    return angles, beam_offset, state_offset, spec_shifts
    



    [docs]def geometric_correction(
    fitsData: Data.FitsData,
    geometric_numbers: Tuple[np.ndarray, np.ndarray, np.ndarray, np.ndarray],
    errData: Optional[Data.FitsData] = None,
) -> Tuple[Data.FitsData, Optional[Data.FitsData]]:
    """Remove hairline angles, beam/modulator state offsets, and spectral shifts from data

    Geometric distortion parameters are provided by the user and are probably generated by SolarGainCalibration.
    Note that the output data will have beam2 reflected so that it exactly matches the orientation of beam1

    Parameters
    ----------
    fitsData : Data.FitsData
        Data object to perform corrections on

    geometric_numbers : tuple
        Object containing the hairline angles, beam/modulator offsets, and spectral shifts of the distorted system

    errData : Data.FitsData
        Uncertainty associated with fitsData

    Returns
    -------
    Data.FitsData
        Un-distorted input data. The two beams are still in a single image, but beam2 is reflected to match the
        orientation of beam1.

    Data.FitsData or None
        If errData is provided, the uncertainty associated with the output data

    """
    angles, beam_offset, state_offset, spec_shifts = geometric_numbers

    # Here we basically copy SolarGainCalibration.get_matching_beams() line for line because we need to deal with the
    #  errors as well
    pivot = CONSTANTS["Data"]["beam_border"]
    nummod = fitsData.headers[0][S122["NumberOfModulatorStates"]]
    numx, numy = fitsData.data[0].shape
    beam1 = np.zeros((pivot, numy, nummod), dtype=np.float32)
    beam2 = np.zeros((numx - pivot, numy, nummod), dtype=np.float32)
    if errData is not None:
        beam1_err = np.zeros(beam1.shape, dtype=np.float32)
        beam2_err = np.zeros(beam2.shape, dtype=np.float32)
        header_list_err = []
    else:
        beam1_err = None
        beam2_err = None

    header_list = []
    for m in range(1, nummod + 1):
        idx = fitsData.find_data_keyword_idx(S122["CurrentState"], m)
        if idx.size > 1:
            raise Exception("Only modulator states with a single HDU are currently supported")
        print(
            "{}: removing distortions from modulator state {} (data HDU {})".format(
                tag(), m, idx[0]
            )
        )
        i = m - 1
        header_list.append(fitsData.data_hdus[idx[0]].header)
        data = fitsData.data[idx[0]]
        tmp1 = data[:pivot, :]
        tmp2 = data[pivot:, :][::-1, :]

        # Clean out any nasty values that will really mess up interpolation functions
        tmp1[np.where(tmp1 == np.inf)] = np.max(tmp1[np.isfinite(tmp1)])
        tmp1[np.where(tmp1 == -np.inf)] = np.min(tmp1[np.isfinite(tmp1)])
        tmp1[np.isnan(tmp1)] = np.nanmedian(tmp1)
        tmp2[np.where(tmp2 == np.inf)] = np.max(tmp2[np.isfinite(tmp2)])
        tmp2[np.where(tmp2 == -np.inf)] = np.min(tmp2[np.isfinite(tmp2)])
        tmp2[np.isnan(tmp2)] = np.nanmedian(tmp2)

        rot1 = -180 * angles[0] / np.pi
        rot2 = -180 * angles[1] / np.pi
        beam1[:, :, i] = spnd.interpolation.rotate(
            spnd.interpolation.shift(tmp1, -state_offset[i, 0], mode="reflect"),
            rot1,
            reshape=False,
            mode="nearest",
        )
        beam2[:, :, i] = spnd.interpolation.rotate(
            spnd.interpolation.shift(tmp2, -state_offset[i, 1] - beam_offset, mode="reflect"),
            rot2,
            mode="nearest",
            reshape=False,
        )

        print(
            "{}: removing spectral tilt/skew from modulator state {} (data HDU {})".format(
                tag(), m, idx[0]
            )
        )
        for j in range(numy):
            beam1[:, j, i] = spnd.interpolation.shift(
                beam1[:, j, i], -spec_shifts[0, j], mode="reflect"
            )
            beam2[:, j, i] = spnd.interpolation.shift(
                beam2[:, j, i], -spec_shifts[1, j], mode="reflect"
            )

        if errData is not None:
            print(
                "{}: removing distortions from err modulator state {} (data HDU {})".format(
                    tag(), i, idx[0]
                )
            )
            header_list_err.append(errData.data_hdus[idx[0]].header)
            err = errData.data[idx[0]]

            # We square these now and don't compute the sqrt until the last interpolation
            tmpe1 = err[:pivot, :] ** 2
            tmpe2 = err[pivot:, :][::-1, :] ** 2

            tmpe1[np.where(tmpe1 == np.inf)] = np.max(tmpe1[np.isfinite(tmpe1)])
            tmpe1[np.where(tmpe1 == -np.inf)] = np.min(tmpe1[np.isfinite(tmpe1)])
            tmpe1[np.isnan(tmpe1)] = np.nanmedian(tmpe1)
            tmpe2[np.where(tmpe2 == np.inf)] = np.max(tmpe2[np.isfinite(tmpe2)])
            tmpe2[np.where(tmpe2 == -np.inf)] = np.min(tmpe2[np.isfinite(tmpe2)])
            tmpe2[np.isnan(tmpe2)] = np.nanmedian(tmpe2)

            beam1_err[:, :, i] = spnd.interpolation.rotate(
                spnd.interpolation.shift(tmpe1, -state_offset[i, 0], mode="reflect"),
                rot1,
                mode="nearest",
                reshape=False,
            )
            beam2_err[:, :, i] = spnd.interpolation.rotate(
                spnd.interpolation.shift(tmpe2, -state_offset[i, 1] - beam_offset, mode="reflect"),
                rot2,
                mode="nearest",
                reshape=False,
            )

            print(
                "{}: removing spectral tilt/skew from err modulator state {} (data HDU {})".format(
                    tag(), i, idx[0]
                )
            )
            for j in range(numy):
                beam1_err[:, j, i] = np.sqrt(
                    spnd.interpolation.shift(beam1_err[:, j, i], -spec_shifts[0, j], mode="reflect")
                )
                beam2_err[:, j, i] = np.sqrt(
                    spnd.interpolation.shift(beam2_err[:, j, i], -spec_shifts[1, j], mode="reflect")
                )

    output = np.vstack((beam1, beam2))
    outData = Data.FitsData()
    outData.init_with_ndarray(output)
    outData.hdulist[0].header = fitsData.headers[0]
    for i in range(nummod):
        outData.hdulist[i + 1].header = header_list[i]

    outErr = None
    if errData is not None:
        out_err = np.vstack((beam1_err, beam2_err))
        outErr = Data.FitsData()
        outErr.init_with_ndarray(out_err)
        outErr.hdulist[0].header = errData.headers[0]
        for i in range(nummod):
            outErr.hdulist[i + 1].header = header_list_err[i]

    return outData, outErr
    



    [docs]def average_beams(
    fitsData: Data.FitsData,
    errData: Optional[Data.FitsData] = None,
    preserve_modulator: bool = False,
) -> Tuple[Data.FitsData, Optional[Data.FitsData]]:
    """Average the two beams together. Real simple.

    This function is modulator state aware, although I'm not sure why.

    Parameters
    ----------
    fitsData : Data.FitsData
        Data containing two beams across N modulator states

    errData : Data.FitsData
        Uncertainty on fitsData

    preserve_modulator : bool
        If True then only beams from the same modulator state will be averaged together. Otherwise, average all beams
        together.

    Returns
    -------
    Data.FitsData
        The average of both beams. Depending on the value of preserve_modulator there will either be a single HDU for
        each modulator state, or just one HDU containing the global average.

    Data.FitsData or None
        If errData is provided then this will be the uncertainty associated with the output

    """
    pivot = CONSTANTS["Data"]["beam_border"]
    nummod = fitsData.headers[0][S122["NumberOfModulatorStates"]]
    numx, numy = fitsData.data[0].shape
    newx = min(pivot, numx - pivot)
    hdulist = [fitsData.hdulist[0]]

    if errData is not None:
        hdulist_err = [errData.hdulist[0]]

    if preserve_modulator:
        for i in range(1, nummod + 1):

            if len(fitsData.find_data_keyword_idx(S122["CurrentState"], i)) > 1:
                raise Exception(
                    "Only single-HDU modulator states are supported. Modulator state {} has {} HDUs".format(
                        i, len(fitsData.find_data_keyword_idx(S122["CurrentState"], i))
                    )
                )

            beam1 = fitsData.data[i][:newx, :]
            beam2 = fitsData.data[i][pivot : pivot + newx, :]
            avg = (beam1 + beam2) * 0.5
            hdulist.append(pyfits.ImageHDU(avg, fitsData.data_hdus[i].header))

            if errData is not None:
                beam1_err = errData.data[i][:newx, :]
                beam2_err = errData.data[i][pivot : pivot + newx, :]
                avg_err = np.sqrt(beam1_err ** 2 + beam2_err ** 2) * 0.5
                hdulist_err.append(pyfits.ImageHDU(avg_err, errData.data_hdus[i].header))

    else:
        beam_sum = np.zeros((newx, numy), dtype=np.float32)
        for i in range(nummod):
            beam_sum += fitsData.data[i][:newx, :]
            beam_sum += fitsData.data[i][pivot : pivot + newx, :]

            if errData is not None:
                beam_sum_err = np.zeros(beam_sum.shape)
                beam_sum_err += errData.data[i][:newx, :] ** 2
                beam_sum_err += errData.data[i][pivot : pivot + newx, :] ** 2

        hdulist.append(pyfits.ImageHDU(beam_sum / (nummod * 2)))
        if errData is not None:
            hdulist_err.append(pyfits.ImageHDU(np.sqrt(beam_sum_err) / (nummod * 2)))

    avgData = Data.FitsData()
    avgData.hdulist = hdulist
    avgData.origin = "average_beams"
    avgErr = None

    if errData is not None:
        avgErr = Data.FitsData()
        avgErr.hdulist = hdulist_err
        avgErr.origin = "average_beams"

    return avgData, avgErr
    



    [docs]def instrument_pol_correction(
    fitsData: Data.FitsData,
    dmodData: Data.DemodulationData,
    errData: Optional[Data.FitsData] = None,
) -> Tuple[List[Data.FitsData], List[Optional[Data.FitsData]]]:
    """
    Split a ViSP frame into its two beams and apply a demodulation matrix to each beam.

    Parameters
    ----------
    fitsData : Data.FitsData
        The data to demodulate.

    dmodData : Data.DemodulationData
        The demodulation matrices to use. Should contain a data HDU for each of the ViSP beams.

    errData : Data.FitsData
        Uncertainty estimates associated with fitsData

    Returns
    -------
    Data.FitsData
        Demodulated beam 1 data

    Data.FitsData
        Demodulated beam 2 data

    Data.FitsData or None
        If errData is provided then this contains uncertainty estimates on the Demodulated beam 1 data

    Data.FitsData or None
        If errData is provided then this contains uncertainty estimates on the Demodulated beam 2 data
    """
    print("{}: splitting beams".format(tag()))
    stacked_data = da.dstack(fitsData.data)
    pivot = CONSTANTS["Data"]["beam_border"]

    beam1 = stacked_data[:pivot, :, :]
    beam2 = stacked_data[pivot:, :, :]

    if errData is not None:
        print("{}: splitting uncertainty beams".format(tag()))
        stacked_data_err = da.dstack(errData.data)
        beam1_err = stacked_data_err[:pivot, :, :]
        beam2_err = stacked_data_err[pivot:, :, :]
    else:
        beam1_err = None
        beam2_err = None

    outdata_lst = []
    outerr_lst = []
    for i, (data, dmod, err) in enumerate(
        zip(
            [beam1, beam2],
            [da.from_array(dmodData.beam1), da.from_array(dmodData.beam2)],
            [beam1_err, beam2_err],
        )
    ):

        print("{}: applying demodulation matrix to beam {}".format(tag(), i + 1))
        dmoded_data, dmoded_err = polarimetric_multiplication(data, dmod, err)

        if is_mp():
            print("{}: subprocess detected. Computing array synchronously".format(tag()))
            with dask.config.set(scheduler="synchronous"):
                dmoded_data = dmoded_data.compute()
        else:
            print("{}: computing array (potentially) asynchronously".format(tag()))
            dmoded_data = dmoded_data.compute()

        outData = Data.FitsData()
        outData.init_with_ndarray(dmoded_data)
        outData.hdulist[0].header = fitsData.hdulist[0].header
        outData.origin = "instrument_pol_correction"

        for i in range(1, len(outData.hdulist)):
            for c in fitsData.hdulist[0].header.cards:
                if c[0] not in ["SIMPLE", "BITPIX", "NAXIS", "NAXIS1", "NAXIS2", "EXTEND", "END"]:
                    outData.update_header(c[0], c[1:], i, True)

        outdata_lst.append(outData)

        outErr = None
        if dmoded_err is not None:
            if is_mp():
                print("{}: subprocess detected. Computing error array synchronously".format(tag()))
                with dask.config.set(scheduler="synchronous"):
                    dmoded_err = dmoded_err.compute()
            else:
                print("{}: computing error array (potentially) asynchronously".format(tag()))
                dmoded_err = dmoded_err.compute()

            outErr = Data.FitsData()
            outErr.init_with_ndarray(dmoded_err)
            outErr.hdulist[0].header = errData.hdulist[0].header
            errData.origin = "instrument_pol_correction"

            for i in range(1, len(outErr.hdulist)):
                for c in errData.hdulist[0].header.cards:
                    if c[0] not in [
                        "SIMPLE",
                        "BITPIX",
                        "NAXIS",
                        "NAXIS1",
                        "NAXIS2",
                        "EXTEND",
                        "END",
                    ]:
                        outErr.update_header(c[0], c[1:], i, True)

        outerr_lst.append(outErr)

    return outdata_lst, outerr_lst
    



    [docs]def telescope_pol_correction(
    fitsData: Data.FitsData, teleCal: np.ndarray, errData: Optional[Data.FitsData] = None
) -> Tuple[Data.FitsData, Optional[Data.FitsData]]:
    """Apply an inverse Telescope Mueller matrix and update header values

    The inverse TM captures all mirrors from M1 to M6 (inclusive)

    Parameters
    ----------
    fitsData : Data.FitsData
        Data to apply the inverse TM to. Should have 4 data HDUs (stokes vectors).

    teleCal : numpy.ndarray
        A (4, 4) array containing the Mueller matrix for the inverse of mirrors 1 - 6

    errData : Data.FitsData
        Uncertainty estimates on fitsData

    include_parallactic : bool
        If True the returned Mueller matrix will include rotation by the parallactic angle and will thus put
        the data into a common solar reference frame

    Returns
    -------
    Data.FitsData
        The input data with the polarimetric contributions of M1 - M6 removed

    Data.FitsData or None
        If uncertainties are provided then this will contain the uncertainty associated with the previous output.
    """
    data = fitsData.get_stacked_data()

    data_err = None
    if errData is not None:
        data_err = errData.get_stacked_data()

    output, out_err = polarimetric_multiplication(data, teleCal, data_err)

    outData = Data.FitsData()
    outData.init_with_ndarray(output)
    outData.hdulist[0].header = fitsData.hdulist[0].header
    outData.origin = "telescope_pol_correction"

    for i in range(1, len(outData.hdulist)):
        for c in fitsData.hdulist[0].header.cards:
            if c[0] not in ["SIMPLE", "BITPIX", "NAXIS", "NAXIS1", "NAXIS2", "EXTEND", "END"]:
                outData.update_header(c[0], c[1:], i, True)

    outErr = None
    if errData is not None:
        outErr = Data.FitsData()
        outErr.init_with_ndarray(out_err)
        outErr.hdulist[0].header = errData.hdulist[0].header
        errData.origin = "telescope_pol_correction"

        for i in range(1, len(outErr.hdulist)):
            for c in errData.hdulist[0].header.cards:
                if c[0] not in ["SIMPLE", "BITPIX", "NAXIS", "NAXIS1", "NAXIS2", "EXTEND", "END"]:
                    outErr.update_header(c[0], c[1:], i, True)

    return outData, outErr

    [docs]def polarimetric_multiplication(
    data: Union[np.ndarray, da.core.Array],
    demod: Union[np.ndarray, da.core.Array],
    err_data: Optional[Union[np.ndarray, da.core.Array]] = None,
    demod_err: Optional[Union[np.ndarray, da.core.Array]] = None,
) -> Tuple[Union[np.ndarray, da.core.Array], Union[np.ndarray, da.core.Array]]:
    """Matrix multiply the input data by a polarization demodulation matrix.

    This is a general-purpose function that requires good understanding of the input parameters. The important thing
    is that the demodulation matrix's last dimension is equal to the number of data HDU's in the input data.

    Parameters
    ----------
    fitsData : Data.FitsData
        Input data to demodulate. For the instrumental demodulation the number of data HDUs should correspond to the
        number of unique modulation states. For the telescope polarimetric calibration, there should be 4 data HDUs,
        corresponding to the 4 Stokes parameters.

    demod : numpy.ndarray
        The demodulation matrix. There are a few different shapes it can have:

            1. (x, y, 4, M), where (x, y) is the shape of a single input frame and M is the number of modulation
            states. In this case, a separate demodulation can be provided for each individual pixel. Total control!

            2. (4, M). This assumes a constant demodulation matrix across the ViSP chip

            3. (4, 4). This form is used for removing the polarizing effect of the telescope optics and requires the
            input data to already be in "Stokes" form.

        Note that in all cases the length of the second to last dimension is 4 (corresponding to the Stokes vector)
        and the length of the last dimension is equal to the number of data HDUs in the input data.

    err_data : Data.FitsData or None
        The uncertainty on the input data

    demod_err : numpy.ndarray
        Uncertainty on the input demodulation matrix

    Returns
    -------
    `numpy.ndarray`
        Demodulated/corrected polarization data. Regardless of the inputs, the output will have 4 data HDUs,
        corresponding to the 4 Stokes parameters.

    `numpy.ndarray`
        If errData is provided, the uncertainty associated with the output data.

    """
    print(
        "{}: matrix-multiplying arrays with shapes {} and {}".format(tag(), data.shape, demod.shape)
    )
    output = np.sum(demod * data[:, :, None, :], axis=3)

    output_err = None
    if err_data is not None:
        # if demod_err is None:
        #     raise Exception('Demodulation matrix uncertainty estimate not provided')
        data_err = err_data.get_stacked_data()

        output_err = np.sum(demod * data_err[:, :, None, :], axis=3)
        # output_err = np.sqrt(np.sum(demod**2 * data_err[:, :, None, :]**2, axis=3) +
        #                      np.sum(demod_err**2 * data[:, :, None, :]**2, axis=3))

    return output, output_err
    



    [docs]def average_polarimetric_beams(
    beam1Data: Data.FitsData,
    beam2Data: Data.FitsData,
    beam1Err: Optional[Data.FitsData] = None,
    beam2Err: Optional[Data.FitsData] = None,
) -> Tuple[Data.FitsData, Optional[Data.FitsData]]:
    """Average beams over each stokes parameter

    Basically the same as average_beams, but just different enough, instead of looking for modulator state (VISP_YYY)
    it looks for stokes parameter (VISP_STK)

    Parameters
    ----------
    beam1Data : Data.FitsData
        Stokes data for beam 1

    beam2Data : Data.FitsData
        Stokes data for beam 2

    beam1Err : Data.FitsData
        Uncertainty estimates for beam1Data

    beam2Err : Data.FitsData
        Uncertainty estimates for beam2Data

    Returns
    -------
    Data.FitsData
        The same as the input data, but where each stokes parameter contains the average of both beams instead of
        both beams separately

    Data.FitsData or None
        If errData is provided then this will be the uncertainty associated with the output
    """

    numx1 = beam1Data.data[0].shape[0]
    numx2 = beam2Data.data[0].shape[0]
    newx = min(numx1, numx2)
    hdulist = [beam1Data.hdulist[0]]

    if beam1Err is not None and beam2Err is not None:
        hdulist_err = [beam1Err.hdulist[0]]

    for i in range(4):

        avg = (beam1Data.data[i][:newx, :] + beam2Data.data[i][:newx, :]) * 0.5
        hdulist.append(pyfits.ImageHDU(avg, beam1Data.data_hdus[i].header))

        if beam1Err is not None and beam2Err is not None:
            avg_err = (
                np.sqrt(beam1Err.data[i][:newx, :] ** 2 + beam2Err.data[i][:newx, :] ** 2) * 0.5
            )
            hdulist_err.append(pyfits.ImageHDU(avg_err, beam1Err.data_hdus[i].header))

    avgData = Data.FitsData()
    avgData.hdulist = hdulist
    avgData.origin = "average_polarimetric_beams"
    avgErr = None

    if beam1Err is not None and beam2Err is not None:
        avgErr = Data.FitsData()
        avgErr.hdulist = hdulist_err
        avgErr.origin = "average_polarimetric_beams"

    return avgData, avgErr
