    [docs]def main(datasetData, slit_pos, outpre, dark_cal=None, solar_gain_cal=None, geometric_cal=None,
         dark_cal_err=None, solar_gain_cal_err=None, gen_err=True,
         dark_lock=None, gain_lock=None, geo_lock=None,
         demodulation=None, telescope_db=None, remove_tele=True):
    """Run raw data through the Science Calibration pipeline.

    If no intermediate Calibration objects are provided then the pipeline simply averages the (potentially
    distorted!) beams. This is probably not what you want.

    Parameters
    ----------
    datasetData : `Data.DataSetData`
        Object containing file metadata for all files associated with the data set to be reduced

    slit_pos : int
        The slit (i.e. spatial) position to load. Files are matched to this parameter with the VISP_CSS header keyword

    outpre : str
        Prefix of the location to save processed data. Can include the full path. The processed data will be saved as
        outpre.fits and associated uncertainties (if provided) will be saved as outpre_err.fits.

    dark_cal : Data.FitsData
        A Dark Calibration object. Probably the result of DarkCalibration.main().

    solar_gain_cal : Data.FitsData
        A Solar Gain Calibration object. Probably the result of SolarGainCalibration.main().

    geometric_cal : Data.FitsData
        A Geometric Calibration object. Probably produced by SolarGainCalibration.main().

    demodulation : Data.DemodulationData
        Object containing an array of shape (X, Y, 4, M) containing the demodulation matrix for each (x, y) pixel.
        This is probably the output of PolarimetricCalibration.upsample_demodulation.

    telescope_db : str
        Path to the database of telescope properties (x, \tau).

    dark_lock : multiprocessing.synchronize.lock
        This lock will be used when reading information from the dark calibration object. This avoids some weirdness in
        astropy's FITS containers when accessing the same data at the same time.

    gain_lock : multiprocessing.synchronize.lock
        This lock will be used when reading information from the gain calibration object. This avoids some weirdness in
        astropy's FITS containers when accessing the same data at the same time.

    geo_lock : multiprocessing.synchronize.lock
        This lock will be used when reading information from the geometric calibration object. This avoids some
        weirdness in astropy's FITS containers when accessing the same data at the same time.

    Returns
    -------
    Data.FitsData
        Level 1 ViSP Science Data with all of the provided Calibrations applied

    Data.FitsData
        If uncertainty on the input data is provided, the uncertainty associated with the Level 1 Science Data product.

    """
    print('{}: Loading ViSP slit position {} data from {}'.format(tag(), slit_pos, datasetData))
    sciData = datasetData.get_slit_pos_data(slit_pos)

    if gen_err:
        sciErr = generic.generate_noise_frame(sciData)
    else:
        sciErr = None

    try:
        if sciData.headers[0]['VISP_PMD'] == 'Full Stokes':
            print('{}: Polarimetric data detected'.format(tag()))
            pol = True
        elif sciData.headers[0]['VISP_PMD'] == 'Stokes-I':
            print('{}: Basic intensity data detected'.format(tag()))
            pol = False
    except KeyError:
        print('{}: Could not determine polarization mode. Assuming Stokes-I'.format(tag()))
        pol = False

    if dark_cal is not None:
        print('{}: doing Dark Correction'.format(tag()))
        with dark_lock:
            print('{}: acquired dark MultiProcessing Lock'.format(tag()))
            verify_useful_dark(sciData, dark_cal)
            generic.dark_correction(sciData, dark_cal, sciErr, dark_cal_err)
        print('{}: released dark MultiProcessing Lock'.format(tag()))

    if solar_gain_cal is not None:
        print('{}: doing Gain Correction with Solar Flat'.format(tag()))
        with gain_lock:
            print('{}: acquired gain MultiProcessing Lock'.format(tag()))
            verify_useful_gain(sciData, solar_gain_cal)
            generic.gain_correction(sciData, solar_gain_cal, sciErr, solar_gain_cal_err)
        print('{}: released gain MultiProcessing Lock'.format(tag()))

    if geometric_cal is not None:
        print('{}: removing geometric/spectral distortions'.format(tag()))
        with geo_lock:
            print('{}: acquired geo MultiProcessing Lock'.format(tag()))
            verify_useful_geometric(sciData, geometric_cal)
            geo_numbers = extract_geometric_numbers(geometric_cal)
        print('{}: released geo MultiProcessing Lock'.format(tag()))
        sciData, sciErr = geometric_correction(sciData, geo_numbers, errData=sciErr)

    if pol:
        if demodulation is None:
            raise Exception('Polarimetric data requires a demodulation matrix and a telescope polaization calibration')
        if telescope_db is None or not os.path.exists(telescope_db):
            raise Exception('Telescope database was either not provided or does not exist')

        print('{}: demodulating ViSP polarlization'.format(tag()))
        verify_useful_demodulation(sciData, demodulation)
        instData, instErr = polarimetric_demodulation(sciData, demodulation.data, sciErr)

        print('{}: combining polarimetric beams'.format(tag()))
        avgBeams, avgErr = average_polarimetric_beams(instData, instErr)

        if remove_tele:
            print('{}: removing telescope polarization matrix'.format(tag()))
            inv_tele = setup_telescope_model(avgBeams, telescope_db)
            outData, outErr = polarimetric_demodulation(avgBeams, inv_tele, avgErr)
        else:
            print('{}: skipping removal of telescope polarization matrix!'.format(tag()))
            outData, outErr = avgBeams, avgErr

    else:
        print('{}: combining non-polarimetric beams'.format(tag()))
        outData, outErr = average_beams(sciData, sciErr)

    print('{}: Generating level-1 science data'.format(tag()))
    ViSPData, ViSPErr = generate_science_calibration(outData, outErr, outpre, pol)

    return ViSPData, ViSPErr
    


    [docs]def verify_useful_dark(sciData, darkData):
    """
    Confirm that the provided DarkCalibration object can be used with the provided Science frame.

    This is done by checking the relevant headers.

    Parameters
    ----------
    sciData : Data.FitsData
        Science Data

    darkData : Data.FitsData
        Dark Calibration object
    """
    check_keys = ['NAXIS1',
                  'NAXIS2',
                  'CAM__001',  # CameraUniqueID
                  'CAM__002',  # FPAExposureTime
                  'CAM__003',  # FPAFrameRate
                  'CAM__014',  # AccumulationsPerFPA
                  'CAM__008',  # Hardware
                  'CAM__009',  #   binning
                  'CAM__010',  # Software
                  'CAM__011',  #   binning
                  'VISP_016', # GratingID
                  'VISP_019', # GratingAngle
                  'VISP_002'] # ArmAngle

    # Add ROI information parametrically because there are a ton of these keywords
    check_keys += ['CAM__{:03}'.format(i) for i in range(15,31)]

    dark_head = darkData.data_hdus[0].header
    for k in check_keys:
        for h in sciData.data_hdus:
            if h.header[k] != dark_head[k]:
                raise ValueError('Dark keyword {} does not match data'.format(k))

    [docs]def generate_science_calibration(sciData, sciErr, outpre, pol=True):
    """Update header information to contain Stokes parameter names and data to disk

    Parameters
    ----------
    sciData : Data.FitsData
        Science data. Should have 4 ImageHDU's, corresponding to the 4 Stokes parameters [I, Q, U, V] IN THAT ORDER!

    sciErr : Data.FitsData or None
        Uncertainties associated with sciData

    outpre : str
        The filename (minus the .fits extension) to save the data to

    pol : bool
        If True then update the header with Stokes information. Otherwise just write the files to disk.

    Returns
    -------
    Data.FitsData
        Passthrough of sciData

    Data.FitsData
        Passthrough of sciErr

    """
    for obj, post in zip([sciData, sciErr], ['.fits', '_err.fits']):
        if obj is not None:
            if pol:
                for i, name in enumerate(['I', 'Q', 'U', 'V']):
                    obj.data_hdus[i].header['VISP_STK'] = (name, 'Stokes parameter')
            outname = outpre + post
            obj.write_out(outname, overwrite=True)

    return sciData, sciErr
    


    [docs]def extract_geometric_numbers(geometric_cal):
    """Simply load the necessary geometric parameters from disk.

    This function exists because, in very rare cases, reduction threads for different slit positions were colliding
    when reading the Geometric Calibration object from its FITS file. The solution is to use a Multiprocessing Lock, but
    in order to still get the speedup from multiprocessing on the geometric correction (a time intensive step) we need
    to load the Geometric Calibration object separately under Lock and then release the Lock to actually do the
    correction.

    Parameters
    ----------
    geometric_cal : Data.FitsData
        Object containing the hairline angles, beam/modulator offsets, and spectral shifts of the distorted system

    Returns
    -------
    numpy.ndarray
         Shape (2,). The hairline angles for the two beams.

    numpy.ndarray
        Shape (2,). The (x, y) offset between beam 2 and beam 1.

    numpy.ndarray
        Shape (M, 2, 2). The first dimension corresponds to M modulator states, the second dimension corresponds to
        the two beams, and the last dimension contains the (x, y) offsets between each modulator state and the mean
        location of all modulator states for each beam.

    numpy.ndarray
        Shape (2, X), where X is the number of pixels in the spatial dimension. For each of the 2 beams, the spectral
        shift at each of the X spatial pixels.
    """
    print('{}: loading relevant values from {}'.format(tag(), geometric_cal))
    angles = geometric_cal.hdulist[geometric_cal.find_keyword_idx('NAME', 'ANGLES')[0]].data
    beam_offset = geometric_cal.hdulist[geometric_cal.find_keyword_idx('NAME', 'BEAMOFF')[0]].data
    state_offset = geometric_cal.hdulist[geometric_cal.find_keyword_idx('NAME', 'STATEOFF')[0]].data
    spec_shifts = geometric_cal.hdulist[geometric_cal.find_keyword_idx('NAME', 'SPECSHFT')[0]].data

    return angles, beam_offset, state_offset, spec_shifts
    


    [docs]def geometric_correction(fitsData, geometric_numbers, errData=None):
    """Remove hairline angles, beam/modulator state offsets, and spectral shifts from data

    Geometric distortion parameters are provided by the user and are probably generated by SolarGainCalibration.
    Note that the output data will have beam2 reflected so that it exactly matches the orientation of beam1

    Parameters
    ----------
    fitsData : Data.FitsData
        Data object to perform corrections on

    geometric_numbers : tuple
        Object containing the hairline angles, beam/modulator offsets, and spectral shifts of the distorted system

    errData : Data.FitsData
        Uncertainty associated with fitsData

    Returns
    -------
    Data.FitsData
        Un-distorted input data. The two beams are still in a single image, but beam2 is reflected to match the
        orientation of beam1.

    Data.FitsData or None
        If errData is provided, the uncertainty associated with the output data

    """
    angles, beam_offset, state_offset, spec_shifts = geometric_numbers

    # Here we basically copy SolarGainCalibration.get_matching_beams() line for line because we need to deal with the
    #  errors as well
    pivot = fitsData.headers[0]['VISP_PPP']
    nummod = fitsData.headers[0]['VISP_XXX']
    numx, numy = fitsData.data_hdus[0].data.shape
    beam1 = np.zeros((numx, pivot, nummod), dtype=np.float32)
    beam2 = np.zeros((numx, numy - pivot, nummod), dtype=np.float32)
    if errData is not None:
        beam1_err = np.zeros(beam1.shape, dtype=np.float32)
        beam2_err = np.zeros(beam2.shape, dtype=np.float32)
        header_list_err = []
    else:
        beam1_err = None
        beam2_err = None

    header_list = []
    for i in range(nummod):
        idx = fitsData.find_data_keyword_idx('VISP_YYY', i)
        if idx.size > 1:
            raise Exception('Only modulator states with a single HDU are currently supported')
        print('{}: removing distortions from modulator state {} (data HDU {})'.format(tag(), i, idx[0]))
        header_list.append(fitsData.data_hdus[idx[0]].header)
        data = fitsData.data_hdus[idx[0]].data
        tmp1 = data[:, :pivot]
        tmp2 = data[:, pivot:][:, ::-1]

        # Clean out any nasty values that will really mess up interpolation functions
        tmp1[np.where(tmp1 == np.inf)] = np.max(tmp1[np.isfinite(tmp1)])
        tmp1[np.where(tmp1 == -np.inf)] = np.min(tmp1[np.isfinite(tmp1)])
        tmp1[np.isnan(tmp1)] = np.nanmedian(tmp1)
        tmp2[np.where(tmp2 == np.inf)] = np.max(tmp2[np.isfinite(tmp2)])
        tmp2[np.where(tmp2 == -np.inf)] = np.min(tmp2[np.isfinite(tmp2)])
        tmp2[np.isnan(tmp2)] = np.nanmedian(tmp2)

        rot1 = -180 * angles[0] / np.pi
        rot2 = -180 * angles[1] / np.pi
        beam1[:, :, i] = spnd.interpolation.rotate(spnd.interpolation.shift(tmp1, -state_offset[i, 0], mode='reflect'),
                                                   rot1, reshape=False, mode='nearest')
        beam2[:, :, i] = spnd.interpolation.rotate(spnd.interpolation.shift(tmp2, -state_offset[i, 1] - beam_offset,
                                                                            mode='reflect'),
                                                   rot2, mode='nearest', reshape=False)

        print('{}: removing spectral tilt/skew from modulator state {} (data HDU {})'.format(tag(), i, idx[0]))
        for j in range(numx):
            beam1[j, :, i] = spnd.interpolation.shift(beam1[j, :, i], -spec_shifts[0, j], mode='reflect')
            beam2[j, :, i] = spnd.interpolation.shift(beam2[j, :, i], -spec_shifts[1, j], mode='reflect')

        if errData is not None:
            print('{}: removing distortions from err modulator state {} (data HDU {})'.format(tag(), i, idx[0]))
            header_list_err.append(errData.data_hdus[idx[0]].header)
            err = errData.data_hdus[idx[0]].data

            # We square these now and don't compute the sqrt until the last interpolation
            tmpe1 = err[:, :pivot]**2
            tmpe2 = err[:, pivot:][:, ::-1]**2

            tmpe1[np.where(tmpe1 == np.inf)] = np.max(tmpe1[np.isfinite(tmpe1)])
            tmpe1[np.where(tmpe1 == -np.inf)] = np.min(tmpe1[np.isfinite(tmpe1)])
            tmpe1[np.isnan(tmpe1)] = np.nanmedian(tmpe1)
            tmpe2[np.where(tmpe2 == np.inf)] = np.max(tmpe2[np.isfinite(tmpe2)])
            tmpe2[np.where(tmpe2 == -np.inf)] = np.min(tmpe2[np.isfinite(tmpe2)])
            tmpe2[np.isnan(tmpe2)] = np.nanmedian(tmpe2)

            beam1_err[:, :, i] = spnd.interpolation.rotate(spnd.interpolation.shift(
                                                                tmpe1, -state_offset[i, 0], mode='reflect'),
                                                           rot1, mode='nearest', reshape=False)
            beam2_err[:, :, i] = spnd.interpolation.rotate(spnd.interpolation.shift(
                                                                tmpe2, -state_offset[i, 1] - beam_offset,
                                                                mode='reflect'),
                                                           rot2, mode='nearest', reshape=False)

            print('{}: removing spectral tilt/skew from err modulator state {} (data HDU {})'.format(tag(), i, idx[0]))
            for j in range(numx):
                beam1_err[j, :, i] = np.sqrt(
                        spnd.interpolation.shift(beam1_err[j, :, i], -spec_shifts[0, j], mode='reflect'))
                beam2_err[j, :, i] = np.sqrt(
                        spnd.interpolation.shift(beam2_err[j, :, i], -spec_shifts[1, j], mode='reflect'))

    output = np.hstack((beam1, beam2))
    outData = Data.FitsData()
    outData.init_with_ndarray(output)
    outData.hdulist[0].header = fitsData.headers[0]
    for i in range(nummod):
        outData.hdulist[i + 1].header = header_list[i]

    outErr = None
    if errData is not None:
        out_err = np.hstack((beam1_err, beam2_err))
        outErr = Data.FitsData()
        outErr.init_with_ndarray(out_err)
        outErr.hdulist[0].header = errData.headers[0]
        for i in range(nummod):
            outErr.hdulist[i + 1].header = header_list_err[i]

    return outData, outErr
    



    [docs]def average_beams(fitsData, errData=None, preserve_modulator=False):
    """Average the two beams together. Real simple.

    This function is modulator state aware, although I'm not sure why.

    Parameters
    ----------
    fitsData : Data.FitsData
        Data containing two beams across N modulator states

    errData : Data.FitsData
        Uncertainty on fitsData

    preserve_modulator : bool
        If True then only beams from the same modulator state will be averaged together. Otherwise, average all beams
        together.

    Returns
    -------
    Data.FitsData
        The average of both beams. Depending on the value of preserve_modulator there will either be a single HDU for
        each modulator state, or just one HDU containing the global average.

    Data.FitsData or None
        If errData is provided then this will be the uncertainty associated with the output

    """
    pivot = fitsData.headers[0]['VISP_PPP']
    nummod = fitsData.headers[0]['VISP_XXX']
    numx, numy = fitsData.data_hdus[0].data.shape
    newy = min(pivot, numy - pivot)
    hdulist = [fitsData.hdulist[0]]

    if errData is not None:
        hdulist_err = [errData.hdulist[0]]

    if preserve_modulator:
        for i in range(nummod):

            if len(fitsData.find_data_keyword_idx('VISP_YYY', i)) > 1:
                raise Exception('Only single-HDU modulator states are supported. Modulator state {} has {} HDUs'. \
                                format(i, len(fitsData.find_data_keyword_idx('VISP_YYY', i))))

            beam1 = fitsData.data_hdus[i].data[:, :newy]
            beam2 = fitsData.data_hdus[i].data[:, pivot:pivot + newy]
            avg = (beam1 + beam2) * 0.5
            hdulist.append(pyfits.ImageHDU(avg, fitsData.data_hdus[i].header))

            if errData is not None:
                beam1_err = errData.data_hdus[i].data[:, :newy]
                beam2_err = errData.data_hdus[i].data[:, pivot:pivot + newy]
                avg_err = np.sqrt(beam1_err**2 + beam2_err**2) * 0.5
                hdulist_err.append(pyfits.ImageHDU(avg_err, errData.data_hdus[i].header))

    else:
        beam_sum = np.zeros((numx, newy), dtype=np.float32)
        for i in range(nummod):
            beam_sum += fitsData.data_hdus[i].data[:, :newy]
            beam_sum += fitsData.data_hdus[i].data[:, pivot:pivot + newy]

            if errData is not None:
                beam_sum_err = np.zeros(beam_sum.shape)
                beam_sum_err += errData.data_hdus[i].data[:, :newy]**2
                beam_sum_err += errData.data_hdus[i].data[:, pivot:pivot + newy]**2

        hdulist.append(pyfits.ImageHDU(beam_sum / (nummod * 2)))
        if errData is not None:
            hdulist_err.append(pyfits.ImageHDU(np.sqrt(beam_sum_err) / (nummod * 2)))

    avgData = Data.FitsData()
    avgData.hdulist = hdulist
    avgData.origin = 'average_beams'
    avgErr = None

    if errData is not None:
        avgErr = Data.FitsData()
        avgErr.hdulist = hdulist_err
        avgErr.origin = 'average_beams'

    return avgData, avgErr
    


    [docs]def get_demodulation_matrix(beam1Data, beam2Data):
    """Read demodulation matrices from disk and prep them for polarimetric_demodulation()

    At some point this function will help the machinery that doesn't up-sample demodulation matrices with
    shape[:2] = (1, 1), but for now it basically just passes the data along.

    Parameters
    ----------
    beam1Data : Data.DemodulationData
        The demodulation matrix for ViSP beam 1. Should have a single data HDU with shape (x, y, 4, M) where (x, y) is
        the shape of the binned ViSP beam 1 frame passed to PA&C and M is the number of unique modulator states.

    beam2Data : Data.DemodulationData
        The demodulation matrix for ViSP beam 2. Should have a single data HDU with shape (x, y, 4, M) where (x, y) is
        the shape of the binned ViSP beam 2 frame passed to PA&C and M is the number of unique modulator states.

    Returns
    -------
    Data.DemodulationData
        Object containing array of shape (X, Y, 4, M) where X and Y are the dimensions of the original ViSP frame.
    """
    print('{}: extracting demoulation matrices'.format(tag()))
    numx1, numy1, _, numm1 = beam1Data.data.shape
    numx2, numy2, _, numm2 = beam2Data.data.shape

    if numx1 != numx2 or numy1 != numy2:
        raise ValueError('The demodulation matrices have different shaped beams')
    if numm1 != numm2:
        raise ValueError("The demodulation matrices don't have the same number of modulator states")

    return PolarimetricCalibration.upsample_demodulation(beam1Data, beam2Data)

    [docs]def polarimetric_demodulation(fitsData, demod, errData=None, demod_err=None):
    """Matrix multiply the input data by a polarization demodulation matrix.

    This is a general-purpose function that requires good understanding of the input parameters. The important thing
    is that the demodulation matrix's last dimension is equal to the number of data HDU's in the input data.

    Parameters
    ----------
    fitsData : Data.FitsData
        Input data to demodulate. For the instrumental demodulation the number of data HDUs should correspond to the
        number of unique modulation states. For the telescope polarimetric calibration, there should be 4 data HDUs,
        corresponding to the 4 Stokes parameters.

    demod : numpy.ndarray
        The demodulation matrix. There are a few different shapes it can have:

            1. (x, y, 4, M), where (x, y) is the shape of a single input frame and M is the number of modulation
            states. In this case, a separate demodulation can be provided for each individual pixel. Total control!

            2. (4, M). This assumes a constant demodulation matrix across the ViSP chip

            3. (4, 4). This form is used for removing the polarizing effect of the telescope optics and requires the
            input data to already be in "Stokes" form.

        Note that in all cases the length of the second to last dimension is 4 (corresponding to the Stokes vector)
        and the length of the last dimension is equal to the number of data HDUs in the input data.

    errData : Data.FitsData or None
        The uncertainty on the input data

    demod_err : numpy.ndarray
        Uncertainty on the input demodulation matrix

    Returns
    -------
    Data.FitsData
        Demodulated/corrected polarization data. Regardless of the inputs, the output will have 4 data HDUs,
        corresponding to the 4 Stokes parameters.

    Data.FitsData
        If errData is provided, the uncertainty associated with the output data.

    """
    data = fitsData.get_stacked_data()

    output = np.sum(demod * data[:, :, None, :], axis=3)

    outData = Data.FitsData()
    outData.init_with_ndarray(output)
    outData.hdulist[0].header = fitsData.hdulist[0].header
    outData.origin = 'polarimetric_demodulation'

    for i in range(1, len(outData.hdulist)):
        for c in fitsData.hdulist[0].header.cards:
            if c[0] not in ['SIMPLE','BITPIX','NAXIS','NAXIS1','NAXIS2','EXTEND','END']:
                outData.update_header(c[0], c[1:], i, True)

    outErr = None
    if errData is not None:
        # if demod_err is None:
        #     raise Exception('Demodulation matrix uncertainty estimate not provided')
        data_err = errData.get_stacked_data()

        output_err = np.sum(demod * data_err[:, :, None, :], axis=3)
        # output_err = np.sqrt(np.sum(demod**2 * data_err[:, :, None, :]**2, axis=3) +
        #                      np.sum(demod_err**2 * data[:, :, None, :]**2, axis=3))

        outErr = Data.FitsData()
        outErr.init_with_ndarray(output_err)
        outErr.hdulist[0].header = errData.hdulist[0].header
        errData.origin = 'polarimetric_demodulation'

        for i in range(1, len(outErr.hdulist)):
            for c in errData.hdulist[0].header.cards:
                if c[0] not in ['SIMPLE','BITPIX','NAXIS','NAXIS1','NAXIS2','EXTEND','END']:
                    outErr.update_header(c[0], c[1:], i, True)

    return outData, outErr
    



    [docs]def average_polarimetric_beams(fitsData, errData=None):
    """Average beams over each stokes parameter

    Basically the same as average_beams, but just different enough, instead of looking for modulator state (VISP_YYY)
    it looks for stokes parameter (VISP_STK)

    Parameters
    ----------
    fitsData : Data.FitsData
        Data containing two beams across 4 stokes parameters

    errData : Data.FitsData
        Uncertainty on fitsData

    Returns
    -------
    Data.FitsData
        The same as the input data, but where each stokes parameter contains the average of both beams instead of
        both beams separately

    Data.FitsData or None
        If errData is provided then this will be the uncertainty associated with the output
    """

    pivot = fitsData.headers[0]['VISP_PPP']
    numx, numy = fitsData.data_hdus[0].data.shape
    newy = min(pivot, numy - pivot)
    hdulist = [fitsData.hdulist[0]]

    if errData is not None:
        hdulist_err = [errData.hdulist[0]]

    for i in range(4):

        beam1 = fitsData.data_hdus[i].data[:, :newy]
        beam2 = fitsData.data_hdus[i].data[:, pivot:pivot + newy]
        avg = (beam1 + beam2) * 0.5
        hdulist.append(pyfits.ImageHDU(avg, fitsData.data_hdus[i].header))

        if errData is not None:
            beam1_err = errData.data_hdus[i].data[:, :newy]
            beam2_err = errData.data_hdus[i].data[:, pivot:pivot + newy]
            avg_err = np.sqrt(beam1_err**2 + beam2_err**2) * 0.5
            hdulist_err.append(pyfits.ImageHDU(avg_err, errData.data_hdus[i].header))

    avgData = Data.FitsData()
    avgData.hdulist = hdulist
    avgData.origin = 'average_polarimetric_beams'
    avgErr = None

    if errData is not None:
        avgErr = Data.FitsData()
        avgErr.hdulist = hdulist_err
        avgErr.origin = 'average_polarimetric_beams'

    return avgData, avgErr
    


    [docs]def setup_telescope_model(fitsData, database):
    """Initialize a TelescopeModel object and load parameters that correspond to when science data were taken

    Parameters
    ----------
    fitsData : Data.FitsData
        Science data object. Used to get wavelength and date

    database : str
        Location of telescope parameter database

    Returns
    -------
    numpy.ndarray
        (4, 4) array containing the DKIST inverse telescope model for the time and wavelength specified
    """
    wavelength = fitsData.headers[0]['WAVELNTH']
    try:
        obstime = Time(fitsData.headers[0]['DATE-OBS'], format='fits')
    except ValueError:
        obstime = Time(fitsData.headers[0]['DATE-OBS'], format='iso')

    TM = TelescopeModel.TelescopeModel(fitsData.headers[0]['TAZIMUTH'],
                                       fitsData.headers[0]['TELEVATN'],
                                       fitsData.headers[0]['TTBLANGL'])

    TM.load_from_database(database, obstime.mjd, wavelength)

    return TM.generate_inverse_telescope_model()
