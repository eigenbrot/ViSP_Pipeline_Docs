    [docs]def main(config_file):
    """The main, runner script for the entire ViSP PolCal Pipeline.

    This function is primarily used for testing or reducing stand-along CS runs (e.g., to determine Telescope parameters
    during IT&C). Day-to-day science Polarimetric Calibration is already handled by the `ViSPPipeline` module.

    The role of this function is to read in the configuration file and make calls to the appropriate pipeline modules.
    Unlike calling the modules directly, this function forces all calibration products to exist, but it will allow
    for missing error frames.
    """
    t1 = time.time()
    config = ConfigParser()
    config.read(config_file)
    print('{}: Loaded configuration file {}'.format(tag(), config_file))

    # Check to make sure raw science exists and is a directory
    if not os.path.exists(config['Main']['raw_pol_dir']):
        print('{}: Calibration Sequence data {} does not exist!'.format(tag(), config['Main']['raw_pol_dir']))
        return 1
    if not os.path.isdir(config['Main']['raw_pol_dir']):
        print('{}: raw_pol_dir must be a directory'.format(tag()))
        return 1

    # Check for necessary calibration data and try to generate them if needed
    pol_kwargs = ViSPPipeline.generate_intermediate_objects(config)

    # Now construct the ScienceCalibration call.
    print('{}: Running science data'.format(tag()))
    pol_kwargs['numpix_x'] = int(config['Main']['num_bins_x'])
    pol_kwargs['numpix_y'] = int(config['Main']['num_bins_y'])

    start_polcal_multiproc(config, pol_kwargs)

    print('{}: All done in {} s'.format(tag(), time.time() - t1))

    return
    


    [docs]def start_polcal_multiproc(config, pol_kwargs):
    """Manage a set of PolCal reduction processes using multiprocessing.

    Each step in the Calibration Sequence (i.e., separate file) is assigned its own process and then the processes are
    started in such a way that there are never more than the user-defined maximum running at a single time.

    Parameters
    ----------
    config : configparser.ConfigParser
        The object containing user-defined reduction options

    pol_kwargs : dict
        The Polarimetric Calibration keyword arguments that contain the intermediate calibration objects and any other
        reduction options

    Returns
    -------
    None
    """

    numthread = int(config['Main']['threads'])
    print('{}: Running PolarimetricCalibration on all files with {} threads'.format(tag(), numthread))
    num_steps = find_number_of_CS_steps(config['Main']['raw_pol_dir'])
    print('{}: Found {} calibration sequence steps'.format(tag(), num_steps))

    # This loop sets up a list of mp processes. They have not been started.
    jobs = []
    pol_kwargs['dark_lock'] = mp.Lock()
    pol_kwargs['gain_lock'] = mp.Lock()
    pol_kwargs['geo_lock'] = mp.Lock()
    for j in range(num_steps):
        outpre = config['Main']['output_dir']
        jobs.append(mp.Process(target=PolarimetricCalibration.main,
                               args=[config['Main']['raw_pol_dir'], j, outpre],
                               kwargs=pol_kwargs,
                               name='step_{:02}'.format(j)))

    # Now, start jobs and keep starting jobs until the max number of threads is reached
    #  Make a check every second to see if any jobs have finished and thus created room for new ones to start.
    i = 0
    while i < len(jobs):
        while [x.is_alive() for x in jobs].count(True) < numthread:
            jobs[i].start()
            i += 1
            if i >= len(jobs):
                break
        time.sleep(1)

    # Finally, make sure we wait for all jobs to complete before moving on.
    for p in jobs:
        p.join()

    return
    


    [docs]def find_number_of_CS_steps(dirname):
    """Discover the number of Calibration Sequence steps present in a SoCC directory

    Parameters
    ----------
    dirname : str
        Path to a directory containing a single SoCC

    Returns
    -------
    int
        The number of Calibration Sequence steps in the SoCC
    """
    step_list = []
    file_list = glob('{}/*.fits'.format(dirname))

    for f in file_list:
        h = pyfits.open(f)
        try:
            if h[0].header['DKIST004'] == 'PolCal':
                step_list.append(h[0].header['PAC__999'])
        except KeyError:
            pass

    unique = np.unique(step_list)

    if not np.all(np.sort(unique) == np.arange(unique.size)):
        raise ValueError('SoCC appears to be missing some CS steps')

    return unique.size
    


    [docs]def gen_blank_config(filename):
    """Create a blank configuration file.

    This is useful as a way to see all of the available options.

    Parameters
    ----------
    filename : str
        The location to save the blank file. Existing files will be overwritten.

    Returns
    -------
    None
    """
    config = ConfigParser(allow_no_value=True)
    config['Main'] = {'raw_pol_dir': '',
                      'output_dir': '',
                      'dark_cal': '',
                      'lamp_gain_cal': '',
                      'solar_gain_cal': '',
                      'geometric_cal': '',
                      'num_bins_x': '',
                      'num_bins_y': '',
                       'threads': '',
                      '': None,
                      "; The raw files point to the location of the raw ViSP frames": None}
    config['DarkCalibration'] = {'raw_dark_dir': ''}
    config['LampGainCalibration'] = {'raw_lamp_gain_dir': ''}
    config['SolarGainCalibration'] = {'raw_solar_gain_dir': ''}

    with open(filename, 'w') as f:
        config.write(f)

    return
    

def command_line():
    import argparse
    import sys

    parser = argparse.ArgumentParser(description='Run the ViSP PolCal Pipeline using the parameters specified in a config file. Config files follow the standard INI format')
    parser.add_argument('config_file', help="INI-style file containing pipeline options. Default is 'config.ini'.",
                        nargs='?', default='config.ini')
    parser.add_argument('-b', '--blank', action='store_true', help='Generate a blank configuration file')
    args = parser.parse_args()

    if args.blank:
        gen_blank_config(args.config_file)
        sys.exit(0)

    if os.path.exists(args.config_file):
        sys.exit(main(args.config_file))
    elif len(sys.argv) < 2:
        parser.print_usage()
        sys.exit(0)
    else:
        print('Could not load config file {}. Aborting.'.format(args.config_file))
        sys.exit(1)
