    [docs]def average_FitsData(
    fitsData: Data.FitsData,
    errData: Optional[Data.FitsData] = None,
    preserve_modulator: bool = True,
    delete_source: bool = False,
) -> Tuple[Data.FitsData, Optional[Data.FitsData]]:
    """Average a Data.FitsData object

    In the most general case this will simply collapse along the last axis of the stacked data (i.e., collapse all
    hdus). However, there is also the option to preserve ViSP modulator states, in which case each state of the
    modulation cycle is collapsed separately.

    In the first case the header information is taken from the first data hdu, and in the second case the header for
    each modulation state is taken from the first data hdu at that position.

    If an uncertainty data set is provided then it will be propagated through the averaging process (i.e.,
    a quadrature sum)

    Parameters
    ----------
    fitsData : Data.FitsData
        Should probably actually have some data in it

    errData: Data.FitsData
        An optional data object containing pixel uncertainty estimates. Should be exactly the same size as fitsData

    preserve_modulator : bool
        If True then keep the separate modulation states separate during the average

    Returns
    -------
    Data.FitsData
        The averaged data

    Data.FitsData (optional)
        If errData was specified then an uncertainty estimate of the average is also returned

    """
    numx, numy = fitsData.data[0].shape
    hdulist = [pyfits.PrimaryHDU(header=fitsData.headers[0])]
    if errData is not None:
        std_hdulist = [pyfits.PrimaryHDU(header=errData.headers[0])]

    if preserve_modulator:
        num_mod = fitsData.headers[0][S122["NumberOfModulatorStates"]]
        for i in range(1, num_mod + 1):  # B/c ViSP starts indexing mod state at 1
            output = np.zeros((numx, numy), dtype=np.float32)
            pos_idx = fitsData.find_data_keyword_idx(S122["CurrentState"], i)
            if len(pos_idx) > 0:
                header = fitsData.data_hdus[pos_idx[0]].header
                print("{}: Averaging modulator state {}".format(tag(), i))
                for j in pos_idx:
                    print("{}:  using HDU {}".format(tag(), j))
                    output += fitsData.data[j]
                    if delete_source:
                        del fitsData.data[j]
                hdulist.append(pyfits.ImageHDU(output / len(pos_idx), header))

            if errData is not None:
                std = np.zeros((numx, numy), dtype=np.float32)
                std_pos_idx = errData.find_data_keyword_idx(S122["CurrentState"], i)
                if len(std_pos_idx) > 0:
                    std_header = errData.data_hdus[std_pos_idx[0]].header
                    print("{}: Averaging uncertainty in modulator state {}".format(tag(), i))
                    for j in std_pos_idx:
                        print("{}:  using err HDU {}".format(tag(), j))
                        std += errData.data[j] ** 2
                    std_hdulist.append(pyfits.ImageHDU(np.sqrt(std) / len(std_pos_idx), std_header))

    else:
        print("{}: Averaging all HDUs".format(tag()))
        output = np.zeros((numx, numy), dtype=np.float32)
        for d in fitsData.data:
            output += d
        hdulist.append(pyfits.ImageHDU(output / len(fitsData.data), fitsData.headers[1]))

        if errData is not None:
            print("{}: Averaging uncertainty from all HDUs".format(tag()))
            std = np.zeros((numx, numy), dtype=np.float32)
            for d in errData.data:
                std += d ** 2
            std_hdulist.append(
                pyfits.ImageHDU(np.sqrt(std) / len(errData.data), errData.headers[1])
            )

    newData = Data.FitsData()
    newData.hdulist = hdulist
    newData.origin = "average_FitsData"

    if errData is not None:
        new_errData = Data.FitsData()
        new_errData.hdulist = std_hdulist
        new_errData.origin = "average_FitsData Uncertainty"
        return newData, new_errData

    return newData, None
    



    [docs]def extract_map(
    fitsData: Data.FitsData, x_start: int, num_x_avg: int, errData: Optional[Data.FitsData] = None
) -> Tuple[Data.FitsData, Optional[Data.FitsData]]:
    """Take a series of N (x,y) 2D images (spectra) and extract an (y,N) map by averaging along the spectral dimension.
    The user specifies the spectral channels over which to average each image.

    If an uncertainty data set is provided the pixel uncertainty will be correctly propagated through the extraction
    process.

    Parameters
    ----------
    fitsData : Data.FitsData
        The data to be extracted. Each HDU in fitsData.data_hdus will become a new row in the output image.

    x_start : int
        The y (spectral) index that defines the start of the region to average

    num_x_avg : int
        The number of spectral channels to be used in the average

    errData : Data.FitsData
        Per-pixel uncertainty estimates corresponding to the data in fitsData

    Returns
    -------
    Data.FitsData
        If N = len(fitsData.data_hdus) and each HDU's data has shape (x,y) then this object will have a single data
        HDU whoes data has shape (x,N). The Primary HDU's header is copied from the input data, and the data HDU will
        have a blank header.

    Data.FitsData
        If errData is provided then a second FitsData object is returned that contains the propagated pixel
        uncertainties. The Primary HDU's header is copied from errData.

    """
    numx, numy = fitsData.data[0].shape
    data_head = fitsData.data_hdus[0].header
    output = np.zeros((numy, len(fitsData.data)), dtype=np.float32)

    if errData is not None:
        outerr = np.zeros((numy, len(errData.data)), dtype=np.float32)
        data_err_head = errData.data_hdus[0].header

    for i in range(len(fitsData.data)):
        print(
            "{}: HDU {}: Averaging slice: [{}:{}, :]".format(tag(), i, x_start, x_start + num_x_avg)
        )
        output[:, i] = np.mean(fitsData.data[i][x_start : x_start + num_x_avg, :], axis=0)
        if errData is not None:
            print(
                "{}: Err HDU {}: Averaging slice: [{}:{}, :]".format(
                    tag(), i, x_start, x_start + num_x_avg
                )
            )
            outerr[:, i] = (
                np.sqrt(np.sum(errData.data[i][x_start : x_start + num_x_avg, :] ** 2, axis=0))
                / num_x_avg
            )

    mapData = Data.FitsData(ignore_shape_check=True)
    mapData.init_with_ndarray(output)
    mapData.hdulist[1].header = data_head

    if errData is not None:
        mapDataerr = Data.FitsData()
        mapDataerr.init_with_ndarray(outerr)
        mapDataerr.hdulist[1].header = data_err_head

        return mapData, mapDataerr

    return mapData, None
    



    [docs]def dark_correction(
    fitsData: Data.FitsData,
    dark_calibration: Data.FitsData,
    errData: Optional[Data.FitsData] = None,
    dark_calibration_err: Optional[Data.FitsData] = None,
) -> None:
    """Subtract dark data from every HDU (probably a modulation state) in the input data.

    If uncertainty estimates are provided then the uncertainty on the input data will be updated to be the quadrature
    sum of itself and the provided dark uncertainty.

    Parameters
    ----------
    fitsData : Data.FitsData
        Data object to perform dark subtraction on

    dark_calibration : Data.FitsData
        Object containing a processed dark calibration. If, for some reason, there are more than 1 data HDU,
        only the first data HDU is used.

    errData : Data.FitsData
        The pixel uncertainty estimates of the input data

    dark_calibration_err : Data.FitsData
        The pixel uncertainty estimates of the dark calibration data

    Returns
    -------
    None
        The data (and uncertainty, if provided) are updated in place

    """
    print("{}: Performing dark correction on {} with {}".format(tag(), fitsData, dark_calibration))
    dark = dark_calibration.data[0]

    if errData is not None:
        if dark_calibration_err is not None:
            print(
                "{}:  and using uncertainty estimates {} and {}".format(
                    tag(), errData, dark_calibration_err
                )
            )
            err = dark_calibration_err.data[0]
        else:
            raise Exception(
                "{}: ERROR must have uncertainty estimates for both data and dark frames".format(
                    tag()
                )
            )

    for i in range(len(fitsData.data)):
        print("{}: subtracting from HDU {}".format(tag(), i))
        fitsData.data[i] = fitsData.data[i] - dark
        if errData is not None:
            print("{}:  and err".format(tag()))
            errData.data[i] = np.sqrt(errData.data[i] ** 2 + err ** 2)

    return
    



    [docs]def gain_correction(
    fitsData: Data.FitsData,
    gain_calibration: Data.FitsData,
    errData: Optional[Data.FitsData] = None,
    gain_calibration_err: Optional[Data.FitsData] = None,
) -> None:
    """Divide gain data from from the input data. Modulation states are preserved.

    If uncertainty estimates are provided then the uncertainty on the input data will be correctly updated to include
    the uncertainty on the gain calibration.

    Parameters
    ----------
    fitsData : Data.FitsData
        Data object to perform gain division on

    gain_calibration : Data.FitsData
        Object containing a processed gain calibration. If, for some reason, there are more than 1 data HDU per
        modulation state only the first data HDU is used.

    errData : Data.FitsData
        The pixel uncertainty estimates of the input data

    gain_calibration_err : Data.FitsData
        The pixel uncertainty estimates of the gain calibration data

    Returns
    -------
    None
        The data (and uncertainty, if provided) are updated in place

    """
    print("{}: Performing gain correction on {} with {}".format(tag(), fitsData, gain_calibration))
    nummod = fitsData.headers[0][S122["NumberOfModulatorStates"]]
    if errData is not None:
        if gain_calibration_err is not None:
            print(
                "{}:  and using uncertainty estimates {} and {}".format(
                    tag(), errData, gain_calibration_err
                )
            )
        else:
            raise Exception(
                "{}: ERROR must have uncertainty estimates for both data and gain frames".format(
                    tag()
                )
            )

    for i in range(1, nummod + 1):
        didx = fitsData.find_data_keyword_idx(S122["CurrentState"], i)
        gidx = gain_calibration.find_data_keyword_idx(S122["CurrentState"], i)
        if gidx.size > 1:
            print(
                "{}: WARNING: Gain object has mulitple HDUs for modulator state {}. This is weird.".format(
                    tag(), i
                )
            )
        if gidx.size == 0:
            raise Exception(
                "{}: ERROR: Could not find gain data for data modulator state {}.".format(tag(), i)
            )
        print("{}: modulator state {}:".format(tag(), i))

        gain = gain_calibration.data[gidx[0]]
        if errData is not None:
            err = gain_calibration_err.data[gidx[0]]
        for j in didx:
            # We do the uncertainty first because the data are corrected in place and we don't want to use corrected
            # data in the uncertainty propagation
            if errData is not None:
                print("{}: propagating err in HDU {}".format(tag(), i))
                errData.data[j] = np.sqrt(
                    (errData.data[j] / gain) ** 2 + (fitsData.data[j] * err / gain ** 2) ** 2
                )

            print("{}:  dividing from HDU {}".format(tag(), j))
            fitsData.data[j] = fitsData.data[j] / gain

    return
    



    [docs]def generate_noise_frame(fitsData: Data.FitsData) -> Data.FitsData:
    """Create a pixel-uncertainty estimate based on known detector properties

    The Poisson (i.e., counting) noise associated with each pixel is (in electrons):

      P = sqrt( DN / N / g )

    where DN is the raw pixel count (in ADU), N is the number of accumulations per FPA, and g is the detector gain (in
    ADU per electron).

    The total noise estimate is then just the quadrature sum of P and the detector read noise.

    Parameters
    ----------
    fitsData : Data.FitsData
        The raw data to use to generate a noise estimate

    Returns
    -------
    Data.FitsData
        The pixel uncertainties

    """
    phdu = pyfits.PrimaryHDU(header=fitsData.headers[0])
    phdu.header["OBJECT"] = "{}_noise".format(phdu.header["OBJECT"])

    hdulist = [phdu]

    for h in fitsData.data_hdus:

        # THIS WILL CHANGE WHEN WE FIGURE OUT WHERE THE CAMERA NOISE CHARACTERISTICS WILL BE KEPT
        gain = 1.0
        read_noise = 0.0
        num_frames = h.header[S122["NumRawFramesInFPA"]]

        poissonsq = h.data / (num_frames * gain)
        total_noise = np.sqrt(poissonsq + read_noise ** 2) * gain

        ihdu = pyfits.ImageHDU(total_noise.astype(np.float32), header=h.header)
        ihdu.header["OBJECT"] = "{}_noise".format(ihdu.header["OBJECT"])
        hdulist.append(ihdu)

    errData = Data.FitsData()
    errData.hdulist = hdulist
    errData.origin = "uncertainty on {}".format(fitsData)

    return errData
